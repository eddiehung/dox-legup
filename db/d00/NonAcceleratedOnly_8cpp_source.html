58 // copied from SwOnly.cpp
59 // this functions add all global variables used in HW
60 // to llvm.used intrinsic global variable, which prohibits the compiler
61 // from optimizing the global variable away
62 // this is need since the SW IR needs to keep the global variable, so that
63 // its address can be parsed later in the Verilog backend
64 void NonAcceleratedOnly::preserveGlobalVariablesUsedInHW(Module &M) {
65 
66  SmallPtrSet<GlobalValue *, 8> GVUsed;
67  // get the llvm.used global variable if it exists already
68  GlobalVariable *LLVMUsed = M.getGlobalVariable("llvm.used");
69  // get all global variables stored in llvm.used
70  findUsedValues(LLVMUsed, GVUsed);
71  // delete the original llvm.used variable
72  if (LLVMUsed)
73  LLVMUsed->eraseFromParent();
74 
75  // create a new vector
76  // to transfer all global variables which was stored in
77  // the existing llvm.used variable
78  std::vector<Constant *> GVUsedNew;
79  llvm::Type *i8PTy = llvm::Type::getInt8PtrTy(getGlobalContext());
80  if (!GVUsed.empty()) {
81  for (auto *GV : GVUsed) {
82  Constant *c = ConstantExpr::getBitCast(GV, i8PTy);
83  GVUsedNew.push_back(c);
84  }
85  }
86 
87  // iterate through all global variables
88  // store into GVUsedNew vector, if the global variable is used in
89  // the HW functions
90  for (Module::global_iterator I = M.global_begin(), E = M.global_end();
91  I != E; ++I) {
92  // bitcast to the format in llvm.used
93  Constant *c = ConstantExpr::getBitCast(I, i8PTy);
94 
95  // add to the vector if it doesn't exists already
96  if (find(GVUsedNew.begin(), GVUsedNew.end(), c) == GVUsedNew.end())
97  GVUsedNew.push_back(c);
98 
99  }
100 
101  // create the llvm.used global variable
102  // which contains the bistcast of all global variables
103  // used by the HW functions
104  llvm::ArrayType *ATy = llvm::ArrayType::get(i8PTy, GVUsedNew.size());
105  LLVMUsed = new llvm::GlobalVariable(
106  M, ATy, false, llvm::GlobalValue::AppendingLinkage,
107  llvm::ConstantArray::get(ATy, GVUsedNew), "llvm.used");
108  // set the correct section, as required by llvm
109  LLVMUsed->setSection("llvm.metadata");
110 }
