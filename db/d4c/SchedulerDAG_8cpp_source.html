31 // Add dependencies for all the operands of iNode
32 // ie. %3 = add %1, %2
33 // %3 is dependent on %1 and %2
34 // %1 is used by %3
35 // %2 is used by %3
36 void SchedulerDAG::regDataDeps(InstructionNode *iNode) {
37  Instruction *inst = iNode->getInst();
38  // these instructions are not scheduled
39  if (isa<AllocaInst>(inst) || isa<PHINode>(inst))
40  return;
41  for (User::op_iterator i = inst->op_begin(), e = inst->op_end(); i != e;
42  ++i) {
43  // we only care about operands that are created by other instructions
44  Instruction *dep = dyn_cast<Instruction>(*i);
45  // also ignore if the dependency is an alloca
46  if (!dep || isa<AllocaInst>(dep))
47  continue;
48 
49  // ignore operands from other basic blocks, these are
50  // guaranteed to be in another state
51  if (dep->getParent() != inst->getParent())
52  continue;
53 
54  InstructionNode *depNode = getInstructionNode(dep);
55  iNode->addDepInst(depNode);
56  depNode->addUseInst(iNode);
57  }
58 }

60 // returns true if there is a dependency from I1 -> I2
61 // based on alias analysis
62 bool SchedulerDAG::hasDependencyAA(Instruction *I1, Instruction *I2) {
63  AliasAnalysis::Location Loc1, Loc2;
64 
65  if (isa<CallInst>(I1)) {
66  // assume that any loads/stores after a call must indeed execute
67  // AFTER the call
68  return true;
69  }
70 
71  // bool store = false;
72  if (LoadInst *lInst = dyn_cast<LoadInst>(I1)) {
73  Loc1 = AliasA->getLocation(lInst);
74  } else if (StoreInst *sInst = dyn_cast<StoreInst>(I1)) {
75  Loc1 = AliasA->getLocation(sInst);
76  // store = true;
77  } else {
78  assert(0 && "Unexpected input");
79  }
80 
81  if (isa<StoreInst>(I1) && isa<LoadInst>(I2)) {
82  // RAW dependency:
83  // I1 is a store and I2 is a load from potentially the same address
84  LoadInst *lInst = dyn_cast<LoadInst>(I2);
85  Loc2 = AliasA->getLocation(lInst);
86  if (!AliasA->isNoAlias(Loc1, Loc2)) {
87  return true;
88  }
89 
90  } else if (isa<StoreInst>(I2)) {
91  // WAW or WAR dependency:
92  // I1 is a store OR a load and I2 is a store to
93  // potentially same address
94  StoreInst *sInst = dyn_cast<StoreInst>(I2);
95  Loc2 = AliasA->getLocation(sInst);
96  if (!AliasA->isNoAlias(Loc1, Loc2)) {
97  return true;
98  }
99  } else {
100  // RAR: no dependency
101  assert(isa<LoadInst>(I1) && isa<LoadInst>(I2));
102  }
103 
104  return false;
105 }

107 // find all memory dependencies: I1 -> I2 (where I2 is given)
108 void SchedulerDAG::memDataDeps(InstructionNode *I2Node) {
109  Instruction *I2 = I2Node->getInst();
110  BasicBlock *bb = I2->getParent();
111 
112  // loop over all candidates for I1 in the BB for dependencies I1 -> I2
113  for (BasicBlock::iterator dep = bb->begin(), ie = bb->end(); dep != ie;
114  ++dep) {
115  Instruction *I1 = dep;
116 
117  // If we reach I2 then there are no more candidates for I1
118  if (I1 == I2)
119  break;
120 
121  if (!isa<LoadInst>(I1) && !isa<StoreInst>(I1) && !isa<CallInst>(I1))
122  continue;
123 
124  if (isaDummyCall(I1))
125  continue;
126 
127  if (hasDependencyAA(I1, I2)) {
128  // errs() << "memDataDeps: I1 -> I2\n" <<
129  // "I1: " << *I1 << "\n" <<
130  // "I2: " << *I2 << "\n";
131  InstructionNode *I1Node = getInstructionNode(I1);
132  I2Node->addMemDepInst(I1Node);
133  I1Node->addMemUseInst(I2Node);
134  }
135  }

138 // find all memory dependencies: I1 -> I2 (where I2 is given)
139 void SchedulerDAG::callDataDeps(InstructionNode *I2Node) {
140  Instruction *I2 = I2Node->getInst();
141  BasicBlock *b = I2->getParent();
142 
143  if (isaDummyCall(I2) && !isaPrintCall(I2))
144  return;
145 
146  RAM *localMem = NULL;
147  if (LEGUP_CONFIG->getParameterInt("LOCAL_RAMS") &&
148  (isa<LoadInst>(I2) || isa<StoreInst>(I2))) {
149  assert(alloc);
150  localMem = alloc->getLocalRamFromInst(I2);
151  }
152 
153  // loop over all candidates for I1 in the BB for dependencies I1 -> I2
154  for (BasicBlock::iterator dep = b->begin(), ie = b->end(); dep != ie;
155  ++dep) {
156  Instruction *I1 = dep;
157 
158  // If we reach I2 then there are no more candidates for I1
159  if (I1 == I2)
160  break;
161 
162  if (!isa<LoadInst>(I1) && !isa<StoreInst>(I1) && !isa<CallInst>(I1))
163  continue;
164 
165  if (isaDummyCall(I1) && !isaPrintCall(I1))
166  continue;
167 
168  if (isa<LoadInst>(I1) || isa<StoreInst>(I1)) {
169  RAM *depMem = alloc->getLocalRamFromInst(I1);
170  if (localMem && depMem) {
171  // only make them dependent if they point to the same memory
172  if (localMem == depMem) {
173  if (LEGUP_CONFIG->getParameterInt("ALIAS_ANALYSIS")) {
174  if (!hasDependencyAA(I1, I2))
175  continue;
176  }
177  InstructionNode *I1Node = getInstructionNode(I1);
178  I2Node->addMemDepInst(I1Node);
179  I1Node->addMemUseInst(I2Node);
180  }
181  continue;
182  }
183  }
184 
185  InstructionNode *I1Node = getInstructionNode(I1);
186  I2Node->addMemDepInst(I1Node);
187  I1Node->addMemUseInst(I2Node);
188  }

200 // hasNoDelay - detect instructions which have no delay. For example a shift by
201 // constant will just be turned into a wire by Quartus
202 bool hasNoDelay(Instruction *instr) {
203  if (instr->isShift()) {
204  return (isa<ConstantInt>(instr->getOperand(1)));
205  }
206  if (isa<GetElementPtrInst>(instr)) {
207  if (LEGUP_CONFIG->getParameterInt("DONT_CHAIN_GET_ELEM_PTR")) {
208  return false;
209  }
210  for (unsigned i = 1; i < instr->getNumOperands(); i++) {
211  if (!isa<ConstantInt>(instr->getOperand(i))) {
212  return false;
213  }
214  }
215  return true;
216  }
217 
218  switch (instr->getOpcode()) {
219  case (Instruction::And):
220  case (Instruction::Or):
221  return (isa<ConstantInt>(instr->getOperand(1)) ||
222  getBitWidth(instr->getType()) == 1);
223  }
224  return (instr->isCast() || isa<PHINode>(instr) || isa<AllocaInst>(instr) ||
225  instr->isTerminator() || isa<LoadInst>(instr) ||
226  isa<CallInst>(instr));
227 }

229 void SchedulerDAG::updateDAGwithInst(Instruction *instr) {
230  // generate Instruction to InstructionNode mapping
231  InstructionNode *iNode = new InstructionNode(instr);
232  nodeLookup[instr] = iNode;
233 
234  // set delay
235  std::string opName = LEGUP_CONFIG->getOpNameFromInst(instr, alloc);
236  if (opName.empty() || isMem(instr)) {
237  if (isa<GetElementPtrInst>(instr)) {
238  if (LEGUP_CONFIG->getParameterInt("DONT_CHAIN_GET_ELEM_PTR")) {
239  iNode->setAtMaxDelay();
240  } else if (hasNoDelay(instr)) {
241  iNode->setDelay(0);
242  } else {
243  iNode->setAtMaxDelay();
244  }
245  } else if (hasNoDelay(instr)) {
246  iNode->setDelay(0);
247  } else {
248  // errs() << "Empty: " << *instr << "\n";
249  // assert(hasNoDelay(instr));
250  iNode->setAtMaxDelay();
251  }
252  } else {
253  Operation *Op = LEGUP_CONFIG->getOperationRef(opName);
254  assert(Op);
255  float critDelay = Op->getCritDelay();
256 
257  if (critDelay > InstructionNode::getMaxDelay()) {
258  // errs() << "Warning delay " << critDelay <<
259  // "ns is greater than clock constraint of " <<
260  // InstructionNode::getMaxDelay() << "ns for instruction: " <<
261  // *instr << "\n";
262  iNode->setAtMaxDelay();
263  } else if (isMul(instr) &&
264  LEGUP_CONFIG->getParameterInt("MULTIPLIER_NO_CHAIN")) {
265  iNode->setAtMaxDelay();
266  } else {
267  iNode->setDelay(critDelay);
268  }
269  }

325 /// setStateTransitions - determine the state transitions given a terminating
326 /// instruction
327 void setStateTransitions(State *lastState, const TerminatorInst *TI,
328  State *waitState,
329  std::map<BasicBlock *, State *> bbFirstState) {
330 
331  lastState->setTerminating(true);
332 
333  // unreachable could occur in a basic block like:
334  // bb: ; preds = %entry
335  // %2 = tail call i32 (i8*, ...)* @printf(i8* noalias getelementptr
336  // inbounds ([32 x i8]* @.str1, i32 0, i32 0)) nounwind ; <i32> [#uses=0]
337  // tail call void @exit(i32 1) noreturn nounwind
338  // unreachable
339  if (isa<UnreachableInst>(TI) || TI->getOpcode() == Instruction::Ret) {
340  lastState->setDefaultTransition(waitState);
341  return;
342  }
343 
344  lastState->setTransitionVariable(TI->getOperand(0));
345 
346  BasicBlock *Default;
347  if (const SwitchInst *SI = dyn_cast<SwitchInst>(TI)) {
348 
349  for (unsigned i = 2, e = SI->getNumOperands(); i != e; i += 2) {
350  Value *value = SI->getOperand(i);
351  assert(value);
352  BasicBlock *Succ = dyn_cast<BasicBlock>(SI->getOperand(i + 1));
353  State *state = bbFirstState[Succ];
354 
355  lastState->addTransition(state, value);
356  }
357 
358  Default = dyn_cast<BasicBlock>(SI->getDefaultDest());
359 
360  } else if (const BranchInst *B = dyn_cast<BranchInst>(TI)) {
361 
362  if (B->isConditional()) {
363  Default = dyn_cast<BasicBlock>(TI->getSuccessor(1));
364 
365  BasicBlock *Succ = dyn_cast<BasicBlock>(B->getSuccessor(0));
366  State *state = bbFirstState[Succ];
367 
368  lastState->addTransition(state);
369  } else {
370  Default = dyn_cast<BasicBlock>(TI->getSuccessor(0));
371  }
372 
373  } else {
374  llvm_unreachable(0);
375  }
376 
377  State *state = bbFirstState[Default];
378  lastState->setDefaultTransition(state);
379 }

381 /// createFSM - create a Finite State Machine object from the scheduler mapping.
382 FiniteStateMachine *SchedulerMapping::createFSM(Function *F,
383  SchedulerDAG *dag) {
384  // create FSM
385  FiniteStateMachine *fsm = new FiniteStateMachine();
386 
387  // very first state
388  State *waitState = fsm->newState();
389  waitState->setDefaultTransition(waitState);
390 
391  // first state in each basic block
392  std::map<BasicBlock *, State *> bbFirstState;
393  // this map is used for labeling the states
394  // each state name will have an index, from this map, as well as being
395  // labeled with a function name, and a basic block name.
396  std::map<BasicBlock *, unsigned> sCount;
397 
398  // create a FSM for the function where each basic block is assigned
399  // an empty state as a placeholder
400  unsigned bbNum = 0;
401  for (Function::iterator b = F->begin(), be = F->end(); b != be; ++b) {
402  if (isEmptyFirstBB(b))
403  continue;
404 
405  State *s = fsm->newState();
406  s->setBasicBlock(b);
407  bbFirstState[b] = s;
408  sCount[b] = 0;
409 
410  bbNum++;
411  }
412 
413  // setup wait state transitions
414  // setTransitionSignal(rtl->find("start"))
415  // will be done later when we have access
416  // to the RTL module
417  Function::iterator firstBB = F->begin();
418  if (BasicBlock *succ = isEmptyFirstBB(firstBB)) {
419  // first BB was empty with just an unconditional branch,
420  // so the waitstate should point to the next state, for instance:
421  // BB_0:
422  // br label %BB_2
423  // also need to set the basic block so phi's are still handled properly
424  // for the empty BB
425  waitState->setTerminating(true);
426  waitState->setBasicBlock(firstBB);
427  assert(bbFirstState.find(succ) != bbFirstState.end());
428  waitState->addTransition(bbFirstState[succ]);
429  } else {
430  assert(bbFirstState.find(firstBB) != bbFirstState.end());
431  waitState->addTransition(bbFirstState[firstBB]);
432  }
433 
434  for (Function::iterator B = F->begin(), BE = F->end(); B != BE; ++B) {
435  std::map<unsigned, State *> orderStates;
436  if (isEmptyFirstBB(B))
437  continue;
438 
439  orderStates[0] = bbFirstState[B];
440  unsigned lastState = getNumStates(B);
441 
442  // int pipelined = getMetadataInt(B->getTerminator(),
443  // "legup.pipelined");
444 
445  // errs() << "BB: " << getLabel(B) << " lastState: " << lastState <<
446  // "\n";
447  createStates(1, lastState, orderStates, fsm);
448 
449  for (BasicBlock::iterator instr = B->begin(), ie = B->end();
450  instr != ie; ++instr) {
451  Instruction *I = instr;
452  unsigned order = getState(dag->getInstructionNode(I));
453 
454  orderStates[order]->push_back(I);
455 
456  // need to ensure multi-cycle instructions finish in the basic block
457  unsigned delayState = Scheduler::getNumInstructionCycles(I);
458 
459  // Normally, loads take two cycles and the loaded values are stored
460  // in shared memory controller output registers (port A or port B).
461  //
462  // In some flows however, we want each load to be stored in a
463  // separate
464  // register (e.g. to enable multi-cycle paths). But storing each
465  // load
466  // in a unique register and keeping the register on the output of
467  // the
468  // memory controller would make loads have a latency of 3, which is
469  // not needed. Instead, when a separate register is created for each
470  // load, make the FSM "think" that loads take 1 cycle (this is done
471  // below). Then the second register is placed at the output of each
472  // load (this is done in GenerateRTL.cpp, visitLoadInst);
473  if (isa<LoadInst>(I) && LEGUP_CONFIG->duplicate_load_reg()) {
474  delayState = 1; // Instead of normal 2 for loads
475  }
476 
477  if (delayState == 0) {
478  fsm->setEndState(I, orderStates[order]);
479  continue;
480  }
481 
482  delayState += order;
483  if (delayState > lastState) {
484 
485  /*
486  // assume iterative module scheduler has already handled
487  // multi-cycle instructions
488  // can't insert a new state - assume its ready in the first
489  // state of next basic block
490  if (pipelined) {
491  // this doesn't work for the kernel
492  assert(isa<LoadInst>(I));
493  //++B;
494  //assert(B != BE);
495  //fsm->setEndState(I, bbFirstState[B]);
496  //--B;
497 
498  // all loads are assumed to be wires
499  fsm->setEndState(I, orderStates[order]);
500 
501  continue;
502  }
503  */
504 
505  createStates(lastState + 1, delayState, orderStates, fsm);
506  lastState = delayState;
507  }
508 
509  fsm->setEndState(I, orderStates[delayState]);
510  }
511 
512  setStateTransitions(orderStates[lastState], B->getTerminator(),
513  waitState, bbFirstState);
514  orderStates[lastState]->setBasicBlock(B);
515 
516  for (unsigned i = 0; i < lastState; i++) {
517  assert(orderStates.find(i) != orderStates.end());
518 
519  State *s = orderStates[i];
520  s->setBasicBlock(B);
521  s->setDefaultTransition(orderStates[i + 1]);
522  }
523  }
524 
525  FiniteStateMachine::iterator stateIter = fsm->begin();
526  for (; stateIter != fsm->end(); stateIter++) {
527  State *state = stateIter;
528  if (!state->getBasicBlock()) {
529  assert(state == waitState);
530  continue;
531  }
532 
533  sCount[state->getBasicBlock()] += 1;
534  std::string newStateName = std::string("LEGUP_F_") +
535  F->getName().str().data() + "_BB_" +
536  getLabelStripped(state->getBasicBlock());
537 
538  stripInvalidCharacters(newStateName);
539  state->setName(newStateName);
540  }
541 
542  if (fsm->begin() != fsm->end())
543  fsm->begin()->setName("LEGUP");
544 
545  return fsm;
546 }
