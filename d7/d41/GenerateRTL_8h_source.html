234  // connect 'signal' to 'driver' during state 'state'.
235  // The 'instr' argument is optional and gives an LLVM instruction to
236  // display in the RTL comments
237  // There are two cases:
238  // 1) Default: A connection in the standard datapath
239  // For example:
240  // connectSignalToDriverInState(memory_enable, ONE, state)
241  // In verilog this might look like:
242  // if ((cur_state == LEGUP_F_legup_lock_BB_1_1)) begin
243  // memory_controller_enable_a = 1'd1;
244  // 2) A connection inside the loop pipeline datapath:
245  // This happens only if the 'state' is a pipeline wait state (i.e.
246  // 'loop_pipeline_wait_loop')
247  // In Verilog this might look like:
248  // if (loop_1_ii_state == 1'd0 & loop_1_valid_bit_1) begin
249  // main_legup_memset_4 <= main_legup_memset_5;
250  // For multi-cycle operations in a loop pipeline, we may need to
251  // specify the optional 'pipelineState' argument. pipelineState
252  // disambiguates _when_ during the pipeline we want to drive the
253  // signal because the 'state' from the central FSM doesn't change
254  // during the entire loop pipeline execution. If we schedule a
255  // multicycle operation, for instance a load, we want to know whether
256  // we want to connect 'signal' when the load is scheduled to start
257  // (INPUT_READY) or when the load is scheduled to finish
258  // (OUTPUT_READY):
259  enum PIPELINE_STATE { INPUT_READY, OUTPUT_READY };
260  void
261  connectSignalToDriverInState(RTLSignal *signal, RTLSignal *driver,
262  State *state, Instruction *instr = 0,
263  PIPELINE_STATE pipelineState = INPUT_READY,
264  bool setToDriverBits = false);

273  // %12 = volatile load i32* %scevgep32, align 4, !tbaa !3,
274  // with the following properties from modulo scheduling:
275  // start_time: 1 avail_time: 2 stage: 0 II: 15 start_ii_state = 1 % 15
276  // = 1
