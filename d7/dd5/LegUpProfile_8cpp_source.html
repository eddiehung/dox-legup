9 // Adds a call to __legup_prof_init() at beginning of main()
10 // Adds calls to __legup_prof_begin() at the beginning of each function
11 // Adds calls to __legup_prof_end() at the end of each function
12 // Adds a call to __legup_prof_print() at the end of main()
13 //
14 // NOTE: the backend for ARM, MIPS, etc. may move the put some prologue/epilogue
15 // instructions before/after these calls.
16 //
17 // Eg. for ARM you may see:
18 // <called_function>:
19 // push {lr}
20 // ...
21 // call __legup_prof_begin
22 // ...
23 // call __legup_prof_end
24 // ...
25 // pop {sp}
26 //
27 //===----------------------------------------------------------------------===//

49  virtual bool runOnFunction(Function &F) {
50  Module *M = F.getParent();
51  Constant *c;
52 
53  c = M->getOrInsertFunction(
54  "__legup_prof_begin", // function name
55  Type::getVoidTy(M->getContext()), // return value (void)
56  Type::getInt8PtrTy(F.getContext()), // pass char pointer for fn name
57  (Type *)0); // end variable args with NULL
58  Function *__legup_prof_begin = cast<Function>(c);
59 
60  // skip over any allocas in the entry block
61  Function::iterator Entry = F.begin();
62  BasicBlock::iterator InsertPos = Entry->begin();
63  while (isa<AllocaInst>(InsertPos))
64  ++InsertPos;
65 
66  IRBuilder<> frontBuilder(InsertPos);
67 
68  /*
69  // if this is the main function, add additional function call to
70  // initialize profiling
71  if (F.getName() == "main") {
72  // build init function
73  c = M->getOrInsertFunction(
74  "__legup_prof_init", // function name
75  Type::getVoidTy(M->getContext()), // return value (void)
76  (Type *)0); // end variable args with NULL
77  Function *__legup_prof_init = cast<Function>(c);
78 
79  // insert instruction at beginning of function
80  frontBuilder.CreateCall(__legup_prof_init);
81  }
82  */
83 
84  Value *strPtr = frontBuilder.CreateGlobalStringPtr(F.getName(), ".str");
85  frontBuilder.CreateCall(__legup_prof_begin, strPtr);
86 
87  // attempt to find return instruction of function
88  // we want to add the profiling end() functions at the very end of
89  // each
90  // function
91  for (Function::iterator BB = F.begin(), E = F.end(); BB != E; BB++) {
92  for (BasicBlock::iterator InsertPos = BB->begin(), EE = BB->end();
93  InsertPos != EE; InsertPos++) {
94  // if it is a return instruction, we want to add function calls
95  // just before it
96  if (isa<ReturnInst>(&(*InsertPos))) {
97  IRBuilder<> backBuilder(InsertPos);
98  // profiling function call to signal end of function
99  c = M->getOrInsertFunction(
100  "__legup_prof_end", // function name
101  Type::getVoidTy(M->getContext()), // return value (void)
102  (Type *)0); // end variable args with NULL
103  Function *__legup_prof_end = cast<Function>(c);
104 
105  // insert instruction just before 'return'
106  backBuilder.CreateCall(__legup_prof_end);
107 
108  if (F.getName() != "main") {
109  break;
110  }
111 
112  c = M->getOrInsertFunction(
113  "__legup_prof_print", // function name
114  Type::getVoidTy(M->getContext()), // return value (void)
115  (Type *)0); // end variable args with NULL
116  Function *__legup_prof_print = cast<Function>(c);
117 
118  // insert instruction just before 'return'
119  backBuilder.CreateCall(__legup_prof_print);
120 
121  break;
122  }
123  }
124  }
125 
126  return true; // each function should have been modified
127  }
