37 bool HwOnlyRename::runOnModule(Module &M) {
38 
39  // If a function is not the accelerating function (the descendents),
40  // then add suffix '_accelerated' to the function names.
41  // In the case that a descendent function appears in both HW and SW side,
42  // this makes sure BB cycles can still be correctly gathered.
43  for (Module::iterator F = M.begin(), FE = M.end(); F != FE; ++F)
44  if ( !F->isIntrinsic () && !LEGUP_CONFIG->isAccelerated(*F))
45  {
46  // do not rename exit() function
47  if (F->getName() == "exit")
48  continue;
49  F->setName(F->getName().str()+"_accelerated");
50  }
51 
52  // Since this IR will be linked back with sw-only IR, global variables should be set
53  // as LinkOnceAnyLinkage
54  for (Module::global_iterator I = M.global_begin(), E = M.global_end(); I != E; ++I) {
55  if (!I->isDeclaration()) {
56  I->setLinkage(GlobalValue::LinkOnceAnyLinkage);
57  }
58  }
59 
60  return true;
61 
62 }
