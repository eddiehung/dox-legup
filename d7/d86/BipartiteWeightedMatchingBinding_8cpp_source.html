7 // LegUp uses a weighted bipartite matching heuristic to solve the
8 // binding problem. See paper: Huang et al, Data Path Allocation Based on
9 // Bipartite Weighted Matching, DAC'90
10 //
11 // The binding problem is represented using a bipartite graph with two vertex
12 // sets. The first vertex set corresponds to the operations being bound (i.e.
13 // LLVM instructions). The second vertex set corresponds to the available
14 // functional units. A weighted edge is introduced from a vertex in the first
15 // set to a vertex in the second set if the corresponding operation is a
16 // candidate to be bound to the corresponding functional unit. We set the cost
17 // (edge weight) of assigning an operation to a functional unit using the
18 // cost function described in the comments of constructWeights().
19 //
20 // Weighted bipartite matching can be solved optimally in polynomial time using
21 // the Hungarian method [Kuhn 2010]. We formulate and solve the
22 // matching problem one clock cycle at a time until the operations in all clock
23 // cycles (states) have been bound.
24 //
25 //===----------------------------------------------------------------------===//

220 // We have three goals when binding operations to shared functional units.
221 // First, we would like to balance the sizes of the multiplexers across
222 // functional units to keep circuit performance high. Multiplexers with
223 // more inputs have higher delay, so it is desirable to avoid having a
224 // functional unit with a disproportionately large multiplexer on its input.
225 // Second, we want to recognize cases where we have shared inputs between
226 // operations, letting us save a multiplexer if the operations are assigned
227 // to the same functional unit. Lastly, during binding if we can assign two
228 // operations that have non-overlapping livetime intervals to the same
229 // functional unit, we can use a single output register for both operations.
230 // In this case we save a register, without needing a multiplexer. We use
231 // the LLVM live variable analysis pass to check for the livetime intervals.
232 // To account for these goals we use the following cost function to measure the
233 // benefit of assigning operation op to function unit fu:
234 // Cost(op, f u) = existingMuxInputsFactor * existingMuxInputs(f u) +
235 // newMuxInputsFactor * newMuxInputs(op, f u) -
236 // outputRegisterSharableFactor * outputRegisterSharable(op, f u)
237 // where existingMuxInputsFactor = 1, newMuxInputsFactor = 10, and
238 // outputRegisterSharableFactor = 5 to give priority to saving new
239 // multiplexer inputs, then output registers, and finally balancing the
240 // multiplexers. Notice that sharing the output register reduces the cost,
241 // while the other factors increase it.

243 void BipartiteWeightedMatchingBinding::constructWeights(raw_ostream &out,
244  Instruction *I, int operationIdx, std::string funcUnitType, int
245  numFuncUnitsAvail, AssignmentInfo &assigned, Table &weights) {
246 
247  int existingMuxInputsFactor = 1;
248  int newMuxInputsFactor = 10;
249  // note: this is negative, a shared output register reduces the cost
250  int outputRegisterSharableFactor = -5;
251 
252 
253  for (int fu = 0; fu < numFuncUnitsAvail; fu++) {
254  int weight = 0;
255  std::string fuId = this->alloc->verilogNameFunction(this->Fp, this->Fp)
256  + "_" + funcUnitType + "_" + utostr(fu);
257 
258  // check both operands
259  for (User::op_iterator i = I->op_begin(), e =
260  I->op_end(); i != e; ++i) {
261  Instruction *operand = dyn_cast<Instruction>(*i);
262  if (!operand) continue;
263  if (assigned.existingOperands[fuId].find(operand) ==
264  assigned.existingOperands[fuId].end()) {
265  weight += newMuxInputsFactor;
266  } else {
267  std::string instStr = getValueStr(I);
268  limitString(instStr, 30);
269  out << instStr << " can share an input with another operation "
270  "already assigned to " << fuId << "\n";
271  }
272 
273  }
274 
275  bool outputRegSharable = false;
276  for (set<Instruction*>::iterator i =
277  assigned.existingInstructions[fuId].begin(), e =
278  assigned.existingInstructions[fuId].end(); i
279  != e; ++i) {
280  Instruction *shared = *i;
281 
282  // check for shared output register
283  if (assigned.IndependentInstructions[I].find(shared) !=
284  assigned.IndependentInstructions[I].end() ) {
285  //errs() << "Shared output: " << *shared << "\n";
286  outputRegSharable = true;
287  break;
288  }
289  }
290 
291  if (outputRegSharable) {
292  weight += outputRegisterSharableFactor;
293  std::string instStr = getValueStr(I);
294  limitString(instStr, 30);
295  out << instStr << " can share an output register with another "
296  "operation already assigned to " << fuId << "\n";
297  }
298 
299  weight += existingMuxInputsFactor * assigned.muxInputs[fuId];
300 
301  weights[fu][operationIdx] = weight;
302 
303  //errs() << "weight " << fu << " " << operationIdx << " = " <<
304  // weights[fu][operationIdx] << "\n";
305  }
306 }

309 void BipartiteWeightedMatchingBinding::UpdateAssignments(raw_ostream &out, int
310  numOperationsToShare, std::string funcUnitType, int numFuncUnitsAvail,
311  AssignmentInfo &assigned, Table &assignments) {
312  std::string s;
313  std::stringstream ss;
314 
315  // Note that assignments (and weights) are square matrices,
316  // numFuncUnitsAvail x numFuncUnitsAvail
317  // However, numOperationsToShare <= numFuncUnitsAvail so the iteration
318  // below is within bounds
319  assert(numOperationsToShare <= numFuncUnitsAvail);
320  for (int fu = 0; fu < numFuncUnitsAvail; fu++) {
321  for (int o = 0; o < numOperationsToShare; o++) {
322  if (assignments[fu][o]) {
323  Instruction *I = opInstr[o];
324  std::string fuId = this->alloc->verilogNameFunction(this->Fp,
325  this->Fp) + "_" + funcUnitType + "_" + utostr(fu);
326  this->setBindingInstrFU(I, fuId);
327 
328  int numOperands = 0;
329  for (User::op_iterator i = I->op_begin(), e =
330  I->op_end(); i != e; ++i) {
331  Instruction *operand = dyn_cast<Instruction>(*i);
332  numOperands++;
333  if (!operand) continue;
334  if (assigned.existingOperands[fuId].find(operand) ==
335  assigned.existingOperands[fuId].end()) {
336  assigned.existingOperands[fuId].insert(operand);
337  assigned.muxInputs[fuId]++;
338  }
339  }
340 
341  std::string instStr = getValueStr(opInstr[o]);
342  limitString(instStr, 30);
343  ss << instStr << " (idx: " << o << ") -> " << fuId <<
344  " (mux inputs: " << assigned.muxInputs[fuId] << ")\n";
345 
346  // number of operands for mem_dual_port is not 2
347  //assert(numOperands == 2);
348  assigned.existingInstructions[fuId].insert(I);
349  }
350  }
351  }
352 
353  ss.flush();
354  out << ss.str();
355 }

409 void BipartiteWeightedMatchingBinding::bindFunctUnitInState(raw_ostream &out,
410  State* state, std::string funcUnitType, int numFuncUnitsAvail,
411  AssignmentInfo &assigned) {
412  // create a numFuncUnitsAvail x numFuncUnitsAvail matrix of integer weights
413  // and assignments for solving the bipartite weighted matching problem
414  Table weights(numFuncUnitsAvail, std::vector<int>(numFuncUnitsAvail));
415  Table assignments(numFuncUnitsAvail, std::vector<int>(numFuncUnitsAvail));
416 
417  std::string tmp;
418  raw_string_ostream weights_stream(tmp);
419 
420  // loop over all operations in this state
421  int operationIdx = 0;
422  for (State::iterator instr = state->begin(), ie = state->end();
423  instr != ie; ++instr) {
424 
425  Instruction *I = *instr;
426  if (shareInstructionWithFU(I, funcUnitType)) {
427  constructWeights(weights_stream, I, operationIdx, funcUnitType,
428  numFuncUnitsAvail, assigned, weights);
429  opInstr[operationIdx] = I;
430  operationIdx++;
431  }
432  }
433 
434  // only share if there is more than one operation using
435  // this functional unit
436  int numOperationsToShare = operationIdx;
437  if (numOperationsToShare >= 1) {
438  out << "State: " << state->getName() << "\n";
439  out << "Binding functional unit type: " << funcUnitType << "\n";
440 
441  assert(numOperationsToShare <= numFuncUnitsAvail && "Invalid schedule!");
442 
443  out << "Weight matrix for operation/function unit matching:\n";
444  weights_stream.flush();
445  out << weights_stream.str();
446  printTable(out, funcUnitType, numOperationsToShare, numFuncUnitsAvail,
447  weights);
448 
449  out << "Solving Bipartite Weighted Matching (minimize weights)...\n";
450  solveBipartiteWeightedMatching(weights, assignments);
451  out << "Assignment matrix after operation/function unit matching:\n";
452  printTable(out, funcUnitType, numOperationsToShare, numFuncUnitsAvail,
453  assignments);
454 
455  out << "Checking that every operator was assigned to a functional unit...";
456  CheckAllWereAssigned(numOperationsToShare, numFuncUnitsAvail,
457  assignments);
458  out << "yes\n";
459 
460  out << "Binding operator -> functional unit assignments:\n";
461  UpdateAssignments(out, numOperationsToShare, funcUnitType,
462  numFuncUnitsAvail, assigned, assignments);
463  }
