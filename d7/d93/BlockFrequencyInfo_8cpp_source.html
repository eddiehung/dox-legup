130 bool BlockFrequencyInfo::runOnFunction(Function &F) {
131  BranchProbabilityInfo &BPI = getAnalysis<BranchProbabilityInfo>();
132  LoopInfo &LI = getAnalysis<LoopInfo>();
133  if (!BFI)
