68 bool is_inst_a_mc_path_src_or_dest(Instruction *i) {
69  // This function determines whether the given instruction acts as the source
70  // or destination of a multi-cycle path. Many instructions will simply be
71  // chained as part of the MC path, but some instructions cannot be chained
72  // and therefore will start or terminate the path ("source" or
73  // "destination").
74  // Destinations are AKA sinks or roots. These source or dest instructions
75  // will
76  // not have their output registers removed, but rather their registers will
77  // stay
78  // and act as the source/dest of the MC path. So these source/dest
79  // instructions
80  // are equivalently "registered instructions" (after all other instructions
81  // have
82  // had their registers removed to enable MC / de-pipelined paths).
83  //
84  // Note that in most cases there is no distinction between sources or
85  // destinations of MC paths, because if an instruction terminated one
86  // MC path then it is also the beginning of the next one (except e.g. for
87  // store instructions which are only dests).
88  //
89  // The instructions which are src/dsts of MC paths, i.e. whose registers
90  // persist
91  // even after de-pipelining of data paths, are:
92  //
93  // - Instructions used across BB, including PHIs (for now)
94  // - call instructions: there are registers before module input
95  // ports (so the register is "outside the function")
96  // - Return instructions
97  // - Loads (registers are either memory_controller_out_reg_a or b)
98  // - Stores do NOT have registers (no input reg to LegUp altsyncrams)
99  // but still terminate data paths
100  // - Instructions with latency > 0
101  //
102  // Handle any more types as they come up (e.g. other terminators)
103 
104  bool is_mc_src_or_dest = false;
105 
106  // Loads and stores
107  if (isMem(i)) {
108  is_mc_src_or_dest = true;
109  }
110  // Instructions with latency
111  // Note: we need to be careful with instructions that are pipelined
112  // because pipelined instructions can't be sources of MC paths (new value
113  // each cycle) unless we guarantee that there won't be new data during the
114  // multi-cycle duration. So for now I'll add instructions with latency > 0
115  // but exclude dividers and remainders, since they cannot be MC sources
116  // (although maybe their inputs can come from MC paths?)
117  /*
118  else if (Scheduler::getNumInstructionCycles(i) > 0 && !isDiv(i) &&
119  !isRem(i)) {
120  is_mc_src_or_dest = true;
121  }
122  */
123  // PHI instructions
124  else if (isa<PHINode>(*i)) {
125  is_mc_src_or_dest = true;
126  }
127  // Instructions used across BB
128  else if (has_uses_in_other_BB(i)) {
129  is_mc_src_or_dest = true;
130  }
131  // Call instructions
132  else if (isa<CallInst>(*i)) {
133  is_mc_src_or_dest = true;
134  }
135  // Ret instructions
136  else if (isa<ReturnInst>(*i)) {
137  is_mc_src_or_dest = true;
138  }
139 
140  return is_mc_src_or_dest;
141 }

225 void SDCScheduler::find_instructions_to_delay(
226  Function *F, std::map<Instruction *, int> &mc_dest_instructions_to_state)
227 // This function is used when removing registers from data paths. It iterates
228 // over all the instructions in the function to find those instructions which:
229 // (1) are destinations of MC paths, and
230 // (2) should be delayed in the second scheduling (scheduled in a later state)
231 // Destinations of MC paths include store instructions, loads (address port),
232 // and instructions which still have registers even after de-pipelining.
233 // But of these, only some should be scheduled differently to have the paths
234 // which they terminate extended, and this function finds those instructions
235 // which we should delay and maps them to their new intended state (with
236 // respect to their BB).
237 {
238  // for all BBs in F
239  for (Function::iterator b = F->begin(), be = F->end(); b != be; ++b) {
240  // Call a helper to find the instructions requiring registers
241  if (alloc->is_BB_executed_infrequently(b)) {
242  find_instructions_to_delay_in_BB(b, mc_dest_instructions_to_state);
243  }
244  }

286 void SDCScheduler::find_instructions_to_delay_in_BB(
287  BasicBlock *b, std::map<Instruction *, int> &mc_dest_instructions_to_state)
288 // Helper to find_instructions_to_delay
289 // It iterates over all the instructions in the function to find
290 // those instructions which are destinations of MC paths and should
291 // be delayed. It then maps each of these instructions to the state
292 // (within their BB) that they should be re-scheduled.
293 {
294  mc_debug() << "\n\nInfrequent Basic Block:";
295 
296  REAL *variables = new REAL[numVars];
297  get_variables(lp, variables);
298 
299  // We may also want to push the terminator (branch, ret, etc.) of
300  // this basic block. But only do that if the BB has some other
301  // multi-cycle path in it.
302  bool bb_has_multicycle_path = false;
303 
304  // for all I in BB
305  for (BasicBlock::iterator i = b->begin(), ie = b->end(); i != ie; ++i) {
306  // The updated state of i is its originally scheduled state
307  // plus whatever latency we want to add
308  unsigned idx = startVariableIndex[dag->getInstructionNode(i)];
309  unsigned current_state = (unsigned)variables[idx];
310 
311  // By default we add latency depending on LLVM_PROFILE_EXTRA_CYCLES
312  // However this does not work well for BB with multiple consecutive
313  // paths. An alternative is to use a multiplier based on the state.
314  unsigned extended_state;
315  if (LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_STRETCH_BB")) {
316  unsigned stretch_factor =
317  LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_EXTRA_CYCLES");
318  extended_state = (unsigned)((float)current_state *
319  (float)(stretch_factor + 0.5));
320  } else {
321  unsigned extra_latency =
322  LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_EXTRA_CYCLES");
323  extended_state = current_state + extra_latency;
324  }
325  // Another idea: Can also extend more if very infrequent (e.g. 1%,
326  // extend 2 cycles)
327 
328  // Determine whether we want to push this instruction to a later
329  // state, i.e. if it terminates a MC path and is not a PHI or
330  // something.
331  if (mc_push_inst_to_later_state(i)) {
332  // Push to a later cycle when we re-schedule.
333  mc_debug() << "\nMulti-cycle paths: Pushing " << *i
334  << " to state >= " << extended_state;
335  mc_dest_instructions_to_state[i] = extended_state;
336  bb_has_multicycle_path = true;
337  }
338  }
339 
340  // If this BB had any multi-cycle paths, now extend the latency of
341  // the entire BB, thereby dilating every path "exiting" this BB.
342  if (bb_has_multicycle_path) {
343  Instruction *t = b->getTerminator();
344  mc_debug() << "\nBB has terminator " << *t;
345  bool push_to_later_state = false;
346  if (isa<ReturnInst>(*t)) {
347  push_to_later_state = true;
348  } else if (isa<BranchInst>(*t)) {
349  push_to_later_state = true;
350  }
351  /*
352  else if (isa<SwitchInst>(*t)) { // Not switches for now (change if
353  needed)
354  push_to_later_state = true;
355  }
356  */
357  if (push_to_later_state) {
358  unsigned idx = startVariableIndex[dag->getInstructionNode(t)];
359  unsigned current_state = (unsigned)variables[idx];
360  unsigned extended_state;
361  if (LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_STRETCH_BB")) {
362  unsigned stretch_factor =
363  LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_EXTRA_CYCLES");
364  extended_state = (unsigned)((float)current_state *
365  (float)(stretch_factor + 0.5));
366  } else {
367  unsigned extra_latency =
368  LEGUP_CONFIG->getParameterInt("LLVM_PROFILE_EXTRA_CYCLES");
369  extended_state = current_state + extra_latency;
370  }
371 
372  mc_debug() << "\nMulti-cycle paths: Pushing Terminator " << *t
373  << " to state >= " << extended_state;
374  mc_dest_instructions_to_state[t] = extended_state;
375  }
376  }
377 
378  delete[] variables;
379 }

392 void SDCScheduler::multicycle_and_modify_schedule(Function *F)
393 // Check whether registers should be removed from paths, and if so remove the
394 // registers here. This is done by re-scheduling all instructions, with
395 // dependency constraints but without timing constraints, forcing all
396 // instructions to be chained (i.e. no pipeline registers for timing)
397 //
398 // Also, if software profiling was performed for this compilation,
399 // extend infrequent BB by a certain number of cycles by performing
400 // a second scheduling phase.
401 {
402  // MULTICYCLE_TODO: For now disable multi-cycle paths in cases where
403  // resources are pipelined (includes loop pipelining). Also only
404  // enable for the pure-HW flow
405  if (!LEGUP_CONFIG->does_flow_support_multicycle_paths()) {
406  return;
407  }
408 
409  // Extend the latency of paths in infrequently executed BB
410  bool re_schedule_to_add_latency =
411  should_scheduling_be_repeated_to_add_latency();
412  if (re_schedule_to_add_latency) {
413  // 1. Find instructions in all basic blocks whose state should be
414  // delayed in the second scheduling.
415  std::map<Instruction *, int> mc_dest_instructions_to_state;
416  find_instructions_to_delay(F, mc_dest_instructions_to_state);
417 
418  // 2. Delete the current LP and create a new one, adding
419  // only dependency constraints
420  delete_lp(lp);
421  numVars = 0;
422  numInst = 0;
423  createLPVariables(F);
424  addMulticycleConstraints(F);
425  addDependencyConstraints(F);
426 
427  // Note: For now I'll also re-add timing constraints
428  addTimingConstraints(F);
429 
430  // 3. Add LP constraints for instructions whose state will persist
431  // in the second scheduling (i.e. MC destination instructions)
432  std::map<Instruction *, int>::const_iterator inst_it =
433  mc_dest_instructions_to_state.begin();
434  for (; inst_it != mc_dest_instructions_to_state.end(); ++inst_it) {
435  // Constraints are of the form: 1.0 * VAR_INDEX >= STATE
436  REAL var_coefficient = 1.0;
437  Instruction *persistent_instruction = inst_it->first;
438  int var_index = 1 + startVariableIndex[dag->getInstructionNode(
439  persistent_instruction)];
440  REAL state = REAL(inst_it->second);
441  // Force it to the same state
442  add_constraintex(lp, 1, &var_coefficient, &var_index, GE, state);
443  }
444 
445  // 4. Schedule ASAP
446  scheduleASAP();
447 
448  // Resource constraints are added here and scheduling is done again
449  do_post_scheduling_steps(F);
450 
451  // 5. Update the states for each instruction, as well as the number
452  // of instructions in each BB in case extending path latencies
453  // increased any BB latencies.
454  depositSchedule(F);
455  }
456 
457  // Store the states of all instructions. This is used in GenerateRTL
458  // when printing multi-cycle timing constraints.
459  save_instruction_states(F);
460 }

466 // This function returns all instructions that are part of the conditional
467 // assignment of a particular signal. For example if signal S is connected
468 // to driver D when signal C==5, then this function returns the instruction
469 // corresponding with signal C (and all other such "conditional instructions").
470 //
471 // Conditions are structured as follows:
472 //
473 // RTLSignal C Constant 5
474 // \ /
475 // \ /
476 // RTLOp (EQ) <-- "S->getCondition(0)" returns this RTLOp
477 // |
478 // |
479 // Signal S (AKA "curr" in this function)
480 //
481 // Except this may be deeper (i.e. multiple levels of operations).
482 // This function returns all conditional instructions, e.g. the one associated
483 // with C in the diagram aboce.
484 // This function is similar to the recursive "printConditions" in VerilogWriter.
485 void recursive_get_all_conditional_instructions(RTLSignal *curr,
486  std::vector<Value *> &preds) {
487  // Base case -- this condition is not an RTLOp
488  if (!curr->isOp()) {
489  // This condition could be cur_state, a memory controller signal,
490  // a parameter (e.g. state #5), or an instruction. Only if it
491  // is an instruction, add it to the predecessor queue
492  Instruction *inst = curr->getInstPtr(0);
493  if (inst) {
494  preds.push_back((Value *)inst);
495  }
496  }
497  // Otherwise continue the recursion
498  else {
499  RTLOp *op = (RTLOp *)curr;
500  for (unsigned i = 0; i < op->getNumOperands(); ++i) {
501  recursive_get_all_conditional_instructions(op->getOperand(i),
502  preds);
503  }
504  }
505 }

605 // getelementptr, and we need to do a "deeper search" of the drivers.
606 // This function handles all these cases (i.e. most of the time only 1 level
607 // of operands is searched, but more levels are searched if needed).
608 bool GenerateRTL::remove_intermediate_register_helper(RTLOp *rtl_op_driver,
609  RTLSignal *src_sig_w,
610  RTLSignal *src_sig_r) {
611  // Set up BFS
612  std::queue<RTLSignal *> frontier;
613  std::set<RTLSignal *> visited;
614  frontier.push(rtl_op_driver);
615  visited.insert(rtl_op_driver);
616 
617  // Do BFS
618  while (!frontier.empty()) {
619  // Pop next
620  RTLSignal *current = frontier.front();
621  frontier.pop();
622 
623  // Check each driver
624  unsigned num_drivers = get_num_signal_drivers_helper(current);
625  // Loop over drivers of current
626  for (unsigned i = 0; i < num_drivers; ++i) {
627  RTLSignal *driver = get_signal_driver_helper(current, i);
628  if (visited.find(driver) != visited.end()) {
629  continue;
630  }
631  visited.insert(driver);
632  frontier.push(driver);
633 
634  if (driver == src_sig_w || driver == src_sig_r) {
635  // If the src is a register, connect to the wire instead
636  if (driver == src_sig_r) {
637  set_signal_operand_helper(current, i, src_sig_w);
638  }
639  // Found the correct source
640  return true;
641  }
642  }
643  }
644  return false;
645 }

707 // This function removes the register between dst and its
708 // operand (src), if a register exists.
709 void GenerateRTL::remove_intermediate_register(Instruction *src,
710  Instruction *dst) {
711  /*
712  * Note: an alternative way to remove registers would be to modify
713  * getOpNonConstant to return the wire instead of the register when
714  * initially constructing the module. However, some instructions have
715  * both register and wire uses. For example, an instruction may
716  * have uses in the same state (chain directly) and also in another BB
717  * (connected to its register).
718  *
719  * An alternative approach (this function) is to post-process the RTL
720  * module to remove connections to registers and replace them with
721  * direct connections to the wire.
722  */
723 
724  /* The LLVM DFG and RTL Data structure graphs are different.
725  For example consider the instruction %1 = add %2, %3 :
726 
727  LLVM Instructions RTL Signals
728 
729  %2 wire %3 wire
730  | |
731  %2 %3 %2 reg %3 reg
732  \ / \ /
733  \/ \ /
734  %1 \ /
735  "Add" RTLOp
736  |
737  |
738  %1 wire
739  |
740  %1 reg
741 
742  The inputs to this function are src and dst instructions, where
743  dst is "%1 = add %2, %3" and src is one of %2 or %3
744 
745  For example if src is %2 and dst is %1, this function aims to
746  modify the RTL above to directly connect the op to the wire src:
747 
748  %2 wire %3 wire
749  / | |
750  %2 reg | %3 reg
751  \ /
752  \ /
753  \ /
754  "Add" RTLOp
755  |
756  |
757  %1 wire
758  |
759  %1 reg
760 
761  Then if needed we call it again later for src = %3.
762  */
763 
764  // Get the RTL signal of dst (wire) and src (wire and reg)
765  RTLSignal *src_sig_w = rtl->findExists(verilogName(src));
766  RTLSignal *src_sig_r = rtl->findExists(verilogName(src) + "_reg");
767  RTLSignal *dst_sig_w;
768  if (isa<StoreInst>(*dst)) {
769  dst_sig_w = rtl->findExists(get_store_port_name(dst, src));
770  } else if (isa<LoadInst>(*dst)) {
771  dst_sig_w = rtl->findExists(get_load_port_name(dst));
772  } else {
773  dst_sig_w = rtl->findExists(verilogName(dst));
774  }
775 
776  if (!dst_sig_w) {
777  mc_debug() << "\n No signal, dst = " << *dst << ", "
778  << verilogName(dst) << "\n";
779  return;
780  }
781  if (!src_sig_w) {
782  mc_debug() << "\n No signal, src = " << *src << ", "
783  << verilogName(src) << "\n";
784  return;
785  }
786  if (!src_sig_r) {
787  return; // Src has no register signal, so there is nothing to remove
788  }
789  if (!src_sig_r->isReg()) {
790  return;
791  }
792 
793  mc_debug() << "\nRemoving reg from " << src_sig_w->getName() << " to "
794  << dst_sig_w->getName();
795 
796  bool found_correct_driver_signal = remove_intermediate_register_helper(
797  (RTLOp *)dst_sig_w, src_sig_w, src_sig_r);
798  if (!found_correct_driver_signal && isa<PHINode>(*dst)) {
799  found_correct_driver_signal =
800  remove_conditional_register(dst_sig_w, src_sig_w, src_sig_r);
801  assert(found_correct_driver_signal);
802  }
803 }

952 // Determine whether there is a register between the predecessor instruction
953 // and the current instruction. In normal flows this defaults to checking
954 // whether they are in different states but in flows where registers are
955 // removed from data paths we need to do additional checks.
956 bool does_pred_have_register(Instruction *pred, Instruction *current,
957  bool remove_registers_from_data_paths,
958  FiniteStateMachine *fsm) {
959 
960  bool does_pred_have_register = false;
961 
962  // If we are removing registers from compare instructions then
963  // return false right away
964  if (isa<ICmpInst>(*pred) &&
965  LEGUP_CONFIG->mc_remove_reg_from_icmp_instructions()) {
966  return false;
967  }
968 
969  // Check instructions which always have registers
970  if (instruction_must_have_register(pred)) {
971  does_pred_have_register = true;
972  }
973  // Check if the predecessor is in a different basic block
974  else if (pred->getParent() != current->getParent()) {
975  mc_debug() << "\n Other BB from " << *current;
976 
977  // Special case of PHI
978  if (!isa<PHINode>(*current)) {
979  does_pred_have_register = true;
980  }
981  }
982  // Check if the predecessor has latency (shift register)
983  // But not things like mults etc. because we de-pipeline those
984  // when mc paths are used
985  else if (is_instruction_pipelined_after_multicycling(pred)) {
986  mc_debug() << "\n Latency > 0 -> " << *pred << " has reg";
987  does_pred_have_register = true;
988  }
989  // Final case:
990  // Only for flows where we are NOT removing registers from data
991  // paths, there is also a register if pred is in a different state.
992  else if (!remove_registers_from_data_paths) {
993  if (fsm->getEndState(pred) != fsm->getEndState(current)) {
994  does_pred_have_register = true;
995  }
996  }
997  return does_pred_have_register;
998 }

1061 // Helper function, takes a vector of instructions (a path), finds all
1062 // adjacent pairs in this path, and removes the pairs from a multimap
1063 // Borrowed from http://stackoverflow.com/questions/3952476/
1064 void remove_all_pairs_in_path_from_multimap(
1065  std::vector<Instruction *> cur_path,
1066  std::multimap<Instruction *, Instruction *> &remove_reg_pairs,
1067  std::set<std::pair<Instruction *, Instruction *>> &keep_reg_pairs) {
1068  typedef std::multimap<Instruction *, Instruction *>::iterator mmap_it;
1069  std::vector<Instruction *>::const_iterator it = cur_path.begin(),
1070  end = cur_path.end();
1071  Instruction *previous = *it;
1072  ++it;
1073 
1074  for (; it != end; ++it) {
1075 
1076  // Get all of the pairs in the multimap with key = previous
1077  std::pair<mmap_it, mmap_it> iterpair =
1078  remove_reg_pairs.equal_range(*it);
1079 
1080  // Erase the pair (previous, *it)
1081  // Note that the vector is in the order root --> src
1082  mmap_it it2 = iterpair.first;
1083  for (; it2 != iterpair.second; ++it2) {
1084  if (it2->second == previous) {
1085  remove_reg_pairs.erase(it2);
1086  break;
1087  }
1088  }
1089 
1090  // Don't remove reg from this pair later
1091  keep_reg_pairs.insert(std::make_pair(*it, previous));
1092 
1093  // And update previous for next iteration
1094  previous = *it;
1095  }
1096 }

1098 // Helper function for dfs
1099 void GenerateRTL::root_dfs_visit_preds_of_current(
1100  Instruction *current, std::stack<Instruction *> &frontier,
1101  std::vector<Value *> &preds, std::vector<std::vector<Instruction *>> &paths,
1102  std::vector<Instruction *> &cur_path,
1103  std::multimap<std::vector<Instruction *>, Value *> &paths_with_arg_srcs,
1104  std::set<Instruction *> &visited,
1105  std::multimap<Instruction *, Instruction *> &remove_reg_pairs,
1106  std::set<std::pair<Instruction *, Instruction *>> &keep_reg_pairs,
1107  std::map<Instruction *, int> &num_unfinished_preds) {
1108  std::vector<Value *>::const_iterator pred_i = preds.begin();
1109  for (; pred_i != preds.end(); ++pred_i) {
1110  Instruction *pred = dyn_cast<Instruction>(*pred_i);
1111 
1112  // Ignore non-instructions and predecessors already visited
1113  if (!pred) {
1114  // Note: Usually we ignore predecessors which are not instructions.
1115  // However, if the predecessor is an input to this function, then
1116  // we can use its register as a multi-cycle source
1117  if (dyn_cast<Argument>(*pred_i)) {
1118  mc_debug() << "\n Found input arg: "
1119  << verilogName(dyn_cast<Argument>(*pred_i));
1120  paths.push_back(cur_path);
1121  paths_with_arg_srcs.insert(std::make_pair(cur_path, *pred_i));
1122  }
1123  // Could also be a constant, etc., so just continue
1124  continue;
1125  }
1126 
1127  bool revisiting_pred = false;
1128  if (visited.find(pred) != visited.end()) {
1129  // Note we don't just continue if a pred is being revisited
1130  // because the path is different (different "current" inst.)
1131  // so we might still have to remove the reg.
1132  revisiting_pred = true;
1133  } else {
1134  visited.insert(pred);
1135  }
1136 
1137  // Check whether removing the register between current and pred
1138  // would be functionally correct. If not, then for now return an
1139  // empty list of predecessors.
1140  // Another solution would be to break up the path into smaller
1141  // paths by keeping some intermediate registers to
1142  // maintain correctness.
1143  mc_debug() << "\n " << *pred;
1144 
1145  if (!can_path_still_be_combinational(pred)) {
1146  mc_debug() << "\n Illegal, aborting current path\n";
1147 
1148  // Stop traversing this path in the DFS
1149  // Also if any instruction in this path was going to have its
1150  // register removed, undo this here
1151  remove_all_pairs_in_path_from_multimap(cur_path, remove_reg_pairs,
1152  keep_reg_pairs);
1153  continue;
1154  }
1155 
1156  // Determine whether or not this predecessor has a register
1157  bool register_between_current_and_pred =
1158  does_pred_have_register(pred, current, true, sched->getFSM(Fp));
1159 
1160  // If there is a register between current and pred, add pred to
1161  // the vector. Otherwise push it onto the queue to examine
1162  // its predecessors.
1163  if (register_between_current_and_pred) {
1164  mc_debug() << "\n Registered";
1165  cur_path.push_back(pred);
1166  print_path_helper(cur_path);
1167  paths.push_back(cur_path);
1168  cur_path.pop_back();
1169  } else {
1170  // No need to push pred on the frontier again
1171  if (!revisiting_pred) {
1172  frontier.push(pred);
1173  mc_debug() << "\n Pushing onto queue";
1174  num_unfinished_preds[current]++;
1175  }
1176 
1177  // If this is a flow where registers are being removed from
1178  // data paths in the RTL, do the removal here.
1179  //
1180  // Don't remove registers between this pair of instructions if
1181  // we previously identified this as a pair which should have a
1182  // register in between
1183  if (keep_reg_pairs.find(std::make_pair(pred, current)) ==
1184  keep_reg_pairs.end()) {
1185  remove_reg_pairs.insert(
1186  std::pair<Instruction *, Instruction *>(pred, current));
1187  }
1188  }
1189  }
1190  mc_debug() << "\n Finished all preds of current = " << *current;
1191 }

1195 void GenerateRTL::create_multicycle_paths_from_root(Instruction *root) {
1196  // Set up DFS
1197  std::stack<Instruction *> frontier;
1198  std::set<Instruction *> visited;
1199  frontier.push(root);
1200  visited.insert(root);
1201 
1202  // Keep track of the current path
1203  std::vector<Instruction *> cur_path;
1204  std::vector<std::vector<Instruction *>> paths;
1205  std::vector<std::string> sdc_constraints;
1206  std::vector<std::string> qsf_constraints;
1207  std::map<Instruction *, int> num_unfinished_preds;
1208

1612 // Return the minimum number of FSM states betweeen these two basic
1613 // blocks (from the end of src to the start of dst)
1614 // If src is a direct predecessor of pred, return 0
1615 int GenerateRTL::min_state_difference(BasicBlock *src, BasicBlock *dst) {
1616  if (is_direct_predecessor(src, dst)) {
1617  return 0;
1618  }
1619 
1620  // MULTICYCLE_TODO:
1621  // Corner case -- if src and dst are the same, that means we called
1622  // this function to find the shortest loop from the BB back to itself.
1623  // However currently the function would return 0 for that case. Since
1624  // we know that this is not a direct loop back (or else above we'd have
1625  // returned 0), for now return 1.
1626  if (src == dst) {
1627  return 1;
1628  }
1629 
1630  // Algorithm: Uniform Cost Search (like Dijkstra for 1 path)
1631  // Start at dst and find shortest path up to src
1632 
1633  // Two data structures, a priority queue of BB to search,
1634  // sorted by key (distance), and a reverse mapping (BB->dist)
1635  std::set<std::pair<int, BasicBlock *>> pq;
1636  std::map<BasicBlock *, int> BB_to_dist;
1637 
1638  // Initialize distance to dst to be 0
1639  pq.insert(std::pair<int, BasicBlock *>(0, dst));
1640  BB_to_dist[dst] = 0;
1641 
1642  // Outer loop: pop BB with lowest dist and iterate over neighbors
1643  while (!pq.empty()) {
1644  // Get the next lowest distance and its BB, and remove it from the pq
1645  int d = pq.begin()->first; // Guaranteed to be the lowest dist
1646  BasicBlock *curr =
1647  pq.begin()->second; // An arbitrary BB with this min dist
1648  pq.erase(pq.begin());
1649 
1650  // Stop if we've reached src
1651  if (curr == src) {
1652  break;
1653  }
1654 
1655  // Inner loop: update dist for all preds
1656  for (pred_iterator PI = pred_begin(curr), E = pred_end(curr); PI != E;
1657  ++PI) {
1658  BasicBlock *pred = *PI;
1659  int new_dist = d + alloc->get_num_states_in_BB(pred);
1660 
1661  // If we've never seen this pred before, add it
1662  if (BB_to_dist.find(pred) == BB_to_dist.end()) {
1663  pq.insert(std::pair<int, BasicBlock *>(new_dist, pred));
1664  BB_to_dist[pred] = new_dist;
1665  continue;
1666  }
1667 
1668  // Otherwise, check if we've found a shorter path
1669  int old_dist = BB_to_dist[pred];
1670  if (new_dist < old_dist) {
1671  pq.erase(std::pair<int, BasicBlock *>(old_dist, pred));
1672  pq.insert(std::pair<int, BasicBlock *>(new_dist, pred));
1673  BB_to_dist[pred] = new_dist;
1674  }
1675  }
1676  }
1677  // Return the distance from dst --> src, however ignore the
1678  // length of src itself
1679  int total_dist = BB_to_dist[src] - alloc->get_num_states_in_BB(src);
1680 
1681  // Corner case: In case the dist is ever 0 return 1, because we know
1682  // that src and dst have at least 1 BB between them
1683  total_dist = std::max(1, total_dist);
1684 
1685  return total_dist;
1686 }

1688 // Return the maximum number of FSM states betweeen these two basic
1689 // blocks (from the end of src to the start of dst)
1690 int GenerateRTL::max_state_difference(BasicBlock *src, BasicBlock *dst) {
1691  // Algorithm: Breadth-First Search with cycle checking
1692  // Start at dst and find all paths up to src
1693 
1694  // Set up BFS
1695  std::queue<BasicBlock *> frontier;
1696  frontier.push(dst);
1697 
1698  // Map each BB to its max distance so far
1699  // This map is also used to track already visited BB
1700  std::map<BasicBlock *, int> BB_to_dist;
1701  BB_to_dist[dst] = 0;
1702 
1703  // Do BFS
1704  while (!frontier.empty()) {
1705  // Pop next
1706  BasicBlock *curr = frontier.front();
1707  frontier.pop();
1708 
1709  // Loop over pred BBs of curr
1710  for (pred_iterator PI = pred_begin(curr), E = pred_end(curr); PI != E;
1711  ++PI) {
1712  BasicBlock *pred = *PI;
1713 
1714  // Get the distance from the start to pred along this path
1715  int new_dist = BB_to_dist[curr] + alloc->get_num_states_in_BB(pred);
1716 
1717  // If we've never seen this pred before, add its distance
1718  // and also add it to the queue
1719  if (BB_to_dist.find(pred) == BB_to_dist.end()) {
1720  BB_to_dist[pred] = new_dist;
1721  frontier.push(pred);
1722  }
1723  // Otherwise, check if we've found a longer path
1724  else if (new_dist > BB_to_dist[pred]) {
1725  BB_to_dist[pred] = new_dist;
1726  }
1727  }
1728  }
1729  // Return the distance from dst --> src, however ignore the
1730  // length of src itself
1731  int total_dist = BB_to_dist[src] - alloc->get_num_states_in_BB(src);
1732 
1733  // Corner case: In case the dist is ever 0 return 1, because we know
1734  // that src and dst have at least 1 BB between them
1735  total_dist = std::max(1, total_dist);

2015 // Get the state # of an instruction given that it is the source
2016 // of an MC path
2017 int GenerateRTL::get_source_state(Instruction *source) {
2018 
2019  int source_state = alloc->get_registered_instruction_state(source);
2020 
2021  // If the source instruction is a PHI then its state is actually "-1"
2022  // in its Basic Block
2023  if (isa<PHINode>(source)) {
2024  source_state -= 1;
2025  }
2026 
2027  // If this source has latency > 0, then only the
2028  // first state was stored, so adjust source_state.
2029  //
2030  // Note: we subtract 1 from the # instruction cycles below
2031  // because the code assumes that an instruction scheduled in
2032  // state X has its value stored in a register at the end of
2033  // state X, and that this register can be read starting from
2034  // state X+1. E.g. if we have an adder in state 9, then this

2039  // be "ready to use" in state 11. Using the analogy of the
2040  // adder, it is as if the "load's logic" is in state 10 and the
2041  // load's register therefore is in state 10 as well, so the load

2043  // to extend the source state of the load by 1, not 2.
2044  if (Scheduler::getNumInstructionCycles(source) > 1) {

2366  // calculated,
2367  // or if there was cycle slack.
2368  //
2369  // The following example illustrates this case:
2370  //
2371  // In a BB, I have some register %a. The value of %a is used in a
2372  // series of computations (a path) and then the final output of
2373  // the path is wire %b.
2374  //
2375  // The wire %b is then used in a check like this:
2376  //
2377  // always @(posedge clk)
2378  // if (cur_state == LEGUP_X & b == 1'b1)
2379  // cur_state <= LEGUP_Y;

2381  // The IR would look something like this:
2382  //

2384  // %b = add i32 %a, 4
2385  // %cmp = icmp eq i32 %b, 1
2386  // br i1 %cmp, label %Y, label %Z

2396  // cannot multi-cycle the path from %a->cur_state.

2407  //

2717  //

2768  //

2803  //

2816  // \|/
2817  // |

2819  // |

2823  // synthesized away
2824  // because only {2,3,4} are needed to uniquely identify the paths.

2827  //

2829  // |
2830  // W1

2832  // W2 |
2833  // \|
2834  // W3
2835  // |
2836  // DST

2839  // unique
2840  // signal per path) I'll just print the minimum of all the paths.
