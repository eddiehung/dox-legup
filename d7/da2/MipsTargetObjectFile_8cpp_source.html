57 /// IsGlobalInSmallSection - Return true if this global address should be
58 /// placed into small data/bss section.
59 bool MipsTargetObjectFile::
60 IsGlobalInSmallSection(const GlobalValue *GV, const TargetMachine &TM,
61  SectionKind Kind) const {
62 
63  const MipsSubtarget &Subtarget = TM.getSubtarget<MipsSubtarget>();
64 
65  // Return if small section is not available.
66  if (!Subtarget.useSmallSection())
67  return false;
68 
69  // Only global variables, not functions.
70  const GlobalVariable *GVA = dyn_cast<GlobalVariable>(GV);
71  if (!GVA)
72  return false;
73 
74  // We can only do this for datarel or BSS objects for now.
75  if (!Kind.isBSS() && !Kind.isDataRel())
76  return false;
77 
78  // If this is a internal constant string, there is a special
79  // section for it, but not in small data/bss.
80  if (Kind.isMergeable1ByteCString())
81  return false;
82 
83  Type *Ty = GV->getType()->getElementType();
84  return IsInSmallSection(TM.getDataLayout()->getTypeAllocSize(Ty));
85 }
