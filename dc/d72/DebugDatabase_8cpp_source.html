612 void DebugDatabase::getGEPBaseAndOffset(ConstantExpr *GEP, Value **getBase,
613  int *getOffset) {
614  int offset = 0;
615  Value *base = NULL;
616 
617  User *baseAddress = GEP->getOperand(0);
618  ConstantExpr *CE = dyn_cast<ConstantExpr>(baseAddress);
619 
620  if (CE && CE->getOpcode() == Instruction::GetElementPtr) {
621  // Nested GEP
622  getGEPBaseAndOffset(CE, &base, &offset);
623  } else if (isa<GlobalVariable>(baseAddress) ||

626  } else {
627  assert(false);
628  }
629 
630  assert(base);
631 
632  gep_type_iterator GTI = gep_type_begin(GEP);
633  for (User::op_iterator i = GEP->op_begin() + 1, e = GEP->op_end(); i != e;
634  ++i, ++GTI) {
635  Value *Op = *i;
636 
637  // Build a mask for high order bits.
638  const DataLayout *TD = dbgInfo->getAlloc()->getDataLayout();
639  unsigned IntPtrWidth = TD->getPointerSizeInBits();

641 
642  // apply mask
