139  bool isLegalValueType(MVT::SimpleValueType VT) const {
140  ArrayRef<MVT::SimpleValueType> LegalVTs = getLegalValueTypes();
141  for (unsigned i = 0, e = LegalVTs.size(); i != e; ++i)
