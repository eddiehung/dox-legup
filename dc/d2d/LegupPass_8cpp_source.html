68 std::vector<Function*> LegupPass::getDepthFirstSortedFunctions(Module &M) {
69  // TODO: Use a more efficient algorithm for this.
70  size_t previousSetSize = 0;
71  std::vector<Function*> sortedSet;
72  std::vector<std::string> addedFunctionNames;
73  size_t addedFunctionCount = 0;
74  do {
75  previousSetSize = sortedSet.size();
76  for (Module::iterator f = M.begin(), FE = M.end(); f != FE; ++f) {
77  Function &F = *f;
78  if (std::find(addedFunctionNames.begin(), addedFunctionNames.end(),
79  F.getName().str()) != addedFunctionNames.end())
80  continue;
81  bool allCalledFunctionsAddedToSet = true;
82 
83  for (Function::iterator b = F.begin(), be = F.end(); b != be; ++b) {
84 
85  for (BasicBlock::iterator instr = b->begin(), ie = b->end();
86  instr != ie; ++instr) {
87  if (isaDummyCall(instr))
88  continue;
89 
90  if (CallInst *CI = dyn_cast<CallInst>(instr)) {
91  llvm::Function *called = getCalledFunction(CI);
92  if (std::find(addedFunctionNames.begin(),
93  addedFunctionNames.end(),
94  called->getName().str())
95  == addedFunctionNames.end()) {
96  allCalledFunctionsAddedToSet = false;
97  break;
98  }
99  }
100  } // end Block Iterator
101  if (!allCalledFunctionsAddedToSet)
102  break;
103  } // end Function Iterator
104  if (allCalledFunctionsAddedToSet) {
105  sortedSet.push_back(&F);
106  addedFunctionNames.push_back(F.getName().str());
107  addedFunctionCount ++;
108  }
109  } // end Module iterator
110  } while (previousSetSize != sortedSet.size()
111  && addedFunctionCount <= M.getFunctionList().size());
112  assert(addedFunctionCount == M.getFunctionList().size());
113  return sortedSet;
114 }

125 // number of instructions and basic blocks in each loop/function
126 void LegupPass::printBBStats(Function &F) {
127  if (!LEGUP_CONFIG->getParameterInt("PRINT_BB_STATS"))
128  return;
129 
130  errs() << "Statistics\n";
131  // note: the loop stats will double count some basic blocks that are
132  // part of multiple loops. For instance, both the inner and the outer
133  // body of a loop nest will count. This may affect the calculated
134  // median/average
135  errs() << "Function: " << F.getName() << "\n";
136  LoopInfo &LI = getAnalysis<LoopInfo>(F);
137  int totalBBs = 0;
138  int totalInst = 0;
139  int totalMemInst = 0;
140  int numLoops = 0;
141  for (LoopInfo::iterator li = LI.begin(), le = LI.end(); li != le; ++li) {
142  numLoops++;
143  Loop *loop = *li;
144  BasicBlock *loopPreheader;
145  loopPreheader = loop->getLoopPreheader();
146  errs() << "Loop " << numLoops << " preheader name: "
147  << getLabel(loopPreheader) << "\n";
148  int numBBs = 0;
149  int numInstLoop = 0;
150  int numMemInstLoop = 0;
151  for (Loop::block_iterator bb = loop->block_begin(), eb =
152  loop->block_end(); bb != eb; ++bb) {
153  BasicBlock *BB = *bb;
154  int numInst = 0;
155  int numMemInst = 0;
156  for (BasicBlock::iterator I = BB->begin(), ie = BB->end(); I != ie;
157  ++I) {
158  numInst++;
159  if (isMem(I))
160  numMemInst++;
161  }
162  errs() << "Number of Instructions (Loop,BB): " << numInst << "\n";
163  errs() << "Number of Mem Instructions (Loop,BB): " << numMemInst
164  << "\n";
165 
166  numInstLoop += numInst;
167  numMemInstLoop += numMemInst;
168  numBBs++;
169  }
170  errs() << "Number of Instructions (Loop): " << numInstLoop << "\n";
171  errs() << "Number of Mem Instructions (Loop): " << numMemInstLoop
172  << "\n";
173  errs() << "Number of Basic Blocks (Loop): " << numBBs << "\n";
174  totalInst += numInstLoop;
175  totalMemInst += numMemInstLoop;
176  totalBBs += numBBs;
177  }
178  errs() << "Number of Loops: " << numLoops << "\n";
179  errs() << "Number of Mem Instructions (Func): " << totalMemInst << "\n";
180  errs() << "Number of Basic Blocks (Func): " << totalBBs << "\n";
181  if (totalBBs > 0)
182  errs() << "Inst/Basic Blocks: " << totalInst / totalBBs << "\n";
183 
184  errs() << "Function: " << F.getName() << "\n";
185  totalBBs = 0;
186  totalInst = 0;
187  totalMemInst = 0;
188  for (Function::iterator BB = F.begin(), be = F.end(); BB != be; ++BB) {
189  int numInst = 0;
190  int numMemInst = 0;
191  for (BasicBlock::iterator I = BB->begin(), ie = BB->end(); I != ie;
192  ++I) {
193  numInst++;
194  if (isMem(I))
195  numMemInst++;
196  }
197  errs() << "Number of Instructions (BB): " << numInst << "\n";
198  errs() << "Number of Mem Instructions (BB): " << numMemInst << "\n";
199 
200  totalInst += numInst;
201  totalMemInst += numMemInst;
202  totalBBs++;
203  }
204  errs() << "Number of Instructions (Func): " << totalInst << "\n";
205  errs() << "Number of Mem Instructions (Func): " << totalMemInst << "\n";
206  errs() << "Number of Basic Blocks (Func): " << totalBBs << "\n";
207  errs() << "Inst/Basic Blocks: " << totalInst/totalBBs << "\n";
208 }

346 void LegupPass::pipelineLabelSanityCheck(Module &M) {
347  std::set<std::string> pipelinedLabels;
348 
349  for (Module::iterator F = M.begin(), E = M.end(); F != E; ++F) {
350  for (Function::iterator b = F->begin(), be = F->end(); b != be; ++b) {
351  TerminatorInst *TI = b->getTerminator();
352  if (getMetadataInt(TI, "legup.pipelined")) {
353  std::string label = getMetadataStr(TI, "legup.label");
354  pipelinedLabels.insert(label);
355  }
356  }
357  }
358 
359  int totalLoopsPipelined = pipelinedLabels.size();
360 
361  int expected = LEGUP_CONFIG->numLoopPipelines();
362 
363  if (totalLoopsPipelined != expected) {
364  errs() << "Error: Expected to find " << expected
365  << " loops to pipeline but only pipelined "
366  << totalLoopsPipelined << "\n";
367  errs() << "Note: loop pipeline labels only work in the main .c file\n";
368  }
369 
370  std::map<std::string, LegupConfig::LOOP_PIPELINE> &loop_pipelines =
371  LEGUP_CONFIG->getAllLoopPipelines();
372 
373  for (std::map<std::string, LegupConfig::LOOP_PIPELINE>::iterator i =
374  loop_pipelines.begin(), ie = loop_pipelines.end(); i != ie;
375  ++i) {
376  std::string label = i->first;
377  if (pipelinedLabels.find(label) == pipelinedLabels.end()) {
378  errs() << "Couldn't pipeline loop with label: " << label << "\n";
379  }
380  }
381 }

383 bool LegupPass::doFinalization(Module &M) {
384  std::set<const Function*> AcceleratedFcts;
385 
386  if (M.begin() == M.end()) {
387  llvm_unreachable("No functions exist in the module!\n");
388  }
389 
390  for (Module::iterator F = M.begin(), FE = M.end(); F != FE; ++F) {
391  if (LEGUP_CONFIG->isAccelerated(*F)) {
392  AcceleratedFcts.insert(F);
393  }
394  }
395 
396  allocation->addGlobalDefines();
397 
398  if (allocation->getDbgInfo()->isDatabaseEnabled()) {
399  // Assign instance IDs to each module instance
400  allocation->getDbgInfo()->assignInstances();
401  }
402 
403  if (allocation->getDbgInfo()->isDebugRtlEnabled()) {
404  allocation->getDbgInfo()->addDebugRtl();
405  }
406 
407  if (allocation->getDbgInfo()->isDatabaseEnabled()) {
408  allocation->getDbgInfo()->outputDebugDatabase();
409  }
410 
411  // TODO - instantiate local RAMs in Generate RTL data structure instead of
412  // verilog writer and re-enable this
413  if (!LEGUP_CONFIG->getParameterInt("LOCAL_RAMS")) {
414  if (!LEGUP_CONFIG->getParameterInt("KEEP_SIGNALS_WITH_NO_FANOUT")) {
415  for (Allocation::const_rtl_iterator i = allocation->rtl_begin(), e =
416  allocation->rtl_end(); i != e; ++i) {
417  RTLModule *rtl = *i;
418 
419  // delete register signals that are unconnected
420  rtl->removeSignalsWithoutFanout();
421 
422  rtl->verifyConnections(allocation);
423 
424  }
425  }
426  }
427 
428  if (!LEGUP_CONFIG->getParameterInt("NO_LOOP_PIPELINING")) {
429  for (Allocation::const_rtl_iterator i = allocation->rtl_begin(), e =
430  allocation->rtl_end(); i != e; ++i) {
431  RTLModule *rtl = *i;
432  if (rtl->getName() == "main") {
433  rtl->buildCircuitStructure();
434  formatted_raw_ostream out(allocation->getPipelineDotFile());
435  rtl->printPipelineDot(out);
436  }
437  }
438  }
439 
440  printVerilog(AcceleratedFcts);
441 
442  printResourcesFile("resources.legup.rpt");
443 
444  std::list<GenerateRTL::PATH *> *overallPaths =
445  allocation->getOverallLongestPaths();
446  std::string fileName = "timingReport.overall.legup.rpt";
447  std::string Error = "Error in printing timing report\n";
448  raw_fd_ostream overallReport(fileName.c_str(), Error,
449  llvm::sys::fs::F_None);
450  overallReport << getFileHeader();
451  GenerateRTL::printPath(overallReport, overallPaths);
452 
453  if (LEGUP_CONFIG->getParameterInt("INSPECT_DEBUG")) {
454  dbger.fillHardwareInfo(allocation);
455  dbger.fillSignals(allocation);
456  dbger.fillVariables(allocation);
457 
458  dbger.StateStoreInfoMapping(allocation);
459  } else if (LEGUP_CONFIG->getParameterInt(
460  "INSPECT_ONCHIP_BUG_DETECT_DEBUG")) {
461  dbger.fillCurStateAndFinishSignals(allocation);
462  }
463 
464  return false;
465 }
