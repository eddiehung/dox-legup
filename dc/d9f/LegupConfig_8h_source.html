562  void check_op_exists(const std::string op_name, int &latency) {
563  if (Operations.find(std::make_pair(op_name, latency)) != Operations.end()) {
564  return;
565  }
566 
567  int i;
568  for (i = 0; i <= maxLatency; i++) {
569  if (Operations.find(std::make_pair(op_name, i)) != Operations.end()) {
570  FuncUnitLatencyWarning[op_name] = std::make_pair(latency, i);
571  latency = i;
572  return;
573  }
574  }
575  errs() << "Error: Couldn't find operation '" << op_name
576  << "' in device tcl file!\n";
577  assert(0);
578  }
