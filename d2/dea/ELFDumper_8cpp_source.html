93 std::error_code createELFDumper(const object::ObjectFile *Obj,
94  StreamWriter &Writer,
95  std::unique_ptr<ObjDumper> &Result) {
96  // Little-endian 32-bit
97  if (const ELF32LEObjectFile *ELFObj = dyn_cast<ELF32LEObjectFile>(Obj))
98  return createELFDumper(ELFObj->getELFFile(), Writer, Result);
99 
100  // Big-endian 32-bit
101  if (const ELF32BEObjectFile *ELFObj = dyn_cast<ELF32BEObjectFile>(Obj))
102  return createELFDumper(ELFObj->getELFFile(), Writer, Result);
103 
104  // Little-endian 64-bit
105  if (const ELF64LEObjectFile *ELFObj = dyn_cast<ELF64LEObjectFile>(Obj))
106  return createELFDumper(ELFObj->getELFFile(), Writer, Result);
107 
108  // Big-endian 64-bit
109  if (const ELF64BEObjectFile *ELFObj = dyn_cast<ELF64BEObjectFile>(Obj))
110  return createELFDumper(ELFObj->getELFFile(), Writer, Result);
111 
112  return readobj_error::unsupported_obj_file_format;
113 }

117 template <typename ELFO>
118 static std::string getFullSymbolName(const ELFO &Obj,
119  typename ELFO::Elf_Sym_Iter Symbol) {
120  StringRef SymbolName = errorOrDefault(Obj.getSymbolName(Symbol));
121  if (!Symbol.isDynamic())
122  return SymbolName;
123 
124  std::string FullSymbolName(SymbolName);
125 
126  bool IsDefault;
127  ErrorOr<StringRef> Version =
128  Obj.getSymbolVersion(nullptr, &*Symbol, IsDefault);
129  if (Version) {
130  FullSymbolName += (IsDefault ? "@@" : "@");
131  FullSymbolName += *Version;
132  } else
133  error(Version.getError());
134  return FullSymbolName;
135 }

137 template <typename ELFO>
138 static void
139 getSectionNameIndex(const ELFO &Obj, typename ELFO::Elf_Sym_Iter Symbol,
140  StringRef &SectionName, unsigned &SectionIndex) {
141  SectionIndex = Symbol->st_shndx;
142  if (SectionIndex == SHN_UNDEF) {
143  SectionName = "Undefined";
144  } else if (SectionIndex >= SHN_LOPROC && SectionIndex <= SHN_HIPROC) {
145  SectionName = "Processor Specific";
146  } else if (SectionIndex >= SHN_LOOS && SectionIndex <= SHN_HIOS) {
147  SectionName = "Operating System Specific";
148  } else if (SectionIndex > SHN_HIOS && SectionIndex < SHN_ABS) {
149  SectionName = "Reserved";
150  } else if (SectionIndex == SHN_ABS) {
151  SectionName = "Absolute";
152  } else if (SectionIndex == SHN_COMMON) {
153  SectionName = "Common";
154  } else {
155  if (SectionIndex == SHN_XINDEX)
156  SectionIndex = Obj.getSymbolTableIndex(&*Symbol);
157  assert(SectionIndex != SHN_XINDEX &&
158  "getSymbolTableIndex should handle this");
159  const typename ELFO::Elf_Shdr *Sec = Obj.getSection(SectionIndex);
160  SectionName = errorOrDefault(Obj.getSectionName(Sec));
161  }
162 }

1065 template <class ELFT>
1066 void MipsGOTParser<ELFT>::parseGOT(const Elf_Shdr &GOTShdr) {
1067  // See "Global Offset Table" in Chapter 5 in the following document
1068  // for detailed GOT description.
1069  // ftp://www.linux-mips.org/pub/linux/mips/doc/ABI/mipsabi.pdf
1070 
1071  ErrorOr<ArrayRef<uint8_t>> GOT = Obj->getSectionContents(&GOTShdr);
1072  if (!GOT) {
1073  W.startLine() << "The .got section is empty.\n";
1074  return;
1075  }
1076 
1077  uint64_t DtLocalGotNum;
1078  uint64_t DtGotSym;
1079  if (!getGOTTags(DtLocalGotNum, DtGotSym))
1080  return;
1081 
1082  if (DtLocalGotNum > getGOTTotal(*GOT)) {
1083  W.startLine() << "MIPS_LOCAL_GOTNO exceeds a number of GOT entries.\n";
1084  return;
1085  }
1086 
1087  Elf_Sym_Iter DynSymBegin = Obj->begin_dynamic_symbols();
1088  Elf_Sym_Iter DynSymEnd = Obj->end_dynamic_symbols();
1089  std::size_t DynSymTotal = std::size_t(std::distance(DynSymBegin, DynSymEnd));
1090 
1091  if (DtGotSym > DynSymTotal) {
1092  W.startLine() << "MIPS_GOTSYM exceeds a number of dynamic symbols.\n";
1093  return;
1094  }
1095 
1096  std::size_t GlobalGotNum = DynSymTotal - DtGotSym;
1097 
1098  if (DtLocalGotNum + GlobalGotNum > getGOTTotal(*GOT)) {
1099  W.startLine() << "Number of global GOT entries exceeds the size of GOT.\n";
1100  return;
1101  }
1102 
1103  GOTIter GotBegin = makeGOTIter(*GOT, 0);
1104  GOTIter GotLocalEnd = makeGOTIter(*GOT, DtLocalGotNum);
1105  GOTIter It = GotBegin;
1106 
1107  DictScope GS(W, "Primary GOT");
1108 
1109  W.printHex("Canonical gp value", GOTShdr.sh_addr + 0x7ff0);
1110  {
1111  ListScope RS(W, "Reserved entries");
1112 
1113  {
1114  DictScope D(W, "Entry");
1115  printGotEntry(GOTShdr.sh_addr, GotBegin, It++);
1116  W.printString("Purpose", StringRef("Lazy resolver"));
1117  }
1118 
1119  if (It != GotLocalEnd && (*It >> (sizeof(GOTEntry) * 8 - 1)) != 0) {
1120  DictScope D(W, "Entry");
1121  printGotEntry(GOTShdr.sh_addr, GotBegin, It++);
1122  W.printString("Purpose", StringRef("Module pointer (GNU extension)"));
1123  }
1124  }
1125  {
1126  ListScope LS(W, "Local entries");
1127  for (; It != GotLocalEnd; ++It) {
1128  DictScope D(W, "Entry");
1129  printGotEntry(GOTShdr.sh_addr, GotBegin, It);
1130  }
1131  }
1132  {
1133  ListScope GS(W, "Global entries");
1134 
1135  GOTIter GotGlobalEnd = makeGOTIter(*GOT, DtLocalGotNum + GlobalGotNum);
1136  Elf_Sym_Iter GotDynSym = DynSymBegin + DtGotSym;
1137  for (; It != GotGlobalEnd; ++It) {
1138  DictScope D(W, "Entry");
1139  printGlobalGotEntry(GOTShdr.sh_addr, GotBegin, It, GotDynSym++);
1140  }
1141  }
1142 
1143  std::size_t SpecGotNum = getGOTTotal(*GOT) - DtLocalGotNum - GlobalGotNum;
1144  W.printNumber("Number of TLS and multi-GOT entries", uint64_t(SpecGotNum));
1145 }
