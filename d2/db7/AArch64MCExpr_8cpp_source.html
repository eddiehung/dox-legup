103 static void fixELFSymbolsInTLSFixupsImpl(const MCExpr *Expr, MCAssembler &Asm) {
104  switch (Expr->getKind()) {
105  case MCExpr::Target:
106  llvm_unreachable("Can't handle nested target expression");
107  break;
108  case MCExpr::Constant:
109  break;
110 
111  case MCExpr::Binary: {
112  const MCBinaryExpr *BE = cast<MCBinaryExpr>(Expr);
113  fixELFSymbolsInTLSFixupsImpl(BE->getLHS(), Asm);
114  fixELFSymbolsInTLSFixupsImpl(BE->getRHS(), Asm);
115  break;
116  }
117 
118  case MCExpr::SymbolRef: {
119  // We're known to be under a TLS fixup, so any symbol should be
120  // modified. There should be only one.
121  const MCSymbolRefExpr &SymRef = *cast<MCSymbolRefExpr>(Expr);
122  MCSymbolData &SD = Asm.getOrCreateSymbolData(SymRef.getSymbol());
123  MCELF::SetType(SD, ELF::STT_TLS);
124  break;
125  }
126 
127  case MCExpr::Unary:
128  fixELFSymbolsInTLSFixupsImpl(cast<MCUnaryExpr>(Expr)->getSubExpr(), Asm);
129  break;
130  }
131 }
