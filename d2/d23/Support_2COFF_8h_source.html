410  /// @brief The PE32 Import Lookup Table.

417  /// when it is initially written out to the image.
418  struct ImportLookupTableEntry32 {
419  uint32_t data;
420 
421  /// @brief Is this entry specified by ordinal, or name?
422  bool isOrdinal() const { return data & 0x80000000; }
423 
424  /// @brief Get the ordinal value of this entry. isOrdinal must be true.
425  uint16_t getOrdinal() const {
426  assert(isOrdinal() && "ILT entry is not an ordinal!");
427  return data & 0xFFFF;
428  }
