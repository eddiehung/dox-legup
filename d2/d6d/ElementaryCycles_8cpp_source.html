260 bool FindElementaryCycles::circuit(int v) { // based on Johnson ’s logical procedure CIRCUIT
261  bool f = false;
262  stackPush(v);
263  blocked[v] = true;
264  for (int wPos = 1; wPos <= Ak[v][0]; wPos++) { // for each w in list Ak[v]:
265  int w = Ak[v][wPos];
266  if (w < start) continue; // ignore relevant parts of Ak
267  if (w == start) { // we have a circuit,
268  if (enumeration) {
269  stackPrint3d(); // print out the stack to record the circuit
270  }
271  assert (stackTop <= nVertices);
272  ++lengthHistogram[stackTop]; // add this circuit ’s length to the length histogram
273  nCircuits++; // and increment count of circuits found
274  if (stackTop > lenLongest) { // keep a copy of the longest circuit found
275  lenLongest = stackTop;
276  longestCircuit = stack;
277  }
278  for (int i = 0; i < stackTop; i ++) // increment [circuit-length][vertex] for all vertices in this circuit
279  ++vertexPopularity[stackTop][stack[i]];
280  f = true;
281  } else if (!blocked[w]) {
282  if (circuit(w)) f = true;
283  }
284  }
285  if (f) {
286  unblock (v);
287  } else {
288  for (int wPos = 1; wPos <= Ak[v][0]; wPos++) { // for each w in list Ak[v]:
289  int w = Ak[v][wPos];
290  if (w < start) continue; // ignore relevant parts of Ak
291  if (notInList(B[w], v)) addToList(B[w], v);
292  }
293  }
294  v = stackPop();
295  return f;
296 }
