102 void AMDGPUTTI::getUnrollingPreferences(Loop *L,
103  UnrollingPreferences &UP) const {
104  for (const BasicBlock *BB : L->getBlocks()) {
105  for (const Instruction &I : *BB) {
106  const GetElementPtrInst *GEP = dyn_cast<GetElementPtrInst>(&I);
107  if (!GEP || GEP->getAddressSpace() != AMDGPUAS::PRIVATE_ADDRESS)
108  continue;
109 
110  const Value *Ptr = GEP->getPointerOperand();
111  const AllocaInst *Alloca = dyn_cast<AllocaInst>(GetUnderlyingObject(Ptr));
112  if (Alloca) {
113  // We want to do whatever we can to limit the number of alloca
114  // instructions that make it through to the code generator. allocas
115  // require us to use indirect addressing, which is slow and prone to
116  // compiler bugs. If this loop does an address calculation on an
117  // alloca ptr, then we want to use a higher than normal loop unroll
118  // threshold. This will give SROA a better chance to eliminate these
119  // allocas.
120  //
121  // Don't use the maximum allowed value here as it will make some
122  // programs way too big.
123  UP.Threshold = 500;
124  }
125  }
126  }
127 }
