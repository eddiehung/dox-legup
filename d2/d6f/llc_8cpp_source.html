103 // GetFileNameRoot - Helper function to get the basename of a filename.
104 static inline std::string
105 GetFileNameRoot(const std::string &InputFilename) {
106  std::string IFN = InputFilename;
107  std::string outputFilename;
108  int Len = IFN.length();
109  if ((Len > 2) &&
110  IFN[Len-3] == '.' &&
111  ((IFN[Len-2] == 'b' && IFN[Len-1] == 'c') ||
112  (IFN[Len-2] == 'l' && IFN[Len-1] == 'l'))) {
113  outputFilename = std::string(IFN.begin(), IFN.end()-3); // s/.bc/.s/
114  } else {
115  outputFilename = IFN;
116  }
117  return outputFilename;
118 }

120 static tool_output_file *GetOutputStream(const char *TargetName,
121  Triple::OSType OS,
122  const char *ProgName) {
123  // If we don't yet have an output filename, make one.
124  if (OutputFilename.empty()) {
125  if (InputFilename == "-")
126  OutputFilename = "-";
127  else {
128  OutputFilename = GetFileNameRoot(InputFilename);
129 
130  switch (FileType) {
131  case TargetMachine::CGFT_AssemblyFile:
132  if (TargetName[0] == 'c') {
133  if (TargetName[1] == 0)
134  OutputFilename += ".cbe.c";
135  else if (TargetName[1] == 'p' && TargetName[2] == 'p')
136  OutputFilename += ".cpp";
137  else
138  OutputFilename += ".s";
139  } else if (TargetName[0] == 'v') {
140  OutputFilename += ".v";
141  } else
142  OutputFilename += ".s";
143  break;
144  case TargetMachine::CGFT_ObjectFile:
145  if (OS == Triple::Win32)
146  OutputFilename += ".obj";
147  else
148  OutputFilename += ".o";
149  break;
150  case TargetMachine::CGFT_Null:
151  OutputFilename += ".null";
152  break;
153  }
154  }
155  }
156 
157  // Decide if we need "binary" output.
158  bool Binary = false;
159  switch (FileType) {
160  case TargetMachine::CGFT_AssemblyFile:
161  break;
162  case TargetMachine::CGFT_ObjectFile:
163  case TargetMachine::CGFT_Null:
164  Binary = true;
165  break;
166  }
167 
168  // Open the file.
169  std::string error;
170  sys::fs::OpenFlags OpenFlags = sys::fs::F_None;
171  if (!Binary)
172  OpenFlags |= sys::fs::F_Text;
173  tool_output_file *FDOut = new tool_output_file(OutputFilename.c_str(), error,
174  OpenFlags);
175  if (!error.empty()) {
176  errs() << error << '\n';
177  delete FDOut;
178  return nullptr;
179  }
180 
181  return FDOut;
182 }

224 static int compileModule(char **argv, LLVMContext &Context) {
225  // Load the module to be compiled...
226  SMDiagnostic Err;
227  std::unique_ptr<Module> M;
228  Module *mod = nullptr;
229  Triple TheTriple;
230 
231  bool SkipModule = MCPU == "help" ||
232  (!MAttrs.empty() && MAttrs.front() == "help");
233 
234  // If user asked for the 'native' CPU, autodetect here. If autodection fails,
235  // this will set the CPU to an empty string which tells the target to
236  // pick a basic default.
237  if (MCPU == "native")
238  MCPU = sys::getHostCPUName();
239 
240  // If user just wants to list available options, skip module loading
241  if (!SkipModule) {
242  M.reset(ParseIRFile(InputFilename, Err, Context));
243  mod = M.get();
244  if (mod == nullptr) {
245  Err.print(argv[0], errs());
246  return 1;
247  }
248 
249  // If we are supposed to override the target triple, do so now.
250  if (!TargetTriple.empty())
251  mod->setTargetTriple(Triple::normalize(TargetTriple));
252  TheTriple = Triple(mod->getTargetTriple());
253  } else {
254  TheTriple = Triple(Triple::normalize(TargetTriple));
255  }
256 
257  if (TheTriple.getTriple().empty())
258  TheTriple.setTriple(sys::getDefaultTargetTriple());
259 
260  // Get the target specific parser.
261  std::string Error;
262  const Target *TheTarget = TargetRegistry::lookupTarget(MArch, TheTriple,
263  Error);
264  if (!TheTarget) {
265  errs() << argv[0] << ": " << Error;
266  return 1;
267  }
268 
269  // Package up features to be passed to target/subtarget
270  std::string FeaturesStr;
271  if (MAttrs.size()) {
272  SubtargetFeatures Features;
273  for (unsigned i = 0; i != MAttrs.size(); ++i)
274  Features.AddFeature(MAttrs[i]);
275  FeaturesStr = Features.getString();
276  }
277 
278  CodeGenOpt::Level OLvl = CodeGenOpt::Default;
279  switch (OptLevel) {
280  default:
281  errs() << argv[0] << ": invalid optimization level.\n";
282  return 1;
283  case ' ': break;
284  case '0': OLvl = CodeGenOpt::None; break;
285  case '1': OLvl = CodeGenOpt::Less; break;
286  case '2': OLvl = CodeGenOpt::Default; break;
287  case '3': OLvl = CodeGenOpt::Aggressive; break;
288  }
289 
290  TargetOptions Options = InitTargetOptionsFromCodeGenFlags();
291  Options.DisableIntegratedAS = NoIntegratedAssembler;
292  Options.MCOptions.ShowMCEncoding = ShowMCEncoding;
293  Options.MCOptions.MCUseDwarfDirectory = EnableDwarfDirectory;
294  Options.MCOptions.AsmVerbose = AsmVerbose;
295 
296  std::unique_ptr<TargetMachine> target(
297  TheTarget->createTargetMachine(TheTriple.getTriple(), MCPU, FeaturesStr,
298  Options, RelocModel, CMModel, OLvl));
299  assert(target.get() && "Could not allocate target machine!");
300 
301  // If we don't have a module then just exit now. We do this down
302  // here since the CPU/Feature help is underneath the target machine
303  // creation.
304  if (SkipModule)
305  return 0;
306 
307  assert(mod && "Should have exited if we didn't have a module!");
308  TargetMachine &Target = *target.get();
309 
310  if (GenerateSoftFloatCalls)
311  FloatABIForCalls = FloatABI::Soft;
312 
313  // Figure out where we are going to send the output.
314  std::unique_ptr<tool_output_file> Out(
315  GetOutputStream(TheTarget->getName(), TheTriple.getOS(), argv[0]));
316  if (!Out) return 1;
317 
318  // Build up all of the passes that we want to do to the module.
319  PassManager PM;
320 
321  // Add an appropriate TargetLibraryInfo pass for the module's triple.
322  TargetLibraryInfo *TLI = new TargetLibraryInfo(TheTriple);
323  if (DisableSimplifyLibCalls)
324  TLI->disableAllFunctions();
325  PM.add(TLI);
326 
327  // Add the target data from the target machine, if it exists, or the module.
328  if (const DataLayout *DL = Target.getDataLayout())
329  mod->setDataLayout(DL);
330  PM.add(new DataLayoutPass(mod));
331 
332  // pass tcl files to LegupConfig
333  if (!ConfigFile.empty()) {
334  for (unsigned i = 0; i < ConfigFile.size();++i) {
335  if (!legup::parseTclFile(ConfigFile[i], legup::LEGUP_CONFIG)) {
336  return 1;
337  }
338  }
339  }
340  if (RelaxAll.getNumOccurrences() > 0 &&
341  FileType != TargetMachine::CGFT_ObjectFile)
342  errs() << argv[0]
343  << ": warning: ignoring -mc-relax-all because filetype != obj";
344 
345  {
346  formatted_raw_ostream FOS(Out->os());
347 
348  AnalysisID StartAfterID = nullptr;
349  AnalysisID StopAfterID = nullptr;
350  const PassRegistry *PR = PassRegistry::getPassRegistry();
351  if (!StartAfter.empty()) {
352  const PassInfo *PI = PR->getPassInfo(StartAfter);
353  if (!PI) {
354  errs() << argv[0] << ": start-after pass is not registered.\n";
355  return 1;
356  }
357  StartAfterID = PI->getTypeInfo();
358  }
359  if (!StopAfter.empty()) {
360  const PassInfo *PI = PR->getPassInfo(StopAfter);
361  if (!PI) {
362  errs() << argv[0] << ": stop-after pass is not registered.\n";
363  return 1;
364  }
365  StopAfterID = PI->getTypeInfo();
366  }
367 
368  // Ask the target to add backend passes as necessary.
369  if (Target.addPassesToEmitFile(PM, FOS, FileType, NoVerify,
370  StartAfterID, StopAfterID)) {
371  errs() << argv[0] << ": target does not support generation of this"
372  << " file type!\n";
373  return 1;
374  }
375 
376  // Before executing passes, print the final values of the LLVM options.
377  cl::PrintOptionValues();
378 
379  PM.run(*mod);
380  }
381 
382  // Declare success.
383  Out->keep();
384 
385  return 0;
386 }
