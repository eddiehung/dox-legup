35 /// lowerSDIV - Given an SDiv expressing a divide by constant,
36 /// replace it by multiplying by a magic number. See:
37 /// <http://the.wall.riscom.net/books/proc/ppc/cwg/code2.html>
38 bool lowerSDiv(Instruction *inst) const {
39  ConstantInt *Op1 = dyn_cast<ConstantInt>(inst->getOperand(1));
40  // check if dividing by a constant and not by a power of 2
41  if (!Op1 || inst->getType()->getPrimitiveSizeInBits() != 32 ||
42  Op1->getValue().isPowerOf2()) {
43  return false;
44  }
45 
46  BasicBlock::iterator ii(inst);
47  Value *Op0 = inst->getOperand(0);
48  APInt d = Op1->getValue();
49 
50  APInt::ms magics = Op1->getValue().magic();
51 
52  IntegerType * type64 = IntegerType::get(Mod->getContext(), 64);
53 
54  Instruction *sext = CastInst::CreateSExtOrBitCast(Op0, type64, "", inst);
55  APInt m = APInt(64, magics.m.getSExtValue());
56  Constant *magicNum = ConstantInt::get(type64, m);
57 
58  Instruction *magInst = BinaryOperator::CreateNSWMul(sext, magicNum, "", inst);
59 
60  APInt ap = APInt(64, 32);
61  Constant *movHiConst = ConstantInt::get(type64, ap);
62  Instruction *movHi = BinaryOperator::Create(Instruction::AShr, magInst,
63  movHiConst, "", inst);
64  Instruction *trunc = CastInst::CreateTruncOrBitCast(movHi, inst->getType(),
65  "", inst);
66  if (d.isStrictlyPositive() && magics.m.isNegative()) {
67  trunc = BinaryOperator::Create(Instruction::Add, trunc, Op0, "", inst);
68  } else if (d.isNegative() && magics.m.isStrictlyPositive()) {
69  trunc = BinaryOperator::Create(Instruction::Sub, trunc, Op0, "", inst);
70  }
71  if (magics.s > 0) {
72  APInt apS = APInt(32, magics.s);
73  Constant *magicShift = ConstantInt::get(inst->getType(), apS);
74  trunc = BinaryOperator::Create(Instruction::AShr, trunc,
75  magicShift, "", inst);
76  }
77 
78  APInt ap31 = APInt(32, 31);
79  Constant *thirtyOne = ConstantInt::get(inst->getType(), ap31);
80  // get sign bit
81  Instruction *sign = BinaryOperator::Create(Instruction::LShr, trunc,
82  thirtyOne, "", inst);
83 
84  Instruction *result = BinaryOperator::Create(Instruction::Add, trunc, sign,
85  "");
86  ReplaceInstWithInst(inst->getParent()->getInstList(), ii, result);
87  return true;
88 }

90 /// lowerUDIV - Given an UDiv expressing a divide by constant,
91 /// replace it by multiplying by a magic number. See:
92 /// <http://the.wall.riscom.net/books/proc/ppc/cwg/code2.html>
93 bool lowerUDiv(Instruction *inst) const {
94  ConstantInt *Op1 = dyn_cast<ConstantInt>(inst->getOperand(1));
95  // check if dividing by a constant
96  if (!Op1 || inst->getType()->getPrimitiveSizeInBits() != 32) {
97  return false;
98  }
99 
100  BasicBlock::iterator ii(inst);
101  Value *Op0 = inst->getOperand(0);
102 
103  APInt::mu magics = Op1->getValue().magicu();
104 
105  IntegerType * type64 = IntegerType::get(Mod->getContext(), 64);
106 
107  Instruction *zext = CastInst::CreateZExtOrBitCast(Op0, type64, "", inst);
108  APInt m = APInt(64, magics.m.getZExtValue());
109  Constant *magicNum = ConstantInt::get(type64, m);
110 
111  Instruction *magInst = BinaryOperator::CreateNSWMul(zext, magicNum, "", inst);
112 
113  if (!magics.a) {
114  APInt apS = APInt(64, 32 + magics.s);
115  Constant *magicShift = ConstantInt::get(type64, apS);
116  Instruction *bigResult = BinaryOperator::Create(Instruction::LShr, magInst,
117  magicShift, "", inst);
118  Instruction *result = CastInst::CreateTruncOrBitCast(bigResult,
119  inst->getType(), "");
120 
121  ReplaceInstWithInst(inst->getParent()->getInstList(), ii, result);
122  } else {
123  APInt ap = APInt(64, 32);
124  Constant *movHiConst = ConstantInt::get(type64, ap);
125  Instruction *movHi = BinaryOperator::Create(Instruction::LShr, magInst,
126  movHiConst, "", inst);
127  Instruction *trunc = CastInst::CreateTruncOrBitCast(movHi, inst->getType(),
128  "", inst);
129  Instruction *sub = BinaryOperator::Create(Instruction::Sub, Op0, trunc, "",
130  inst);
131  APInt ap1 = APInt(32, 1);
132  Constant *one = ConstantInt::get(inst->getType(), ap1);
133  Instruction *shift = BinaryOperator::Create(Instruction::LShr, sub, one, "",
134  inst);
135  Instruction *add = BinaryOperator::Create(Instruction::Add, shift, trunc,
136  "", inst);
137  APInt apShr = APInt(32, magics.s - 1);
138  Constant *shr = ConstantInt::get(inst->getType(), apShr);
139  Instruction *result = BinaryOperator::Create(Instruction::LShr, add,
140  shr, "");
141 
142  ReplaceInstWithInst(inst->getParent()->getInstList(), ii, result);
143  }
144  return true;
145 }

147 virtual bool runOnFunction(Function &F) {
148  bool isChanged = false;
149  for (Function::iterator BB = F.begin(), EE = F.end(); BB != EE; ++BB) {
150  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;) {
151  Instruction *instr = I;
152  ++I;
153  if (instr->getOpcode() == Instruction::UDiv) {
154  isChanged |= lowerUDiv(instr);
155  } else if (instr->getOpcode() == Instruction::SDiv) {
156  isChanged |= lowerSDiv(instr);
157  }
158  }
159  }
160  return isChanged;
161 }
