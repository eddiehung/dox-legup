94 // Given a LiveBlocks map which contains a set of live blocks for each
95 // instruction, convert this to a map which maps every instruction to a set of
96 // other instructions with which it is independent Note, in addition to
97 // considering only live blocks, states in the FSM are also considered when
98 // determining independence
99 //
100 // @brief: this function fills the IndependentInstructions map
101 void Binding::FindIndependentInstructionsUsingLiveBlocks
102 (
103  std::map<Instruction*, std::set<BasicBlock*> > &LiveBlocks,
104  std::map<Instruction*, std::set<Instruction*> > &IndependentInstructions,
105  LiveVariableAnalysis* LVA,
106  FiniteStateMachine *fsm
107 )
108 {
109  // We need to consider every instruction with every other instruction, i.e.
110  // every pair of instructions. So, iterate over all instructions, and in
111  // each iteration iterate over all instructions again, but starting from
112  // the next iteration to avoid repetition
113 
114  if (LEGUP_CONFIG->getParameterInt("DISABLE_REG_SHARING")) {
115  return;
116  }
117 
118  for (std::map<Instruction*, std::set<BasicBlock*> >::iterator i =
119  LiveBlocks.begin(), end = LiveBlocks.end(); i != end; ++i) {
120 
121  for (std::map<Instruction*, std::set<BasicBlock*> >::iterator j = i; j
122  != end; ++j) {
123 
124  Instruction *i1 = i->first, *i2 = j->first;
125 
126  if (IndependentInstructions[i1].find(i2) !=
127  IndependentInstructions[i1].end() ) {
128  continue;
129  }
130 
131  // only pair if neither or both will be optimized away
132  if ( is_register_optimized_away(i1, fsm->getEndState(i1), fsm) !=
133  is_register_optimized_away(i2, fsm->getEndState(i2), fsm) ) {
134  continue;
135  }
136 
137  std::set<BasicBlock*> Intersection;
138 
139  std::set_intersection(i->second.begin(), i->second.end(),
140  j->second.begin(), j->second.end(),
141  std::inserter(Intersection, Intersection.begin()) );
142 
143  if (Intersection.empty()) { // two instructions are independent
144  IndependentInstructions[i1].insert(i2);
145  IndependentInstructions[i2].insert(i1);
146  continue;
147  }
148 
149  if (FromIndependentStates(i1, i2, Intersection, LVA, fsm)) {
150  // two instructions are independent
151  IndependentInstructions[i1].insert(i2);
152  IndependentInstructions[i2].insert(i1);
153  }
154  }
155  }
156 }

159 // The Live Variable Analysis pass (LVA is a reference to this pass) calculated
160 // live blocks for every instruction. This function uses this analysis,
161 // taking a set of Instructions (Instructions) as input and returning a map
162 // from every instruction to a set of basic blocks it is live in (LiveBlocks)
163 // @brief: this function fills the LiveBlocks map
164 void Binding::FindLiveBlocks
165 (
166  std::set<Instruction*> &Instructions,
167  std::map<Instruction*, std::set<BasicBlock*> > & LiveBlocks,
168  LiveVariableAnalysis *LVA
169 )
170 {
171  for (std::set<Instruction*>::iterator i = Instructions.begin(), ie =
172  Instructions.end(); i != ie; ++i) {
173  LiveBlocks.insert( std::make_pair(*i, std::set<BasicBlock*>()) );
174 
175  int index = LVA->instToLatticeBit[*i];
176 
177  for (LiveVariableAnalysis::BlockInfoMapping::iterator it =
178  LVA->blockToInfo.begin(); it != LVA->blockToInfo.end(); ++it) {
179  BasicBlock *bb = it->first;
180  LiveVariableAnalysis::BasicBlockLivenessInfo *info = it->second;
181  // instruction is live if it is a use, def, in or our of the block
182  if( (info->use->test(index))
183  || (info->def->test(index))
184  || (info->in->test(index))
185  || (info->out->test(index)) )
186  LiveBlocks[*i].insert( bb );
187  }
188  }

191 // Determines if the register assigned to an instruction will be optimized away.
192 // This happens if the instruction is not used across states. However, it is
193 // also possible that an instruction is used across states but its register is
194 // still optimized away if the use is a phi into another block and the phi is
195 // in a transition state of the instruction (see below)
196 bool is_register_optimized_away
197 (
198  Instruction *I,
199  State *state,
200  FiniteStateMachine *fsm
201 )
202 {
203  std::set<Instruction*> PHIs; // all Phi nodes using the instruction I
204  if (check_for_successors_in_other_states(I, state, fsm, PHIs)) {
205  return false; // can't optimize register away
206  }
207 
208  // Now, all successors of I are either PHIs or in the same state
209  if (PHIs.empty()) return true;
210 
211  // All successors are PHIs. However, if the PHIs are in a transition state
212  // of I, then the register will not be used
213  // Visit all transition states and remove all from the PHIs set.
214  visit_all_transition_states (state, PHIs);
215 
216  // Finally, if PHIs is empty now, then all the uses of I use the wire, not
217  // the reg. But if PHIs is not empty, at least one of its PHI successors
218  // is in a non-transition state
219  if (PHIs.empty()) return true;
220  return false;
221 }

223 // Checks for successors in other states and also filles the PHIs
224 // map with all PHI successors found
225 bool check_for_successors_in_other_states
226 (
227  Instruction *I,
228  State *state,
229  FiniteStateMachine *fsm,
230  std::set<Instruction*> &PHIs
231 )
232 {
233  // First, iterate over all successors. If they are all either PHI's or
234  // in the same state, keep iterating, else return true
235  for (Value::use_iterator i = I->use_begin(), e = I->use_end(); i!=e; ++i) {
236  Instruction *successor = dyn_cast<Instruction>(*i);
237  if (!successor) continue;
238  if (isa<PHINode>(*successor)) {
239  PHIs.insert(successor);
240  continue;
241  }
242  // Not a PHI, check if it's in the same state
243  if (fsm->getEndState(successor) != state) return true;
244  }
245  return false;
246 }

249 void visit_all_transition_states
250 (
251  State *state,
252  std::set<Instruction*> &PHIs
253 )
254 {
255  // visit default transition first
256  State *ts = state->getDefaultTransition();
257  for (State::iterator i = ts->begin(), e = ts->end(); i != e; ++i) {
258  if (!isa<PHINode>(*(*i))) break;
259  PHIs.erase(*i);

261 
262  if (state->getNumTransitions() > 1) {
263  for (unsigned t = 0, te = state->getNumTransitions()-1; t != te; ++t) {
264  State *ts = state->getTransitionState(t);
265  for (State::iterator i = ts->begin(), e = ts->end(); i != e; ++i) {
266  if (!isa<PHINode>(*(*i))) break;
267  PHIs.erase(*i);
268  }
269  }
270  }

388 // Calculate the order of a state within a basic block:
389 // State: LEGUP_function_call_58 order: 0
390 // State: LEGUP_function_call_59 order: 1
391 // State: LEGUP_F_main_BB_49_60 order: 2
392 // State: LEGUP_F_main_BB_49_61 order: 3
393 // State: LEGUP_F_main_BB_49_62 order: 4
394 // State: LEGUP_F_main_BB_49_63 order: 5
395 // State: LEGUP_F_main_BB_49_64 order: 6
396 // State: LEGUP_F_main_BB_49_65 order: 7
397 // State: LEGUP_function_call_66 order: 8
398 // State: LEGUP_function_call_67 order: 9
399 // State: LEGUP_F_main_BB_49_68 order: 10
400 // last state for BB %49 would be set to 10
401 void Binding::getBBStateEncodings(Function *Fp,
402  FiniteStateMachine *fsm,
403  std::map<State*, int> &StateEncoding,
404  std::map<BasicBlock*, int> &lastState) {
405 
406  for (Function::iterator BB = Fp->begin(), be = Fp->end(); BB != be; ++BB) {
407  unsigned order=0;
408 
409  State *state = NULL;
410  for (FiniteStateMachine::iterator stateIter = fsm->begin(); stateIter
411  != fsm->end(); stateIter++) {
412  if (stateIter->getBasicBlock() == BB) {
413  // will handle cases where the basic block is empty
414  state = stateIter;
415  break;
416  }
417  for (State::iterator instr = stateIter->begin(), ie =
418  stateIter->end(); instr != ie; ++instr) {
419  Instruction *I = *instr;
420  if (I->getParent() == BB) {
421  state = stateIter;
422  break;
423  }
424  }
425  if (state) break;
426  }
427  assert(state);
428 
429  while (state) {
430  if ( StateEncoding.find(state) == StateEncoding.end() ) {
431  StateEncoding[state] = order;
432  order++;
433  }
434  //errs() << "State: " << state->getName() << " order: " <<
435  // StateEncoding[state] << "\n";
436  if (state->isTerminating()) break;
437 
438  if (state == state->getDefaultTransition()) {
439  // special case for function calls
440  assert(state->getNumTransitions() == 2);
441  state = state->getTransitionState(0);
442  } else {
443  state = state->getDefaultTransition();
444  }
445  }
446  lastState[BB] = order-1;
447  }
448 }
