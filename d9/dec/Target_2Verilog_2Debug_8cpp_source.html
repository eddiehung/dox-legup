707  getline(fs, s);
708  regsFillRate = atof(s.c_str());
709  fs.close();
710  } else if (optionSizeBufsStaticAnalysis) {
711  // Buffer allocation method: Static analysis
712  ctrlFillRate = 0.21;
713  memFillRate = numWrites / (float)numStates;
714  regsFillRate =
715  traceScheduler->getNumStatesWithTracing() / (float)numStates;
716  } else {
717  // Buffer allocation method: Constant factor
718  ctrlFillRate = 0.21;
719  memFillRate = 0.11;
720 
721  if (optionRegBufferDualPorted && optionTraceRegsDelayAll &&
722  optionTraceRegsDelayAll) {
723  regsFillRate = 0.34;
724  } else if ((!optionRegBufferDualPorted) && optionTraceRegsDelayAll &&
725  optionTraceRegsDelayWorst) {
726  regsFillRate = 0.32;
727  } else if ((!optionRegBufferDualPorted) && (!optionTraceRegsDelayAll) &&
728  optionTraceRegsDelayWorst) {
729  regsFillRate = 0.34;
730  } else if ((!optionRegBufferDualPorted) && (!optionTraceRegsDelayAll) &&
731  (!optionTraceRegsDelayWorst)) {
732  regsFillRate = 0.20;
733  } else {
734  assert(false);
735  }
736  }
737 
738  traceCtrlWidth = instanceBits + stateBits + optionCtrlSequenceBits;
739 
740  // 2 = number of bits for memory size signal
741  traceMemWidth = 2 + alloc->getDataLayout()->getPointerSizeInBits() +
742  alloc->getDataSize();
743 
744  if (optionTraceRegs) {
745  // We want to solve this equation:
746  // ControlWidth * ControlDepth + MemWidth * MemDepth + RegsWidth *
747  // RegsDepth = Total Mem Bits
748  // Where: MemDepth / CtrlDepth = memFillRate / ctrlFillRate
749  // and: RegsDepth / CtrlDepth = regsFillRate / ctrlFillRate
750 
751  traceCtrlDepth =
752  (ctrlFillRate * optionTraceMemBits) /
753  (ctrlFillRate * traceCtrlWidth + memFillRate * traceMemWidth +
754  regsFillRate * traceRegsWidth);
755  traceMemDepth = (traceCtrlDepth * memFillRate / ctrlFillRate);
756  traceRegsDepth = (traceCtrlDepth * regsFillRate / ctrlFillRate);
757 
758  if (!traceRegsDepth)
759  traceRegsDepth = 1;
760  } else {
761  // We want to solve this equation:
762  // ControlWidth * ControlDepth + MemWidth * MemDepth = Total Mem
763  // Bits
764  // Where: MemDepth / CtrlDepth = memFillRate / ctrlFillRate
765 
766  traceCtrlDepth =
767  (ctrlFillRate * optionTraceMemBits) /
768  (ctrlFillRate * traceCtrlWidth + memFillRate * traceMemWidth);
769  traceMemDepth = (traceCtrlDepth * memFillRate / ctrlFillRate);
770  traceRegsDepth = 0;
771  }
