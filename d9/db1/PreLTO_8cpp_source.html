34 struct LegUp : public FunctionPass {
35  static char ID; // Pass identification, replacement for typeid
36  LegUp() : FunctionPass(ID) {}
37 
38  virtual bool doInitialization(Module &M) {
39  Mod = &M;
40 
41  TD = new DataLayout(&M);
42 
43  IL = new IntrinsicLowering(*TD);
44  return false;
45  }
46 
47  virtual bool runOnFunction(Function &F) {
48  bool modified = false;
49 
50  // Examine all the instructions in this function to find the intrinsics
51  // that need to be lowered.
52  for (Function::iterator BB = F.begin(), EE = F.end(); BB != EE; ++BB) {
53  bool localChange = true;
54 
55  // the below functions can modify instruction CI invalidating the
56  // iterator so we need to keep checking the BB until there are no
57  // more changes
58  while (localChange) {
59  localChange = false;
60  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I !=
61  E; ++I) {
62  if (CallInst *CI = dyn_cast<CallInst>(I)) {
63 
64  Function *calledFunction = CI->getCalledFunction();
65  // ignore indirect function calls
66  if (!calledFunction) continue;
67 
68  localChange = lowerIfIntrinsic(CI, calledFunction);
69 
70  // recheck the BB again
71  if (localChange) {
72  modified = true;
73  break;
74  }
75  }
76  }
77  }
78  }
79 
80  return modified;
81  }

103 /// This is used to keep track of intrinsics that get generated to a lowered
104 /// function. We must generate the prototypes before the function body which
105 /// will only be expanded on first use (by the loop below).
106 /// Note: This function was adapted from lowerIntrinsics() in CBackend.cpp
107 bool LegUp::lowerIfIntrinsic(CallInst *CI, Function *calledFunction) {
108 
109  switch (calledFunction->getIntrinsicID()) {
110  case Intrinsic::not_intrinsic:
111  //case Intrinsic::memory_barrier: # replaced by fence atomic
112  case Intrinsic::vastart:
113  case Intrinsic::vacopy:
114  case Intrinsic::vaend:
115  case Intrinsic::returnaddress:
116  case Intrinsic::frameaddress:
117  case Intrinsic::setjmp:
118  case Intrinsic::longjmp:
119  case Intrinsic::prefetch:
120  case Intrinsic::powi:
121  case Intrinsic::x86_sse_cmp_ss:
122  case Intrinsic::x86_sse_cmp_ps:
123  case Intrinsic::x86_sse2_cmp_sd:
124  case Intrinsic::x86_sse2_cmp_pd:
125  case Intrinsic::ppc_altivec_lvsl:
126  // We directly implement these intrinsics
127  return false;
128 
129  case Intrinsic::dbg_declare:
130  case Intrinsic::dbg_value:
131  // This is just debug info - it doesn't affect program
132  // behavior
133  return false;
134 
135  case Intrinsic::memcpy:
136  case Intrinsic::memmove:
137  case Intrinsic::memset:
138  return lowerLegupInstrinsic(CI, calledFunction);
139 
140  case Intrinsic::uadd_with_overflow:
141  return lowerOverflowIntrinsic(CI, calledFunction);
142 
143  default:
144 
145  // All other intrinsic calls we must lower.
146  DEBUG(errs() << "Lowering: " << *CI << "\n");
147  IL->LowerIntrinsicCall(CI);
148 
149  return true;
150  }
151 
152 }

154 // determine the type of the memcpy, memmove, memset destination pointer
155 const Type *get_dest_ptr_type(CallInst *CI) {
156  const Value *destOp = CI->getOperand(0);
157  PointerType *destPtr;
158  if (const BitCastInst *bci = dyn_cast<BitCastInst>(destOp)) {
159  destPtr = dyn_cast<PointerType>(bci->getSrcTy());
160  } else if (const GetElementPtrInst *gep = dyn_cast<GetElementPtrInst>(
161  destOp)) {
162  destPtr = dyn_cast<PointerType>(gep->getPointerOperand()->getType());
163  } else if (PointerType *p = dyn_cast<PointerType>(destOp->getType())) {
164  destPtr = p;
165  } else {
166  errs() << "CallInst: " << *CI << "\n";
167  errs() << "Dest Pointer: " << *destOp << "\n";
168  llvm_unreachable("Unknown pointer destination in intrinsic argument");
169  }
170 
171  const Type *destType = destPtr->getElementType();
172  while (const ArrayType *at = dyn_cast<ArrayType>(destType)) {
173  destType = at->getElementType();
174  }
175  return destType;
176 }

178 std::string LegUp::getIntrinsicMemoryAlignment(CallInst *CI) {
179 
180  std::string legupFunctionAlign;
181 
182  const Type *destType = get_dest_ptr_type(CI);
183 
184  // use modified struct size if first argument is a struct pointer
185  if (isa<StructType>(destType)) {
186  // Get the alignment, and decide which memcpy to use
187 
188  // the following code would always work if clang supported the
189  // -malign-double flag
190  // we need the above code to handle 64-bit aligned accesses on a 32-bit
191  // machine
192  ConstantInt *align = dyn_cast<ConstantInt>(CI->getOperand(3));
193  for (int i = 8; i > 0; i >>= 1) {
194  if (align->equalsInt(i)) {
195  legupFunctionAlign = utostr(i);
196  }
197  }
198  } else if (const IntegerType *it = dyn_cast<IntegerType>(destType)) {
199  legupFunctionAlign = utostr(it->getBitWidth() / 8);
200  } else if (isa<PointerType>(destType)) {
201  // pointers are 32 bits
202  legupFunctionAlign = "4";
203  } else if (destType->isFloatTy()) {
204  // floats are 32 bits
205  legupFunctionAlign = "4";
206  } else if (destType->isDoubleTy()) {
207  // double are 64 bits
208  legupFunctionAlign = "8";
209  } else {
210  errs() << "CallInst: " << *CI << "\n";
211  llvm_unreachable("unknown destination pointer type");
212  }
213 
214  return legupFunctionAlign;
215 }

217 void LegUp::createOverflowSumCarry(CallInst *CI, Instruction* &sum,
218  Instruction* &carry) {
219  Value *op0 = CI->getArgOperand(0);
220  Value *op1 = CI->getArgOperand(1);
221  unsigned size = op0->getType()->getPrimitiveSizeInBits();
222  unsigned newSize = size + 1;
223 
224  IntegerType * newType = IntegerType::get(Mod->getContext(),
225  newSize);
226 
227  string name = "overflow_intrinsic";
228  // assume llvm.uadd.* so we zero extend
229  Instruction *zextOp0 = CastInst::CreateZExtOrBitCast(op0, newType,
230  name, CI);
231  Instruction *zextOp1 = CastInst::CreateZExtOrBitCast(op1, newType,
232  name, CI);
233 
234  Instruction *add = BinaryOperator::Create(Instruction::Add, zextOp0,
235  zextOp1, name, CI);
236 
237  sum = CastInst::CreateTruncOrBitCast(add, op0->getType(),
238  name + "_sum", CI);
239 
240  APInt ap1 = APInt(newSize, size);
241  Constant *one = ConstantInt::get(newType, ap1);
242  Instruction *shift = BinaryOperator::Create(Instruction::LShr, add,
243  one, name, CI);
244 
245  IntegerType * sizeOne = IntegerType::get(Mod->getContext(), 1);
246  carry = CastInst::CreateTruncOrBitCast(shift, sizeOne,
247  name + "_carry", CI);
248 
249 }

251 void LegUp::replaceOverflowIntrinsic(CallInst *CI, Instruction *sum,
252  Instruction *carry) {
253 
254 
255  // find sum/carry extractvalue
256  // sum: %108 = extractvalue %0 %uadd.i, 0
257  // carry: %109 = extractvalue %0 %uadd.i, 1
258  vector<Instruction*> dead;
259  for (Value::user_iterator i = CI->user_begin(), e = CI->user_end(); i !=
260  e; ++i) {
261  if (ExtractValueInst *EV = dyn_cast<ExtractValueInst>(*i)) {
262  Instruction *replace = NULL;
263  // sum
264  if (*EV->idx_begin() == 0) {
265  replace = sum;
266  // carry
267  } else {
268  assert(*EV->idx_begin() == 1);
269  replace = carry;
270  }
271  DEBUG(errs() << "Replacing " << *EV << "\n");
272  DEBUG(errs() << "With " << *replace << "\n");
273  EV->replaceAllUsesWith(replace);
274  dead.push_back(EV);
275 
276  } else {
277  errs() << **i << "\n";
278  llvm_unreachable("Expecting extractvalue for overflow intrinsic\n");
279  }
280  }
281 
282  // remove dead extractvalue instructions
283  for (vector<Instruction*>::iterator i = dead.begin(), e =
284  dead.end(); i != e; ++i) {
285  (*i)->eraseFromParent();
286  }
287  CI->eraseFromParent();
288 }

291 // Handle uadd.with.overflow.* intrinsics. For example:
292 // %uadd.i = call %0 @llvm.uadd.with.overflow.i64(i64 %105, i64 %106)
293 // %108 = extractvalue %0 %uadd.i, 0
294 // %109 = extractvalue %0 %uadd.i, 1
295 // Replace this n-bit addition with a (n + 1) bit addition and shift out the
296 // carry bit manually
297 bool LegUp::lowerOverflowIntrinsic(CallInst *CI, Function *calledFunction) {
298 
299  DEBUG(errs() << "Lowering overflow intrinsic: " << *CI << "\n");
300 
301 
302  Instruction *sum = 0, *carry = 0;
303 
304  createOverflowSumCarry(CI, sum, carry);
305 
306  replaceOverflowIntrinsic(CI, sum, carry);
307 
308  return true;
309 }

312 // handle memcpy, memmove, memset by replacing with legup specific functions
313 bool LegUp::lowerLegupInstrinsic(CallInst *CI, Function *calledFunction) {
314 
315  DEBUG(errs() << "Lowering for LegUp: " << *CI << "\n");
316 
317  std::string legupFunction;
318  switch (calledFunction->getIntrinsicID()) {
319  case Intrinsic::memcpy:
320  legupFunction = "legup_memcpy";
321  break;
322  case Intrinsic::memmove:
323  legupFunction = "legup_memmove";
324  break;
325  case Intrinsic::memset:
326  legupFunction = "legup_memset";
327  break;
328  }
329 
330  std::string fullLegupFunction =
331  legupFunction + "_" + getIntrinsicMemoryAlignment(CI);
332 
333  // we only need the first 3 operands
334  vector<Value *> Ops;
335  Ops.push_back(CI->getOperand(0));
336  Ops.push_back(CI->getOperand(1));
337  Ops.push_back(CI->getOperand(2));
338 
339  if (CI->getOperand(2)->getType()->isIntegerTy(64))
340  fullLegupFunction += "_i64";
341 
342  DEBUG(errs() << "Replacing calls with: " << fullLegupFunction << "\n");
343  ReplaceCallWith(fullLegupFunction.c_str(), CI, Ops,
344  calledFunction->getReturnType());
345 
346  return true;
347 }
