45 /// RTLWidth - Represents the bitwidth of a RTLSignal i.e. [5:2]
46 /// @brief RTLWidth
47 class RTLWidth {
48 public:
49  RTLWidth() {
50  nativeHi = hi = "";
51  nativeLo = lo = "0";
52  isSigned = false;
53  }
54 
55  // shortcut for LLVM instructions
56  RTLWidth(const Type* T, std::string lo="0") : hi(getMSBIndex(T)), lo(lo),
57  nativeHi(getMSBIndex(T)), nativeLo(lo), newStyleRTLWidth(false) {}
58  RTLWidth(const Value *I, MinimizeBitwidth *MBW);
59  RTLWidth(unsigned minW, unsigned nativeW, bool isSigned) : lo("0"), nativeLo("0"),
60  isSigned(isSigned) {
61  if(minW==1) hi = "";
62  else hi = utostr(minW-1);
63  if(nativeW==1) nativeHi = "";
64  else nativeHi = utostr(nativeW-1);
65  }
66 
67  // Standard width. ie. size=32 will give 31:0
68  RTLWidth(unsigned size) : lo("0"), nativeLo("0"), newStyleRTLWidth(false) {
69  assert(size != 0);
70  if (size == 1) {
71  hi = "";
72  } else {
73  hi = utostr(size-1);
74  }
75  nativeHi = hi;
76  }
77  // This can be used to create non-standard widths. ie. 6:3
78  // Or it can also be used for parameterized widths
79  // ie. `MEMORY_CONTROLLER_DATA_SIZE-1:0
80  RTLWidth(std::string hi, std::string lo="0") : hi(hi), lo(lo), nativeHi(hi), nativeLo(lo), newStyleRTLWidth(false) {}
81 
82  RTLWidth(unsigned hiIn, unsigned loIn) : lo(utostr(loIn)), nativeLo(utostr(loIn)), newStyleRTLWidth(false) {
83  if(hiIn==0) hi="";
84  else hi=utostr(hiIn);
85  nativeHi = hi;
86  }
87 
88  std::string str() const;
89  std::string getHi() const { return hi; };
90  std::string getLo() const { return lo; };
91  bool getSigned(){ return isSigned; };
92  void setSigned(bool sign){ isSigned=sign; };
93  bool isNewStyleRTLWidth(){ return newStyleRTLWidth; }
94 
95  // an integer for the number of bits
96  // we need the RTLModule and Allocation to handle global `define and
97  // parameterized widths
98  unsigned numBits(const RTLModule *rtl, const Allocation *alloc) const;
99  unsigned numNativeBits(const RTLModule *rtl, const Allocation *alloc) const;
100 
101  bool operator==(const RTLWidth &other) const;
102 
103  // Returns a vector of widths that doesn't include the bits
104  // from the argument width
105  // e.g. passing in a width of 5:2 with a "this" width of 7:0
106  // will return [1:0,7:6]
107  std::vector<RTLWidth> widthsFromExclusionWithWidth(const RTLWidth &width) const;
108 
109 private:
110  std::string hi, lo;
111  std::string nativeHi, nativeLo;
112  bool isSigned;
113  bool newStyleRTLWidth;
