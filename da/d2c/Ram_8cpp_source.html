183 // assign appropriate bits of 64-bit value
184 // for example ci is a 2-byte short:
185 // ci = 0xAABB
186 // ci->getZExtValue = 0x000000000000AABB (64-bits)
187 // wordAlignedByteOffset = 2
188 // offset 7 6 5 4 3 2 1 0
189 // val = 00 0F FF FF 00 00 00 00 (previous value)
190 // bitwise or val with:
191 // val |= 00 00 00 00 AA BB 00 00 (ci << (2 * 8))
192 // val = 00 0F FF FF AA BB 00 00
193 uint64_t RAM::getConstantVal(const Constant *c, unsigned offset, uint64_t val) {
194  const ConstantInt *ci = dyn_cast<ConstantInt>(c);
195 
196  //NC changes
197  if (LEGUP_CONFIG->getParameterInt("INSPECT_DEBUG")) {
198  insertDebugInitialValue(ci->getValue());
199  }
200  unsigned wordByteSize = 8;
201  unsigned wordAlignedByteOffset = offset % wordByteSize;
202 
203  unsigned intSize;
204  if (ci) {
205  intSize = ci->getBitWidth() / 8;
206  } else {
207  intSize = alloc->getDataLayout()->getPointerSize();
208  }
209 
210  // assert alignment
211  assert(wordAlignedByteOffset % intSize == 0);
212  // assert that we are not overwriting any values
213  assert(val >> (wordAlignedByteOffset * 8) == 0);
214 
215  if (ci) {
216  val |= ci->getZExtValue() << (wordAlignedByteOffset * 8);
217  } else {
218  val = alloc->getRamTagNum(dyn_cast<Value>(c));
219  val <<= (32 - alloc->getTagSize());
220  }
221  return val;
222 }

456 APInt RAM::getPointerFromGEP(const ConstantExpr *expr) {
457  if (!expr->isGEPWithNoNotionalOverIndexing()) {
458  llvm_unreachable(0);
459  }
460  uint64_t val = alloc->getRamTagNum(dyn_cast<Value>(expr->getOperand(0)));
461  val <<= (32 - alloc->getTagSize());
462  // Compute the offset that this GEP adds to the pointer.
463  SmallVector<Value*, 8> Indices(expr->op_begin()+1, expr->op_end());
464  uint64_t GEPOffset = alloc->getDataLayout()->getIndexedOffset(
465  expr->getOperand(0)->getType(), Indices);
466  return APInt(getDataWidth(), val + GEPOffset);
467 }

614 void RAM::getRAMTypeForArray(ArrayType *ATy) {
615  elements = ATy->getNumElements();
616 
617  // arrays can hold other arrays
618  while (ArrayType *ATy2 = dyn_cast<ArrayType>(ATy->getElementType())) {
619  elements = elements * ATy2->getNumElements();
620  ATy = ATy2;
621  }
622 
623  // calculate data width
624  Type *dataTy = ATy->getElementType();
625  DEBUG(errs() << "elements: " << elements << ", type: " << *dataTy << "\n");
626  if (const IntegerType *ITy = dyn_cast<IntegerType>(dataTy)) {
627  datawidth = ITy->getBitWidth();
628  } else if (dataTy->isFloatTy()){
629  datawidth = 4 * 8; //float type is 4 byte
630  } else if (dataTy->isDoubleTy()){
631  datawidth = 8 * 8; //double type is 8 byte
632  } else if (StructType *STy = dyn_cast<StructType>(dataTy)) {
633  IsStruct = true;
634  datawidth = 64;
635 
636  int structSize = getStructElements(STy);
637  elements *= structSize;
638  } else if (isa<PointerType>(dataTy)) {
639  datawidth = alloc->getDataLayout()->getPointerSizeInBits();
640  } else {
641  llvm_unreachable("Unsupported Array Type");
642  }
643 
644  // remember that we can ignore the 0th element
645  // ie. 16 elements will require 4 bits (not 5 bits required to store the
646  // _number_ 16)
647  addresswidth = requiredBits(elements-1);
648 }

762 // return the latency in cycles of the RAM
763 // this is set in the tcl file by:
764 // set_operation_latency mem_dual_port 2
765 // set_operation_latency local_mem_dual_port 1
766 // TODO: be able to set the latency of a specific local ram
767 int RAM::getLatency(Allocation *alloc) {
768  RAM *R = this;
769  int latency = 1;
770  if (LEGUP_CONFIG->getParameterInt("LOCAL_RAMS")) {
771  std::string FuName = "mem_dual_port";
772  if (alloc->isGlobalRams.find(R) == alloc->isGlobalRams.end()) {
773  FuName = "local_mem_dual_port";
774  }
775  int constraint;
776  //errs() << "FuName: " << FuName << "\n";
777  assert(LEGUP_CONFIG->getOperationLatency(FuName, &constraint));
778  if (alloc->isGlobalRams.find(R) == alloc->isGlobalRams.end()) {
779  // local ram
780  latency = constraint;
781 
782  // If these local RAMs can be sources of multi-cycle paths, each load
783  // will need a unique output register. This can be done by reducing the
784  // latency of the RAMs by 1.
785  if (LEGUP_CONFIG->duplicate_load_reg()) {
786  if (latency < 2) {
787  errs() << "\nRAM latency must be >1 when pulling out load registers";
788  assert(0);
789  }
790  --latency;
791  }
792  } else {
793  // global RAM take 1 off latency
794  // (1 cycle is taken by the memory controller)
795  latency = constraint-1;
796  }
797  }
798  return latency;
799 }

846 // allocate the individual arrays/RAMs assigned to this physical RAM
847 // at specific address offsets in the physical RAM
848 void PhysicalRAM::staticMemoryAllocation(formatted_raw_ostream &out) {
849  out << "Running static memory allocation for grouped physical ram: "
850  << getName() << "\n";
851 
852  unsigned datawidth = getDataWidth();
853  unsigned bytes = (datawidth > 8) ? datawidth / 8 : 1;
854 
855  // sort memory arrays in descending order by address alignment, then
856  // descending by array size
857  std::sort(ramList.begin(), ramList.end(), memAllocSort);
858 
859  // list of the RAMs in actual order placed in the RAM
860  std::list<RAM *> allocation;
861 
862  struct HOLE {
863  unsigned start;
864  unsigned end;
865  HOLE(unsigned s, unsigned e) : start(s), end(e) {}
866  };
867 
868  // list of the available "holes" or unused space in actual order of
869  // the RAM
870  std::list<HOLE> holes;
871  // available space: 0 -> 2^23-1
872  HOLE init = HOLE(0, 8388607);
873  holes.push_back(init);
874 
875  for (PhysicalRAM::ram_iterator r = ram_begin(), re = ram_end(); r != re;
876  ++r) {
877  RAM *R = *r;
878 
879  unsigned size = R->getElements() * bytes;
880  unsigned alignment = R->getAddressAlignment();
881  // if (!R->isROM() && datawidth == 32) alignment = alignment*2;
882 
883  unsigned arrayStart = 0;
884  bool found = false;
885  std::list<HOLE>::iterator i = holes.begin();
886 
887  // greedily place at first available hole (unused space) in the memory
888  // that meets address alignment constraint
889  // i.e. for an alignment of 4096 we can only place the RAM
890  // at offsets of 0, 4096, 8192, 12288, ...
891  while (!found) {
892  HOLE &hole = *i;
893  while (hole.start > arrayStart) {
894  arrayStart += alignment;
895  }
896  unsigned arrayEnd = arrayStart + size - 1;
897  if (arrayEnd <= hole.end) {
898  // we can place this RAM in this available hole
899  ramOffset[R] = arrayStart;
900  out << "Placing RAM: " << R->getName() << " at: " << arrayStart
901  << "\n";
902  // split hole
903  int start1 = hole.start;
904  int end1 = arrayStart - 1;
905  int start2 = end1 + 1 + size;
906  int end2 = hole.end;
907  // remove the current hole and replace with new ones...
908  i = holes.erase(i);
909  if (start1 <= end1) {
910  holes.insert(i, HOLE(start1, end1));
911  }
912  if (start2 <= end2) {
913  holes.insert(i, HOLE(start2, end2));
914  }
915  found = true;
916  }
917  ++i;
918  }
919 
920  out << "Holes after allocating ram: " << R->getName()
921  << " size: " << size << " alignment: " << alignment
922  << " offset: " << ramOffset[R] << "\n";
923  for (std::list<HOLE>::iterator i = holes.begin(), ie = holes.end();
924  i != ie; ++i) {
925  HOLE &hole = *i;
926  assert(hole.end >= hole.start);
927  out << "Hole: " << hole.start << " - " << hole.end
928  << " size = " << (hole.end - hole.start + 1) << "\n";
929  }
930  }
931 
932  // re-sort by offset
933  offsetSortObj.ramOffset = &ramOffset;
934  std::sort(ramList.begin(), ramList.end(), offsetSortObj);
935 
936  out << "Final memory allocation for physical ram: " << getName() << "\n";
937  char buf[200];
938  unsigned prevOffset = 0;
939  unsigned prevSize = 0;
940  unsigned totalUnused = 0;
941  unsigned totalMemory = 0;
942  for (PhysicalRAM::ram_iterator r = ram_begin(), re = ram_end(); r != re;
943  ++r) {
944  RAM *R = *r;
945  unsigned datawidth = R->getDataWidth();
946  unsigned bytes = (datawidth > 8) ? datawidth / 8 : 1;
947  unsigned size = R->getElements() * bytes;
948 
949  unsigned alignment = R->getAddressAlignment();
950  unsigned offset = ramOffset[R];
951  // offset should be word-aligned
952  assert(offset % bytes == 0);
953  unsigned unused = offset - prevOffset - prevSize;
954  totalMemory = offset + size;
955  totalUnused += unused;
956  prevOffset = offset;
957  prevSize = size;
958  snprintf(buf, 200, "ram: %-40s\t"
959  "size (bytes): %-5d (hex:0x%-5x)\t"
960  "alignment (bytes): %-5d (hex:0x%-5x) "
961  "offset (bytes): %-5d (hex:0x%-5x) "
962  "unused (bytes): %-5d\n",
963  R->getName().c_str(), size, size, alignment, alignment, offset,
964  offset, unused);
965  out << buf;
966 
967  // sanity check...all possible addresses should not overlap with offset
968  for (unsigned i = 0; i < size - 1; i++) {
969  assert((offset + i) == (offset | i));
970  /*
971  unsigned ADD = (offset + i);
972  unsigned OR = (offset | i);
973  if (ADD != OR) {
974  out << "Error: i = " << i << " " << ADD << " != " << OR << "\n";
975  }
976  */
977  }
978  }
979 
980  out << "Total Unused (B): " << totalUnused << "\n";
981  out << "Total Memory (B): " << totalMemory << "\n";
982  double frag = (double)totalUnused / totalMemory;
983  int fragP = frag * 100;
984  out << "Fragmentation Ratio (Total Unused / Total Memory): " << frag << " ("
985  << fragP << "%)\n";
986 
987  numwords = totalMemory / bytes;
988 }
