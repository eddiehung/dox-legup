244 // select time where operation should be scheduled
245 int SDCModuloScheduler::findTimeSlot(Instruction *I, int minTime, int maxTime) {
246  int curTime = minTime;
247  int schedSlot = 0;
248  bool found = false;
249  while (!found && curTime <= maxTime) {
250  if (resourceConflict(I, curTime)) {
251  File() << "Resource conflict at time " << curTime
252  << ". Incrementing time slot\n";
253  // try the next time slot
254  curTime++;
255  } else {
256  // no resource conflicts. Select this time slot.
257  // Dependence constraints due to predecessors were honoured in the
258  // computation of minTime
259  schedSlot = curTime;
260  found = true;
261  }
262  }
263 
264  // if a legal slot was not found, then pick the first option from the
265  // following:
266  // 1) minTime - if this is the first time the operation is being scheduled
267  // or if minTime is greater than the time the operation was last scheduled
268  // 2) previously scheduled time + 1
269  if (!found) {
270  File() << "No legal slot found\n";
271  if (neverScheduled[I] || minTime > prevSchedTime[I]) {
272  File() << "Forcing to minTime\n";
273  schedSlot = minTime;
274  } else {
275  File() << "Forcing to prev sched time + 1\n";
276  schedSlot = prevSchedTime[I] + 1;
277  }
278  }
279 
280  return schedSlot;
281 }

360 // schedule operation at time timeSlot. Displace previous scheduled
361 // operations that conflict with it either due to resource conflicts or
362 // dependence constraints
363 void SDCModuloScheduler::schedule(Instruction *I, int timeSlot) {
364 
365  moduloScheduler.schedTime[I] = timeSlot;
366  std::string FuName =
367  LEGUP_CONFIG->getOpNameFromInst(I, moduloScheduler.alloc);
368 
369  if (moduloScheduler.isResourceConstrained(I)) {
370  int moduloTimeSlot = timeSlot % moduloScheduler.II;
371 
372  bool resourceConflict = true;
373  for (int i = 0; i < numIssueSlots(FuName); i++) {
374  // the instruction shouldn't already be in the table
375  assert(
376  I !=
377  moduloScheduler.getReservationTable(FuName, i, moduloTimeSlot));
378  // find an empty slot
379  if (moduloScheduler.getReservationTable(FuName, i,
380  moduloTimeSlot) == NULL) {
381  moduloScheduler.setReservationTable(FuName, i, moduloTimeSlot,
382  I);
383  resourceConflict = false;
384  break;
385  }
386  }
387 
388  if (resourceConflict) {
389  // unschedule *all* potential resource conflicts
390  for (int i = 0; i < numIssueSlots(FuName); i++) {
391  Instruction *prev = moduloScheduler.getReservationTable(
392  FuName, i, moduloTimeSlot);
393  assert(I != prev);
394  if (prev) {
395  File() << "Resource conflict\n";
396  unschedule(prev);
397  }
398  }
399  // use the first issue slot (arbitrary)
400  moduloScheduler.setReservationTable(FuName, 0, moduloTimeSlot, I);
401  }
402  }
403 
404  // loop over successors and displace previous scheduled operations that
405  // have a dependency conflict
406  for (Value::use_iterator use = I->use_begin(), e = I->use_end(); use != e;
407  ++use) {
408  Instruction *succ = dyn_cast<Instruction>(*use);
409  if (!succ)
410  continue;
411 
412  // successor has been scheduled
413  if (moduloScheduler.schedTime.find(succ) !=
414  moduloScheduler.schedTime.end()) {
415  int min = moduloScheduler.schedTime[I] + moduloScheduler.delay(I) -
416  moduloScheduler.II * distance(I, succ);
417  if (moduloScheduler.schedTime[succ] < min) {
418  assert(succ != I);
419  File() << "Conflict with successor\n";
420  unschedule(succ);
421  }
422  }
423  }
424 
425  InstructionNode *op = dag->getInstructionNode(I);
426  for (InstructionNode::iterator use = op->mem_use_begin(),
427  e = op->mem_use_end();
428  use != e; ++use) {
429  Instruction *succ = (*use)->getInst();
430  assert(succ);
431 
432  // successor has been scheduled
433  if (moduloScheduler.schedTime.find(succ) !=
434  moduloScheduler.schedTime.end()) {
435  int min = moduloScheduler.schedTime[I] + moduloScheduler.delay(I) -
436  moduloScheduler.II * distance(I, succ);
437  if (moduloScheduler.schedTime[succ] < min) {
438  assert(succ != I);
439  File() << "Conflict with successor\n";
440  unschedule(succ);
441  }
442  }
443  }
444 
445  neverScheduled[I] = false;
446  unscheduledInsts.erase(I);
447  prevSchedTime[I] = timeSlot;
448 }
