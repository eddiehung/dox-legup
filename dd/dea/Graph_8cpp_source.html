22 // Return the string corresponding to this instruction for the
23 // graph. Alternatively, alloc->getConfig->getOpnameFromInst can be used.
24 string get_instruction_string_for_graph (const Instruction * I) {
25  string op_name = I->getOpcodeName();
26 
27  // If the instruction is icmp, concatenate compare condition to the end
28  // (e.g. icmp -> icmpeq)
29  if (const ICmpInst *cmp = dyn_cast<ICmpInst>(I)) {
30  switch (cmp->getPredicate()) {
31  case ICmpInst::ICMP_EQ: op_name += "eq"; break;
32  case ICmpInst::ICMP_NE: op_name += "ne"; break;
33  case ICmpInst::ICMP_SLT: op_name += "slt"; break;
34  case ICmpInst::ICMP_ULT: op_name += "ult"; break;
35  case ICmpInst::ICMP_SLE: op_name += "sle"; break;
36  case ICmpInst::ICMP_ULE: op_name += "ule"; break;
37  case ICmpInst::ICMP_SGT: op_name += "sgt"; break;
38  case ICmpInst::ICMP_UGT: op_name += "ugt"; break;
39  case ICmpInst::ICMP_SGE: op_name += "sge"; break;
40  case ICmpInst::ICMP_UGE: op_name += "uge"; break;
41  default: ;
42  }
43  }
44 
45  // Both operands should have same type, concatenate it to operation name,
46  // e.g. change from mul to muli32, or icmpeq from above to icmpeqi64
47  op_name += "i" + legup::IntToString( I->getOperand(0)->getType()->getPrimitiveSizeInBits() );
48 
49  return op_name;
50 }

56 // Create, initialize and return a new node given an instruction
57 Node * Graph::NewNode(Instruction * _I) {
58  N++;
59  Node *temp = new Node (
60  /* I= */ _I,
61  /* op= */ get_instruction_string_for_graph(_I),
62  /* label= */ 0
63  );
64 
65  // Create nodes for the two predecessors. The left and right
66  // predecessors which are either "constants" or "inputs".
67 
68  // Check if the left predecessor of I is a constant or an input
69  if ( isa<ConstantInt>( temp->I->getOperand(0) )) {
70 
71  // prepend "POS" or "NEG" to the const name
72  string constantName;
73  int value = cast<ConstantInt>(temp->I->getOperand(0))->getZExtValue();
74  if (value<0) constantName = "NEG" + IntToString( value*-1 );
75  else constantName = "POS" + IntToString(value*1);
76 
77  // if the constant has been found already, no need to create new node
78  if (CONSTANTS.find(constantName) != CONSTANTS.end()) {
79  temp->p1 = CONSTANTS[constantName];
80  }
81  else { // create new const node
82  Node * new_constant = new Node (
83  /* I= */ NULL,
84  /* op= */ constantName,
85  /* label= */ 0,
86  /* p1= */ NULL,
87  /* p2= */ NULL
88  );
89  CONSTANTS.insert( pair<string,Node*>(constantName, new_constant) );
90  temp->p1 = new_constant;
91  }
92  }
93  else temp->p1 = INPUT;
94 
95  // Check if the right predecessor of I is a constant or an input
96  if ( isa<ConstantInt>( temp->I->getOperand(1) )) {
97 
98  // prepend "POS" or "NEG" to the const name
99  string constantName;
100  int value = cast<ConstantInt>(temp->I->getOperand(1))->getZExtValue();
101  if (value<0) constantName = "NEG" + IntToString( value*-1 );
102  else constantName = "POS" + IntToString(value*1);
103 
104  // if the constant has been found already, no need to create new node
105  if (CONSTANTS.find(constantName) != CONSTANTS.end()) {
106  temp->p2 = CONSTANTS[constantName];
107  }
108  else { // create new node
109  Node * new_constant = new Node (
110  /* I= */ NULL,
111  /* op= */ constantName,
112  /* label= */ 0,
113  /* p1= */ NULL,
114  /* p2= */ NULL
115  );
116  CONSTANTS.insert( pair<string,Node*>(constantName, new_constant) );
117  temp->p2 = new_constant;
118  }
119  }
120  else temp->p2 = INPUT;
121 
122  return temp;
123 }

236 /*******************************************************************************
237  * Assignment Operator
238  ******************************************************************************
239  *
240  * Deep copy of graph objects. g2 = g1 allocates memory for new nodes
241  * so that g2 is the same size as g1 with unique nodes, while the data
242  * in the nodes stays the same (LLVM Instructions, names, etc.).
243  * Graphs can be thought of as "containers" for LLVM Instructions, so
244  * this operator has the same functionality as e.g. copying a vector of
245  * Instructions, i.e. vector <Instruction*> v2 = v1.
246  *
247  * Frequently with Graph object we need to traverse all nodes in the graph and
248  * perform an operation.
249  *
250  * This is the algorithm used (BFS):
251  *
252  * 1. Create a set of nodes visited. The set initially contains only the root
253  * 2. Create also a queue of nodes to visit, and insert the root
254  * 3. While the queue is not empty, pop the front element. If its left and
255  * right predecessors are operations and have not been visited, add them to
256  * the queue and mark them visited.
257  *
258  ******************************************************************************/
259 
260 Graph & Graph::operator = ( Graph & rhs ) { // g2 = g1, rhs is g1
261  if ( this != &rhs ) { // Check for self-assignment (if so just return)
262 
263  // This map maps nodes in the old graph (rhs) to its copy (this).
264  // It also keeps track of visited nodes during the BFS of the rhs graph.
265  map <Node*, Node*> visited;
266  this->DeleteAllNodes();
267 
268  if (rhs.size() == 0) return (*this);
269 
270  this->root = this->NewNode(rhs.getRoot()->I); // copy the root operation first
271 
272  // add this instruction and node to our map
273  this->GraphNodes.insert( pair<Instruction*, Node*> (this->root->I, this->root) );
274  this->current = this->root;
275 
276  queue <Node*> lhs_nodes; // queue of Nodes we still need to visit, in the "this" (new) graph
277  queue <Node*> rhs_nodes; // queue of Nodes we still need to visit, in rhs (old) graph
278  Node * t2; // unlike before, while we still do not want to change the
279  // current pointer of the rhs graph, the current pointer of
280  // the "this" graph cannot be in use elsewhere yet and hence
281  // may be used now
282 
283  rhs_nodes.push(rhs.root); // initialize our 2 queues
284  lhs_nodes.push(this->root);
285  visited.insert( pair<Node*, Node*> (rhs.getRoot(), this->root) ); // insert the roots
286 
287  // while still more nodes to process (rhs and lhs queues always
288  // have same size, no need to check both)
289  while (!rhs_nodes.empty()) {
290  this->current = lhs_nodes.front(); lhs_nodes.pop(); // pop the top node from lhs node queue
291  t2 = rhs_nodes.front(); rhs_nodes.pop(); // pop the top node from rhs node queue
292 
293  // left predecessor
294 
295  // if the node we just popped has a left predecessor in the rhs
296  // graph, we have to add the left predecessor to the lhs graph too
297  if (t2->p1->is_op) {
298  // if we haven't visited that predecessor before, then we can
299  // create a new node in lhs graph
300  if ( visited.find(t2->p1) == visited.end() ) {
301  this->GrowLeft(t2->p1->I); // add left to our new subtree
302  rhs_nodes.push(t2->p1); // push it onto both queues
303  lhs_nodes.push(this->current->p1); // the node we just added
304 
305  // mark this pair as visited
306  visited.insert( pair<Node*, Node*> (t2->p1, this->current->p1) );
307  }
308  else { // case where the predecessor exists but has been visited
309  this->current->p1 = visited[t2->p1];
310  }
311  }
312 
313  // repeat for right predecessor
314  if (t2->p2->is_op) {
315  if ( visited.find(t2->p2) == visited.end() ) {
316  this->GrowRight(t2->p2->I);
317  rhs_nodes.push(t2->p2);
318  lhs_nodes.push(this->current->p2);
319  visited.insert( pair<Node*, Node*> (t2->p2, this->current->p2) );
320  }
321  else {
322  this->current->p2 = visited[t2->p2];
323  }
324  }
325  }// while
326 
327  // Finally, set this->current to be the node in this corresponding to
328  // rhs.current, so the graphs are completely similar
329  this->current = visited[rhs.current];
330 
331  }
332  return * this;
333 }
