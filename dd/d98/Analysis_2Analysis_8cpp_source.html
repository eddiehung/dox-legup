75 LLVMBool LLVMVerifyModule(LLVMModuleRef M, LLVMVerifierFailureAction Action,
76  char **OutMessages) {
77  raw_ostream *DebugOS = Action != LLVMReturnStatusAction ? &errs() : nullptr;
78  std::string Messages;
79  raw_string_ostream MsgsOS(Messages);
80 
81  LLVMBool Result = verifyModule(*unwrap(M), OutMessages ? &MsgsOS : DebugOS);
82 
83  // Duplicate the output to stderr.
84  if (DebugOS && OutMessages)
85  *DebugOS << MsgsOS.str();
86 
87  if (Action == LLVMAbortProcessAction && Result)
88  report_fatal_error("Broken module found, compilation aborted!");
89 
90  if (OutMessages)
91  *OutMessages = strdup(MsgsOS.str().c_str());
92 
93  return Result;
94 }
