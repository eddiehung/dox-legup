83 /*
84  * triStateMask represents a 64 bit bit mask for an up to 64 bit wide signal, where each bit in
85  * the bitmask corresponds to one of four things.
86  * 1. A static 0 value independent of the circuit inputs
87  * 2. A static 1 value independent of the circuit inputs
88  * 3. An unknown value (could be 0 or 1 based on circuit inputs)
89  * 4. A sign extended value
90  * To represent such a mask, three uint128_t values are maintained.
91  * bitIsDynamic - each bit is 0 if the corresponding bit in the signal is known, and 1 otherwise
92  * bitValue - If bitIsDynamic[i] is 0, bitValue[i] is equal signal[i]
93  * - else, bitValue[i] = 0
94  * bitIsSext - If bitIsDynamic[i] is 1, bitIsSext[i] is 1 if signal[i] is sign extended from signal[MSB]
95  * - else bitIsSext[i] = 0
96  * MSB is the index of the left-most non-zero bit
97  * LSB is the index of the right-most non-zero bit
98  */
99 class triStateMask {
100  friend bool operator==(const triStateMask& a, const triStateMask& b);
101  public:
102  uint128_t bitIsDynamic;
103  uint128_t bitValue;
104  uint8_t bitwidth,MSB,LSB,extendFrom;

126  static uint128_t biggestVal(uint8_t bitwidth)
127  {
128  uint128_t temp;
129  if(bitwidth==0) temp=uint128_t(0,0);
130  else if(bitwidth<=64) temp=uint128_t(0,(uint64_t)0xFFFFFFFFFFFFFFFFLL>>((uint8_t)(64-bitwidth)));
131  else if(bitwidth<=128)
132  {
133  temp=uint128_t((uint64_t)0xFFFFFFFFFFFFFFFFLL>>((uint8_t)(128-bitwidth)),(uint64_t)0xFFFFFFFFFFFFFFFFLL);
134  }
135  else
136  {
137  printf("error bitwidth>128 = %u\n",bitwidth);
138  assert(bitwidth<=128);
139  }
140  return temp;
141  }
142  static uint128_t getOnesMask(uint8_t right,uint8_t left)
143  {
144  uint128_t onesLeft = biggestVal(left);
145  uint128_t onesRight = biggestVal(right);
146  return (onesLeft^onesRight);
147  }
