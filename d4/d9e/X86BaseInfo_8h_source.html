557  /// getSizeOfImm - Decode the "size of immediate" field from the TSFlags field
558  /// of the specified instruction.
559  inline unsigned getSizeOfImm(uint64_t TSFlags) {
560  switch (TSFlags & X86II::ImmMask) {
561  default: llvm_unreachable("Unknown immediate size");
562  case X86II::Imm8:
563  case X86II::Imm8PCRel: return 1;
564  case X86II::Imm16:
565  case X86II::Imm16PCRel: return 2;
