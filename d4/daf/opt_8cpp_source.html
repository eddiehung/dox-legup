201 static inline void addPass(PassManagerBase &PM, Pass *P) {
202  // Add the pass to the pass manager...
203  PM.add(P);
204 
205  // If we are verifying all of the intermediate steps, add the verifier...
206  if (VerifyEach) {

208  PM.add(createDebugInfoVerifierPass());

212 /// AddOptimizationPasses - This routine adds optimization passes
213 /// based on selected optimization level, OptLevel. This routine
214 /// duplicates llvm-gcc behaviour.
215 ///
216 /// OptLevel - Optimization Level
217 static void AddOptimizationPasses(PassManagerBase &MPM,FunctionPassManager &FPM,
218  unsigned OptLevel, unsigned SizeLevel) {
219  FPM.add(createVerifierPass()); // Verify that input is correct
220  MPM.add(createDebugInfoVerifierPass()); // Verify that debug info is correct
221 
222  PassManagerBuilder Builder;
223  Builder.OptLevel = OptLevel;
224  Builder.SizeLevel = SizeLevel;
225 
226  if (DisableInline) {
227  // No inlining pass
228  } else if (OptLevel > 1) {
229  Builder.Inliner = createFunctionInliningPass(OptLevel, SizeLevel);
230  } else {
231  Builder.Inliner = createAlwaysInlinerPass();
232  }
233  Builder.DisableUnitAtATime = !UnitAtATime;
234  Builder.DisableUnrollLoops = (DisableLoopUnrolling.getNumOccurrences() > 0) ?
235  DisableLoopUnrolling : OptLevel == 0;
236 
237  // This is final, unless there is a #pragma vectorize enable
238  if (DisableLoopVectorization)
239  Builder.LoopVectorize = false;
240  // If option wasn't forced via cmd line (-vectorize-loops, -loop-vectorize)
241  else if (!Builder.LoopVectorize)
242  Builder.LoopVectorize = OptLevel > 1 && SizeLevel < 2;
243 
244  // When #pragma vectorize is on for SLP, do the same as above
245  Builder.SLPVectorize =
246  DisableSLPVectorization ? false : OptLevel > 1 && SizeLevel < 2;
247 
248  Builder.populateFunctionPassManager(FPM);
249  Builder.populateModulePassManager(MPM);
250 }

252 static void AddStandardCompilePasses(PassManagerBase &PM) {
253  PM.add(createVerifierPass()); // Verify that input is correct
254 
255  // If the -strip-debug command line option was specified, do it.
256  if (StripDebug)
257  addPass(PM, createStripSymbolsPass(true));
258 
259  // Verify debug info only after it's (possibly) stripped.
260  PM.add(createDebugInfoVerifierPass());
261 
262  if (DisableOptimizations) return;
263 
264  // -std-compile-opts adds the same module passes as -O3.
265  PassManagerBuilder Builder;
266  if (!DisableInline)
267  Builder.Inliner = createFunctionInliningPass();
268  Builder.OptLevel = 3;
269  Builder.populateModulePassManager(PM);
270 }

338 int main(int argc, char **argv) {
339  sys::PrintStackTraceOnErrorSignal();
340  llvm::PrettyStackTraceProgram X(argc, argv);
341 
342  // Enable debug stream buffering.
343  EnableDebugBuffering = true;
344 
345  llvm_shutdown_obj Y; // Call llvm_shutdown() on exit.
346  LLVMContext &Context = getGlobalContext();
347 
348  InitializeAllTargets();
349  InitializeAllTargetMCs();
350  InitializeAllAsmPrinters();
351 
352  // Initialize passes
353  PassRegistry &Registry = *PassRegistry::getPassRegistry();
354  initializeCore(Registry);
355  initializeDebugIRPass(Registry);
356  initializeScalarOpts(Registry);
357  initializeObjCARCOpts(Registry);
358  initializeVectorization(Registry);
359  initializeIPO(Registry);
360  initializeAnalysis(Registry);
361  initializeIPA(Registry);
362  initializeTransformUtils(Registry);
363  initializeInstCombine(Registry);
364  initializeInstrumentation(Registry);
365  initializeTarget(Registry);
366  // For codegen passes, only passes that do IR to IR transformation are
367  // supported.
368  initializeCodeGenPreparePass(Registry);
369  initializeAtomicExpandLoadLinkedPass(Registry);
370 
371 #ifdef LINK_POLLY_INTO_TOOLS
372  polly::initializePollyPasses(Registry);
373 #endif
374 
375  cl::ParseCommandLineOptions(argc, argv,
376  "llvm .bc -> .bc modular optimizer and analysis printer\n");
377 
378  if (AnalyzeOnly && NoOutput) {
379  errs() << argv[0] << ": analyze mode conflicts with no-output mode.\n";
380  return 1;
381  }
382 
383  SMDiagnostic Err;
384 
385  // Load the input module...
386  std::unique_ptr<Module> M;
387  M.reset(ParseIRFile(InputFilename, Err, Context));
388 
389  if (!M.get()) {
390  Err.print(argv[0], errs());
391  return 1;
392  }
393 
394  // If we are supposed to override the target triple, do so now.
395  if (!TargetTriple.empty())
396  M->setTargetTriple(Triple::normalize(TargetTriple));
397 
398  // Figure out what stream we are supposed to write to...
399  std::unique_ptr<tool_output_file> Out;
400  if (NoOutput) {
401  if (!OutputFilename.empty())
402  errs() << "WARNING: The -o (output filename) option is ignored when\n"
403  "the --disable-output option is used.\n";
404  } else {
405  // Default to standard output.
406  if (OutputFilename.empty())
407  OutputFilename = "-";
408 
409  std::string ErrorInfo;
410  Out.reset(new tool_output_file(OutputFilename.c_str(), ErrorInfo,
411  sys::fs::F_None));
412  if (!ErrorInfo.empty()) {
413  errs() << ErrorInfo << '\n';
414  return 1;
415  }
416  }
417 
418  // If the output is set to be emitted to standard out, and standard out is a
419  // console, print out a warning message and refuse to do it. We don't
420  // impress anyone by spewing tons of binary goo to a terminal.
421  if (!Force && !NoOutput && !AnalyzeOnly && !OutputAssembly)
422  if (CheckBitcodeOutputToConsole(Out->os(), !Quiet))
423  NoOutput = true;
424 
425  if (PassPipeline.getNumOccurrences() > 0) {
426  OutputKind OK = OK_NoOutput;
427  if (!NoOutput)
428  OK = OutputAssembly ? OK_OutputAssembly : OK_OutputBitcode;
429 
430  VerifierKind VK = VK_VerifyInAndOut;
431  if (NoVerify)
432  VK = VK_NoVerifier;
433  else if (VerifyEach)
434  VK = VK_VerifyEachPass;
435 
436  // The user has asked to use the new pass manager and provided a pipeline
437  // string. Hand off the rest of the functionality to the new code for that
438  // layer.
439  return runPassPipeline(argv[0], Context, *M.get(), Out.get(), PassPipeline,
440  OK, VK)
441  ? 0
442  : 1;
443  }
444 
445  // Create a PassManager to hold and optimize the collection of passes we are
446  // about to build.
447  //
448  PassManager Passes;
449 
450  // Add an appropriate TargetLibraryInfo pass for the module's triple.
451  TargetLibraryInfo *TLI = new TargetLibraryInfo(Triple(M->getTargetTriple()));
452 
453  // The -disable-simplify-libcalls flag actually disables all builtin optzns.
454  if (DisableSimplifyLibCalls)
455  TLI->disableAllFunctions();
456  Passes.add(TLI);
457 
458  // Add an appropriate DataLayout instance for this module.
459  const DataLayout *DL = M.get()->getDataLayout();
460  if (!DL && !DefaultDataLayout.empty()) {
461  M->setDataLayout(DefaultDataLayout);
462  DL = M.get()->getDataLayout();
463  }
464 
465  if (DL)
466  Passes.add(new DataLayoutPass(M.get()));
467 
468  // pass tcl files to LegupConfig
469  if (!ConfigFile.empty()) {
470  for (unsigned i = 0; i < ConfigFile.size();++i) {
471  if (!legup::parseTclFile(ConfigFile[i], legup::LEGUP_CONFIG)) {
472  return 1;
473  }
474  }
475  }
476 
477  // run LegUp tcl commands
478  if (!LegUpTclCommands.empty()) {
479  if (!legup::parseTclString(LegUpTclCommands, legup::LEGUP_CONFIG)) {
480  return 1;
481  }
482  return 0;
483  }
484 
485 
486  Triple ModuleTriple(M->getTargetTriple());
487  TargetMachine *Machine = nullptr;
488  if (ModuleTriple.getArch())
489  Machine = GetTargetMachine(Triple(ModuleTriple));
490  std::unique_ptr<TargetMachine> TM(Machine);
491 
492  // Add internal analysis passes from the target machine.
493  if (TM.get())
494  TM->addAnalysisPasses(Passes);
495 
496  std::unique_ptr<FunctionPassManager> FPasses;
497  if (OptLevelO1 || OptLevelO2 || OptLevelOs || OptLevelOz || OptLevelO3) {
498  FPasses.reset(new FunctionPassManager(M.get()));
499  if (DL)
500  FPasses->add(new DataLayoutPass(M.get()));
501  if (TM.get())
502  TM->addAnalysisPasses(*FPasses);
503 
504  }
505 
506  if (PrintBreakpoints) {
507  // Default to standard output.
508  if (!Out) {
509  if (OutputFilename.empty())
510  OutputFilename = "-";
511 
512  std::string ErrorInfo;
513  Out.reset(new tool_output_file(OutputFilename.c_str(), ErrorInfo,
514  sys::fs::F_None));
515  if (!ErrorInfo.empty()) {
516  errs() << ErrorInfo << '\n';
517  return 1;
518  }
519  }
520  Passes.add(createBreakpointPrinter(Out->os()));
521  NoOutput = true;
522  }
523 
524  // If the -strip-debug command line option was specified, add it. If
525  // -std-compile-opts was also specified, it will handle StripDebug.
526  if (StripDebug && !StandardCompileOpts)
527  addPass(Passes, createStripSymbolsPass(true));
528 
529  // Create a new optimization pass for each one specified on the command line
530  for (unsigned i = 0; i < PassList.size(); ++i) {
531  // Check to see if -std-compile-opts was specified before this option. If
532  // so, handle it.
533  if (StandardCompileOpts &&
534  StandardCompileOpts.getPosition() < PassList.getPosition(i)) {
535  AddStandardCompilePasses(Passes);
536  StandardCompileOpts = false;
537  }
538 
539  if (StandardLinkOpts &&
540  StandardLinkOpts.getPosition() < PassList.getPosition(i)) {
541  AddStandardLinkPasses(Passes);
542  StandardLinkOpts = false;
543  }
544 
545  if (OptLevelO1 && OptLevelO1.getPosition() < PassList.getPosition(i)) {
546  AddOptimizationPasses(Passes, *FPasses, 1, 0);
547  OptLevelO1 = false;
548  }
549 
550  if (OptLevelO2 && OptLevelO2.getPosition() < PassList.getPosition(i)) {
551  AddOptimizationPasses(Passes, *FPasses, 2, 0);
552  OptLevelO2 = false;
553  }
554 
555  if (OptLevelOs && OptLevelOs.getPosition() < PassList.getPosition(i)) {
556  AddOptimizationPasses(Passes, *FPasses, 2, 1);
557  OptLevelOs = false;
558  }
559 
560  if (OptLevelOz && OptLevelOz.getPosition() < PassList.getPosition(i)) {
561  AddOptimizationPasses(Passes, *FPasses, 2, 2);
562  OptLevelOz = false;
563  }
564 
565  if (OptLevelO3 && OptLevelO3.getPosition() < PassList.getPosition(i)) {
566  AddOptimizationPasses(Passes, *FPasses, 3, 0);
567  OptLevelO3 = false;
568  }
569 
570  const PassInfo *PassInf = PassList[i];
571  Pass *P = nullptr;
572  if (PassInf->getTargetMachineCtor())
573  P = PassInf->getTargetMachineCtor()(TM.get());
574  else if (PassInf->getNormalCtor())
575  P = PassInf->getNormalCtor()();
576  else
577  errs() << argv[0] << ": cannot create pass: "
578  << PassInf->getPassName() << "\n";
579  if (P) {
580  PassKind Kind = P->getPassKind();
581  addPass(Passes, P);
582 
583  if (AnalyzeOnly) {
584  switch (Kind) {
585  case PT_BasicBlock:
586  Passes.add(createBasicBlockPassPrinter(PassInf, Out->os(), Quiet));
587  break;
588  case PT_Region:
589  Passes.add(createRegionPassPrinter(PassInf, Out->os(), Quiet));
590  break;
591  case PT_Loop:
592  Passes.add(createLoopPassPrinter(PassInf, Out->os(), Quiet));
593  break;
594  case PT_Function:
595  Passes.add(createFunctionPassPrinter(PassInf, Out->os(), Quiet));
596  break;
597  case PT_CallGraphSCC:
598  Passes.add(createCallGraphPassPrinter(PassInf, Out->os(), Quiet));
599  break;
600  default:
601  Passes.add(createModulePassPrinter(PassInf, Out->os(), Quiet));
602  break;
603  }
604  }
605  }
606 
607  if (PrintEachXForm)
608  Passes.add(createPrintModulePass(errs()));
609  }
610 
611  // If -std-compile-opts was specified at the end of the pass list, add them.
612  if (StandardCompileOpts) {
613  AddStandardCompilePasses(Passes);
614  StandardCompileOpts = false;
615  }
616 
617  if (StandardLinkOpts) {
618  AddStandardLinkPasses(Passes);
619  StandardLinkOpts = false;
620  }
621 
622  if (OptLevelO1)
623  AddOptimizationPasses(Passes, *FPasses, 1, 0);
624 
625  if (OptLevelO2)
626  AddOptimizationPasses(Passes, *FPasses, 2, 0);
627 
628  if (OptLevelOs)
629  AddOptimizationPasses(Passes, *FPasses, 2, 1);
630 
631  if (OptLevelOz)
632  AddOptimizationPasses(Passes, *FPasses, 2, 2);
633 
634  if (OptLevelO3)
635  AddOptimizationPasses(Passes, *FPasses, 3, 0);
636 
637  if (OptLevelO1 || OptLevelO2 || OptLevelOs || OptLevelOz || OptLevelO3) {
638  FPasses->doInitialization();
639  for (Module::iterator F = M->begin(), E = M->end(); F != E; ++F)
640  FPasses->run(*F);
641  FPasses->doFinalization();
642  }
643 
644  // Check that the module is well formed on completion of optimization
645  if (!NoVerify && !VerifyEach) {
646  Passes.add(createVerifierPass());
647  Passes.add(createDebugInfoVerifierPass());
648  }
649 
650  // Write bitcode or assembly to the output as the last step...
651  if (!NoOutput && !AnalyzeOnly) {
652  if (OutputAssembly)
653  Passes.add(createPrintModulePass(Out->os()));
654  else
655  Passes.add(createBitcodeWriterPass(Out->os()));
656  }
657 
658  // Before executing passes, print the final values of the LLVM options.
659  cl::PrintOptionValues();
660 
661  // Now that we have all of the passes ready, run them.
662  Passes.run(*M.get());
663 
664  // Declare success.
665  if (!NoOutput || PrintBreakpoints)
666  Out->keep();
667 
668  return 0;
669 }
