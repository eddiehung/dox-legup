612  if (StandardCompileOpts) {
613  AddStandardCompilePasses(Passes);

615  }

618  AddStandardLinkPasses(Passes);

640  FPasses->run(*F);
641  FPasses->doFinalization();
642  }

645  if (!NoVerify && !VerifyEach) {
646  Passes.add(createVerifierPass());
647  Passes.add(createDebugInfoVerifierPass());

649

651  if (!NoOutput && !AnalyzeOnly) {

654  else

659  cl::PrintOptionValues();

662  Passes.run(*M.get());
663

665  if (!NoOutput || PrintBreakpoints)
666  Out->keep();
