119 const MCExpr *
120 MCAsmInfo::getExprForFDESymbol(const MCSymbol *Sym,
121  unsigned Encoding,
122  MCStreamer &Streamer) const {
123  if (!(Encoding & dwarf::DW_EH_PE_pcrel))
124  return MCSymbolRefExpr::Create(Sym, Streamer.getContext());
125 
126  MCContext &Context = Streamer.getContext();
127  const MCExpr *Res = MCSymbolRefExpr::Create(Sym, Context);
128  MCSymbol *PCSym = Context.CreateTempSymbol();
129  Streamer.EmitLabel(PCSym);
130  const MCExpr *PC = MCSymbolRefExpr::Create(PCSym, Context);
131  return MCBinaryExpr::CreateSub(Res, PC, Context);
132 }
