731 int ModuloScheduler::getCycleRecMII(std::list<Instruction *> &path) {
732 
733  int cycleDelay = 0;
734  int cycleDistance = 0;
735  Instruction *first = NULL;
736  Instruction *prev = NULL;
737  int edgeDistance = 0;
738  bool multipleBackEdges = false;
739  for (std::list<Instruction *>::iterator pi = path.begin(), pe = path.end();
740  pi != pe; ++pi) {
741  Instruction *I = *pi;
742  if (!prev) {
743  first = I;
744  } else {
745  edgeDistance = distance(prev, I);
746  if (edgeDistance) {
747  if (cycleDistance) {
748  multipleBackEdges = true;
749  }
750  cycleDistance = edgeDistance;
751  }
752  }
753  prev = I;
754  cycleDelay += delay(I);
755  // do I need this?
756  // well if we have a recurrence path where the latency of all
757  // operations is 0 then the recMII for that path will be 0...
758  if (isa<PHINode>(I)) {
759  cycleDelay++;
760  }
761  File() << "delay: " << cycleDelay << " I: " << *I << "\n";
762  }
763  assert(prev);
764  assert(first);
765  edgeDistance = distance(prev, first);
766  if (edgeDistance) {
767  if (cycleDistance) {
768  multipleBackEdges = true;
769  }
770  cycleDistance = edgeDistance;
771  }
772  if (multipleBackEdges) {
773  File() << "More than one back edge -- ignoring\n";
774  return 0;
775  }
776  assert(cycleDistance);
777  File() << "Total cycle delay: " << cycleDelay << "\n";
778  File() << "Total cycle dependency distance: " << cycleDistance << "\n";
779  int cycleRecMII = ceil((float)cycleDelay / (float)cycleDistance);
780  File() << "recMII = ceil(delay/distance) = " << cycleRecMII << "\n";
781  File() << "\n";
