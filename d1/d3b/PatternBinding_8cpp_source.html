125 void PatternBinding::PatternAnalysis
126 (
127  PatternMap &Patterns,
128  std::map<Value*, Value*> &AllBindingPairs,
129  std::map<Graph*, Graph*> &GraphPairs,
130  std::set<Instruction*> &InstructionsInGraphs
131 )
132 {
133  if (LEGUP_CONFIG->getParameterInt("ENABLE_PATTERN_SHARING") == 0) return;
134  if (LEGUP_CONFIG->getParameterInt("PS_MAX_SIZE") < 1) return;
135 
136  // Output a file describing all patterns discovered
137  formatted_raw_ostream out(this->alloc->getPatternFile());
138  string function_name = this->Fp->getName().str();
139  out << "Function: " << function_name << "\n";
140  create_dot_directory_for_function(function_name);
141 
142  // Maps every instruction to a set of Graphs which contain it
143  map<Instruction*, set<Graph*> > InstructionGraphUses;
144 
145  // Discover all patterns and graphs in this function and store
146  // it in the PatternMap
147  DiscoverAllPatternsInFunction (
148  Patterns,
149  InstructionGraphUses
150  );
151 
152  // Pair the graphs together, starting from the largest size found
153  // and ending with graphs of size PS_MIN_SIZE (in legup.tcl).
154 
155  // Note that many overlapping nodes exist, so every time a pair is
156  // formed, every graph which it is part of needs to be 'removed'
157  // (or ignored from now on). This is done with the AlreadyUsed map
158  std::map<Graph*, bool> AlreadyUsed; // could be a set
159 
160  // iterate from largest size to smallest (backwards in the map)
161  // and pair together the graphs found above, taking into
162  // consideration the liveness information of each graph
163  // (from the Live Variable Analysis, or LVA, pass)
164  PatternMap::PatternMap_iterator PMi = Patterns.end();
165  PatternMap::PatternMap_iterator PMe = Patterns.begin();
166  for ( ; PMi != PMe; )
167  {
168  --PMi;
169 
170  // Don't share for pattern sizes < PS_MIN_SIZE
171  if (PMi->first < LEGUP_CONFIG->getParameterInt("PS_MIN_SIZE")) {
172  continue;
173  }
174 
175  GraphLVAPairing(
176  PMi,
177  PMi->first,
178  AllBindingPairs,
179  GraphPairs,
180  InstructionsInGraphs,
181  InstructionGraphUses,
182  AlreadyUsed,
183  function_name,
184  out
185  );
186  }
187  out << "\n";
188 }

237 // See Resource Sharing Paper for algorithm description (Repeatedly uses BFS)
238 void PatternBinding::DiscoverAllPatternsInBasicBlock
239 (
240  Function::iterator BB,
241  PatternMap &Patterns,
242  map<Instruction*, set<Graph*> > &InstructionGraphUses,
243  map<Instruction*, vector<Graph*> > &GraphsFound
244 )
245 {
246  queue <Graph*> Graphs;
247 
248  // Fill queue with all subgraphs of size 1
249  DiscoverAllPatternsOfSizeOneInBasicBlock(
250  BB,
251  Patterns,
252  Graphs,
253  InstructionGraphUses
254  );
255 
256  int current_size = 1; // current size of subgraphs we are considering
257 
258  // Now that we have all graphs of size 1 in a queue, add every possible
259  // predecessor for every graph of size 1 to make every possible graph of
260  // size 2. Place each new graph of size 2 discovered at the end of the
261  // Graphs queue, then repeat this for all graphs of size 2 to produce
262  // graphs of size 3, etc.
263  while (!Graphs.empty()) {
264  if ( Graphs.front()->size() >=
265  LEGUP_CONFIG->getParameterInt("PS_MAX_SIZE") )
266  break; // stop growing
267 
268  // remove next graph from the queue and all possible predecessors
269  Graph * g = Graphs.front(); Graphs.pop();
270 
271  if (g->size() > current_size) {
272  current_size++; // beginning a new set of subgraphs
273 
274  // two graphs cannot be duplicates if they are not of the same size,
275  // so clear whenever the size of Graphs in the queue changes
276  GraphsFound.clear();
277  }

342 // Given graph g of size N, add all possible predecessors of every node
343 // to g, one by one, to create graphs of size N+1. Push these graphs
344 // on to the Graphs queue and add each new graph to the PatternMap
345 
346 // Algorithm: starting from the root of g, perform a BFS of all
347 // the nodes in the graph. Every time a node is reached which
348 // does not have a predecessor in the graph, find what that
349 // predecessor is from the LLVM DFG and create a copy of g
350 // with that new predecessor. This copy has size one greater
351 // than g. Add the copy to Patterns and Graphs.
352 
353 void PatternBinding::DiscoverGraphsOfSizeOneGreater
354 (
355  Function::iterator BB,
356  Graph *g,
357  PatternMap &Patterns,
358  queue <Graph*> &Graphs,
359  map<Instruction*, set<Graph*> > &InstructionGraphUses,
360  map<Instruction*, vector<Graph*> > &GraphsFound
361 )
362 {
363  set<Node*> visited; // traversed nodes in the graph
364  queue <Node*> Nodes; // queue of nodes to consider (BFS)
365  Nodes.push(g->getRoot());
366 
367  while (!Nodes.empty()) { // while still nodes to visit
368 
369  // set "current" to be the next top node in the queue
370  g->setCurrent( Nodes.front() ); Nodes.pop();
371 
372  // For both predecessors of the current node
373  for (int predecessor_number=0; predecessor_number < 2; ++predecessor_number)
374  {
375  Node *pred_node;
376  if (predecessor_number==0) pred_node = g->getCurrent()->p1;
377  else pred_node = g->getCurrent()->p2;
378 
379  // if predecessor is an operation in the graph g
380  if (pred_node->is_op) {
381  // and if we haven't visited it, then add it to our BFS
382  // queue to visit later, and move on
383  if (visited.find(pred_node) == visited.end()) {
384  Nodes.push(pred_node); // add to queue
385  visited.insert(pred_node); // mark as visited
386  }
387  }
388  // Otherwise, the predecessor is not in the current graph g.
389  // This means we can create a copy of g with this predecessor,
390  // and add the copy to our data structures
391  else {
392  Instruction * pred = dyn_cast<Instruction> (
393  g->getCurrent()->I->getOperand(predecessor_number)
394  );
395  if (!check_predecessor_legality(pred, g, BB, this->MBW,
396  this->fsm, this->alloc)) {
397  continue;
398  }
399  CreateGraphWithNewPredecessor(
400  pred,
401  predecessor_number, // 0=left, 1=right
402  g,
403  Patterns,
404  Graphs,
405  InstructionGraphUses,
406  GraphsFound
407  );
408  }
409  } // for both predecessors
410  } // while
411 }

414 // Check legality of a predecessor. It must be an instruction,
415 // in the same basic block as its successor, and can't have
416 // fanouts other than its successor in the same state
417 bool check_predecessor_legality
418 (
419  Instruction *pred,
420  Graph *g,
421  Function::iterator BB,
422  MinimizeBitwidth *MBW,
423  FiniteStateMachine *fsm,
424  Allocation *alloc
425 )
426 {
427  if (!pred) return false; // if it is not an instruction
428  if (pred->getParent() != BB) return false; // or not in the same block
429  string name = LEGUP_CONFIG->getOpNameFromInst(pred, alloc);
430  if (!( check_validity_for_chaining(pred, name, MBW) )) return false;
431 
432  // Final check: we don't want to include this instruction
433  // in the graph if it has fanouts which are in the same
434  // state and not in the current graph.
435  bool IllegalFanouts = false;
436 
437  Value::use_iterator uit = pred->use_begin();
438  Value::use_iterator end = pred->use_end();
439  for ( ; uit != end; ++uit) {
440  Instruction * successor = dyn_cast<Instruction> ( *uit );
441  if (!successor) continue; // successor is not an instruction
442  // if the successor is in the same state
443  if ( fsm->getEndState(successor) == fsm->getEndState(pred) ) {
444  // and if the graph does not contain the successor
445  if ( g->GraphNodes.find(successor) == g->GraphNodes.end() ) {
446  IllegalFanouts = true;
447  break;
448  }
449  }
450  }
451  if (IllegalFanouts) return false;
452  return true;
453 }

624 // Uses BFS to traverse all corresponding nodes in both graphs
625 bool are_graphs_identical_bfs(Graph * g1, Graph * g2) {
626 
627  // BFS data structures
628  set <Node*> visited;
629  queue <Node*> g1_nodes;
630  queue <Node*> g2_nodes;
631 
632  // Initialize BFS
633  g1_nodes.push(g1->getRoot());
634  g2_nodes.push(g2->getRoot());
635  visited.insert(g1->getRoot());
636 
637  Node *t1; // use these temp pointers as opposed to the graph's "current" pointer in order
638  Node *t2; // to not change the value of the current pointer, which may be in use by another function
639 
640  // Main BFS Loop
641  while (!g1_nodes.empty()) {
642  // Pop next nodes from queue
643  t1 = g1_nodes.front(); g1_nodes.pop();
644  t2 = g2_nodes.front(); g2_nodes.pop();
645 
646  // Check equality of both the left and right predecessors
647  Node *g1_pred;
648  Node *g2_pred;
649  for (int predecessor_number = 0; predecessor_number < 2; ++predecessor_number) {
650  if (predecessor_number == 0) { // left predecessor
651  g1_pred = t1->p1;
652  g2_pred = t2->p1;
653  }
654  else { // right predecessor
655  g1_pred = t1->p2;
656  g2_pred = t2->p2;
657  }
658  if (g1_pred->is_op) { // If the predecessor is an operation
659  if (!(g2_pred)) return false; // g1->current->p1 exists. but not g2->current->p1
660  if ( (g1_pred->I) != (g2_pred->I) ) return false; // both exist, but node doesn't match
661  if ( visited.find(g1_pred) == visited.end() ) {
662  g1_nodes.push(g1_pred);
663  g2_nodes.push(g2_pred);
664  visited.insert(g1_pred);
665  }
666  }
667  else if (g2_pred->is_op) return false; // g2->current->p1 exists. but not g1->current->p1
668  }
669  }

681 // Given a PatternMap with graphs which are potential candidates for pairing,
682 // determine which graphs can be paired. To do this, liveness intervals are
683 // calculated for all instructions in the graphs. Then, select a matching.
684 void PatternBinding::GraphLVAPairing
685 (
686  PatternMap::PatternMap_iterator iter1,
687  int current_size,
688  std::map<Value*, Value*> &AllBindingPairs,
689  std::map<Graph*, Graph*> &GraphPairs,
690  std::set<Instruction*> &InstructionsInGraphs,
691  map<Instruction*, set<Graph*> > &InstructionGraphUses,
692  map<Graph*, bool> &AlreadyUsed, string & foldername,
693  formatted_raw_ostream &out
694 )
695 {
696  // Hierarchy of iteration can be confusing. Recall a PatternMap maps
697  // an int (pattern size) to a list of patterns with that size.
698  // But each pattern itself is a list of Graphs (all pattern instances).
699  // See PatternMap.h for more information.
700 
701  // iter1, passed as a parameter, is an iterator to a PatternList
702  // (i.e. a list containing every pattern of that size)
703 
704  // iterate over every pattern with that size (e.g. all patterns of size 2)
705  PatternMap::PatternList_iterator iter2;
706 
707  // iterate over every graph in that pattern (all graphs of this same pattern)
708  PatternMap::Graph_iterator iter3;
709 
710  // iterate over every instruction in that graph
711  Graph::GraphNodes_iterator iter4;
712 
713  // for each GraphList in the PatternList (Pattern list iterator)
714  for (iter2 = iter1->second.begin(); iter2 != iter1->second.end(); ++iter2) {
715 
716  if (iter2->size() == 1) continue; // Can't share if frequency 1
717 
718  // 0. Create a set of all Graphs in this pattern and all instructions
719  // in this pattern (every instruction in every graph)
720 
721  std::set<Graph*> Graphs; // set of all Graphs of this pattern
722  std::set<Instruction*> Instructions; // all instructions in all graphs in this pattern
723 
724  // for each Graph in this Pattern (Graph iterator)
725  for (iter3 = iter2->begin(); iter3 != iter2->end(); ++iter3) {
726  if ( AlreadyUsed[*iter3] ) continue;
727  Graphs.insert( *iter3 );
728 
729  // for all instructions in this graph
730  for (iter4 = (*iter3)->GraphNodes.begin();
731  iter4 != (*iter3)->GraphNodes.end(); ++iter4) {
732  Instructions.insert(iter4->first);
733  assert(iter4->second->I == iter4->first);
734  }
735  }
736 
737  // 1. Create a map that maps every instruction in this pattern to every other
738  // instruction in this pattern with which it is independent (independent
739  // means in different states).
740  std::map<Instruction*, std::set<Instruction*> > IndependentInstructions;
741 
742  // Fills the Independent instructions map (adjacency list)
743  FindIndependentInstructions(Instructions, IndependentInstructions,
744  this->LVA, this->fsm);
745 
746  // Now, every instruction in this pattern is mapped to a set of every
747  // other instruction that it is independent with in terms of live
748  // intervals
749 
750  // 2. Next, create a similar map as IndependentInstructions, but for
751  // Graphs: this maps every Graph to every other Graph that it is
752  // independent with (i.e. all instructions have non-overlapping lifetimes)
753  std::map<Graph*, std::set<Graph*> > IndependentGraphs;
754 
755  std::set<Graph*>::iterator g = Graphs.begin();
756  std::set<Graph*>::iterator ge = Graphs.end();
757  for ( ; g != ge; ++g ) {
758  IndependentGraphs.insert(std::make_pair(*g, std::set<Graph*>()));
759  }
760 
761  // As above with instructions, now fill IndependentGraphs
762  FindIndependentGraphs(IndependentGraphs, IndependentInstructions);
763 
764  // 3. Finally, given independence information for every graph, pair them.
765  //
766  // Consider the results of steps 0-2 above:
767  // - Step 1 mapped every Instruction in this pattern to every other
768  // instruction with which it is independent
769  // - Step 2 used the information in Step 1 to map every LegUp Graph
770  // in this pattern to every other LegUp Graph with which it is
771  // independent.
772  //
773  // The result is IndependentGraphs, which itself can be thought of
774  // as an adjacency list representing a graph: IndependentGraphs
775  // maps every LegUp Graph to a set of other LegUp Graphs. Treating
776  // each LegUp Graph as a vertex and independence as edges, the result
777  // is an adjacency list representation of an "Independence graph"
778  // (in which nodes representing LegUp Graphs are adjacent if their
779  // LegUp Graphs have independent lifetimes).
780  //
781  // The goal now is to pair up the maximum number of adjacent vertices
782  // in this Independence graph (i.e. pair up as many independent LegUp
783  // Graphs as possible for sharing). Moreover, edge weights can be
784  // added to the Independence graph reflecting preferred pairing for
785  // LegUp Graphs based on bit width and other optimizations.
786  //
787  // This weighted maximum matching problem is solved optimally by the
788  // Blossom Algorithm: http://en.wikipedia.org/wiki/Blossom_algorithm,
789  // also called Jack Edmond's Maximum Matching Algorithm,
790  // which solves for a maximum matching in polytime.
791  //
792  // Currently however, a sub-optimal algorithm is used in which graphs
793  // are paired greedily and the optimizations mentioned above are
794  // used as heuristics in a cost function to guide the greedy algorithm.
795  //
796  // -- SHADJIS 2012
797 
798  int numPairs_old = GraphPairs.size();

801  MakePairs(
802  IndependentGraphs,
803  AllBindingPairs,
804  GraphPairs,
805  InstructionsInGraphs,
806  InstructionGraphUses,
807  AlreadyUsed
808  );

810  int numPairs_new = GraphPairs.size();

812  // 4. Now all pairs have been made. Write binding results to output files
813  int numPairs = numPairs_new - numPairs_old;
814  if (!numPairs) continue; // No new pairs added, so don't output anything
815 
816  // We need to select a graph whose nodes we can write to the text and
817  // dot files. Any graph will do, so arbitrarily pick the first one
818  Graph *representative = iter2->front();
819  assert(representative);
820

822  current_size,
823  numPairs,
824  representative,
825  AlreadyUsed,
826  foldername,
827  iter2,
828  out
829  );
830 
831  out << "\tInstructions:\n";
832  for (std::set<Instruction*>::iterator I = InstructionsInGraphs.begin(),
833  E = InstructionsInGraphs.end(); I != E; ++I) {
834  out << "\t" << **I << "\n";
835  }
836  out << "\n";
837 
838  WriteDotAndVerilogFiles (
839  current_size,
840  numPairs,
841  representative,
842  foldername
843  );

859 // all equivalent nodes in each graph.
860 // For example, if the pattern being considered is add->sub, then two
861 // graphs are "independent" if their adds have non-overlapping liftimes
862 // AND if their subtracts have non-overlapping lifetimes.

865 // bits of one another (e.g. don't pair an 8-bit adder with a 64-bit adder)
866 void PatternBinding::FindIndependentGraphs
867 (
868  std::map<Graph*, std::set<Graph*> > & IndependentGraphs,
869  std::map<Instruction*, std::set<Instruction*> > & IndependentInstructions
870 )

872  // For all graphs (g1)
873  std::map<Graph*, std::set<Graph*> >::iterator i = IndependentGraphs.begin();
874  std::map<Graph*, std::set<Graph*> >::iterator end = IndependentGraphs.end();
875  for ( ; i != end; ++i) {
876 
877  // For all graphs (g2)
878  std::map<Graph*, std::set<Graph*> >::iterator j = i;
879  for ( ; j != end; ++j) {
880 
881  Graph *g1 = i->first, *g2 = j->first;
882

884  continue;

888  // possible that two graphs share a common instruction. Make sure
889  // this is not the case.
890  if (share_common_instruction(g1, g2)) continue;

899  }

903  i = IndependentGraphs.begin();

905  for ( ; i != end; ++i) {
906  IndependentGraphs[i->first].erase(i->first); // erase self

927 // Given two graphs, check that all of their corresponding instructions
928 // are independent
929 bool PatternBinding::CheckAllOperationsForIndependence
930 (
931  Graph *g1,
932  Graph *g2,
933  std::map<Instruction*, std::set<Instruction*> > &IndependentInstructions
934 )
935 {
936  // In order to handle commutative operations, two maps will be created:
937 
938  // Recall that nodes in a graph are labeled with integers from
939  // 1 to GraphSize such that two identical graphs with different
940  // topology (due to commutative operations) will have corresponding
941  // operations labelled consitently (i.e. the node labelled "1" on
942  // both graphs will be the same operation, even if it is in a
943  // different location)
944 
945  // Therefore, for accessibility convenience two maps are created
946  // here mapping the label integer to the Instruction it represents
947  // on both maps
948 
949  std::map<unsigned, Instruction*> Graph1_Labels;
950  std::map<unsigned, Instruction*> Graph2_Labels;
951 
952  // Fill these maps
953 
954  Graph::GraphNodes_iterator GNi;
955  for (GNi = g1->GraphNodes.begin(); GNi != g1->GraphNodes.end();
956  ++GNi) {
957  // GNi->second is a Node *, hence it has an int label, and
958  // GNi->first is the instruction of that Node
959  Graph1_Labels[GNi->second->label] = GNi->first;
960  assert(GNi->second->I == GNi->first);
961  assert(GNi->second->label > 0);
962  assert(GNi->second->label <= g1->size());
963  }
964  for (GNi = g2->GraphNodes.begin(); GNi != g2->GraphNodes.end();
965  ++GNi) {
966  Graph2_Labels[GNi->second->label] = GNi->first;
967  assert(GNi->second->I == GNi->first);
968  assert(GNi->second->label > 0);
969  assert(GNi->second->label <= g2->size());
970  }
971 
972  // The two maps are now filled, so check every operation in both
973  // graphs for live interval independence
974  // If every operation in the graph is independent with its partner
975  // in the other graph, then we can share these Graphs
976  bool share_graphs = true;
977  assert(g1->size() == g2->size());
978  for (int index = 1; index <= g1->size(); ++index) {
979  Instruction * i1 = Graph1_Labels[index];
980  Instruction * i2 = Graph2_Labels[index];
981  assert( strcmp(i1->getOpcodeName(), i2->getOpcodeName()) == 0 );
982 
983  // can't share if the operations are not independent
984  if ( IndependentInstructions[i1].find(i2) ==
985  IndependentInstructions[i1].end() ) {
986  share_graphs = false;
987  break;
988  }
989 
990  // or if their bit widths are too different
991  if ( !within_bitwidth_threshold(i1, i2, this->MBW) ) {
992  share_graphs = false;
993  break;
994  }
995 
996  // or if sharing the instructions causes a combinational loop
997  if (combinationalLoop(i1, i2)) {
998  share_graphs = false;
999  break;
1000  }
1001  }
1002  return share_graphs;
1003 }

1254 // Given graph pairing candidate "partnerCandidate" and
1255 // a mapping of all the nodes of partnerCandidate to
1256 // the nodes of the original graph, return a cost associated
1257 // with sharing these two graphs.
1258 // See Resource Sharing paper for more information on the cost
1259 // function and optimizations considered.
1260 // Also return if a combinational loop is found as this
1261 // disqualifies the pairing.
1262 int PatternBinding::CalculatePairingCost
1263 (
1264  bool &combinationalLoop,
1265  Graph *partnerCandidate,
1266  std::map<unsigned, Node*> &Graph1_NodeLabels,
1267  map<Instruction*, Graph* > &InstructionAssignedGraph1,
1268  map<Instruction*, Graph* > &InstructionAssignedGraph2,
1269  std::set<Instruction*> &InstructionsInGraphs
1270 )
1271 {
1272  int current_cost = 0;
1273 
1274  // Iterate over all the nodes of the pairing candidate
1275  Graph::GraphNodes_iterator GNi;
1276  for (GNi = partnerCandidate->GraphNodes.begin(); GNi !=
1277  partnerCandidate->GraphNodes.end(); ++GNi) {
1278 
1279  // n2 is the node in the pairing candidate
1280  Node *n2 = GNi->second;
1281 
1282  // n1 is the equivalent node in Graph1 to n2
1283  Node *n1 = Graph1_NodeLabels[n2->label];

1286  current_cost += abs( (int)(this->MBW->getMinBitwidth(n1->I) - this->MBW->getMinBitwidth(n2->I)) );

1288  this->MBW->getMinBitwidth(n2->I->getOperand(0))) );
1289  current_cost += abs( (int)(this->MBW->getMinBitwidth(n1->I->getOperand(1)) -
1290  this->MBW->getMinBitwidth(n2->I->getOperand(1))) );

1293  // nodes share a common input, the cost of a mux is saved.

1298  n1,
1299  n2,
1300  InstructionAssignedGraph1,
1301  InstructionAssignedGraph2,
1302  InstructionsInGraphs
1303  );
1304

1306  }
