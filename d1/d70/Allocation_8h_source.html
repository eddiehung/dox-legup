350 class PropagatingSignal {
351 public:
352 
353  // All public constructors take an RTLSignal.
354  //
355  // The main constructor optionally takes a function
356  // at which the signal should stop propagating and
357  // exist as a wire.
358  //
359  PropagatingSignal(RTLSignal *sig, Function *finalFunction = 0)
360  { init(false, false, finalFunction, sig); }
361 
362  // Constructor for signals that stop propagating at
363  // the top level module. This special case must be
364  // handled because the top level module has no
365  // function representation.
366  //
367  // Optionally takes a flag that will identify a
368  // propagating signal as a memory signal.
369  // Unfortunately, there is some special handling
370  // for memory signals.
371  //
372  PropagatingSignal(RTLSignal *sig, bool stopsAtTopLevel, bool isMemory = false)
373  { init(isMemory, stopsAtTopLevel, 0, sig); }
374 
375  PropagatingSignal(const PropagatingSignal &toCopy) {
376 
377  _isMemory = toCopy._isMemory;
378  _stopsAtTopLevelModule = toCopy._stopsAtTopLevelModule;
379  _finalFunction = toCopy._finalFunction;
380  _merged = toCopy._merged;
381  _pthreadsSignal = toCopy._pthreadsSignal;
382 
383  assert(toCopy._signal);
384  _signal = toCopy._signal;
385 
386  // For debugging
387  getName();
388 
389  _widthOverriden = toCopy._widthOverriden;
390  if (_widthOverriden)
391  _overridenWidth = RTLWidth(toCopy._overridenWidth.getHi(), toCopy._overridenWidth.getLo());
392  else
393  _overridenWidth = RTLWidth();
394 
395  }
396 
397  // These methods are hacks that allow the memory signal
398  // code to be replaced with propagating signal code
399  //
400  bool isMemory() { return _isMemory; }
401  void setMemory(bool flag) { _isMemory = flag; }
402 
403  // Set this for any signals that will be wires at the top
404  // level module (as opposed to external I/O)
405  //
406  bool stopsAtTopLevelModule() { return _stopsAtTopLevelModule; }
407  void setStopsAtTopLevelModule(bool flag) { _stopsAtTopLevelModule = flag; }
408 
409  // The function at which the propagating signal will stop
410  // propagating and become a wire
411  //
412  void setFinalFunction(Function *f) { _finalFunction = f; }
413  bool stopsPropagatingAtFunction(Function *f) { assert(f); return _finalFunction == f; }
414 
415  // Sets the underlying RTLSignal
416  //
417  void setSignal(RTLSignal *sig) { _signal = sig; }
418  RTLSignal *getSignal() { assert(_signal); return _signal; }
419 
420  // Propagating signals can "merge" with other signals
421  // The merging should not change the actual (RTL) signal,
422  // instead, we will allow the width of the signal to
423  // appear as the merged width once the signals have
424  // merged.
425  void overrideWidth(RTLWidth width) { assert(_signal); _overridenWidth = width; _widthOverriden = true; }
426  RTLWidth getWidth() { assert(_signal); return _widthOverriden?_overridenWidth:_signal->getWidth(); }
427 
428  // Convenience method for getting the name of the
429  // underlying (RTL) signal

432
