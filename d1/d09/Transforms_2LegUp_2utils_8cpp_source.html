235 bool isolateGV(Module &M, std::set<GlobalValue*> &Named) {
236 
237  // Mark all globals internal
238  // FIXME: what should we do with private linkage?
239  set_all_linkage(M, GlobalValue::InternalLinkage);
240 
241  // All of the functions may be used by global variables or the named
242  // globals. Loop through them and create a new, external functions that
243  // can be "used", instead of ones with bodies.
244  std::vector<Function*> NewFunctions;
245 
246  Function *Last = --M.end(); // Figure out where the last real fn is.
247 
248  for (Module::iterator I = M.begin(); ; ++I) {
249  if (std::find(Named.begin(), Named.end(), &*I) == Named.end()) {
250  Function *New = Function::Create(I->getFunctionType(),
251  GlobalValue::ExternalLinkage);
252  New->copyAttributesFrom(I);
253 
254  // If it's not the named function, delete the body of the function
255  I->dropAllReferences();
256 
257  M.getFunctionList().push_back(New);
258  NewFunctions.push_back(New);
259  New->takeName(I);
260  }
261 
262  if (&*I == Last) break; // Stop after processing the last function
263  }
264 
265  // Now that we have replacements all set up, loop through the module,
266  // deleting the old functions, replacing them with the newly created
267  // functions.
268  if (!NewFunctions.empty()) {
269  unsigned FuncNum = 0;
270  Module::iterator I = M.begin();
271  do {
272  if (std::find(Named.begin(), Named.end(), &*I) == Named.end()) {
273  // Make everything that uses the old function use the new dummy fn
274  I->replaceAllUsesWith(NewFunctions[FuncNum++]);
275 
276  Function *Old = I;
277  ++I; // Move the iterator to the new function
278 
279  // Delete the old function!
280  M.getFunctionList().erase(Old);
281 
282  } else {
283  ++I; // Skip the function we are extracting
284  }
285  } while (&*I != NewFunctions[0]);
286  }
287 
288  // set function linkage back to external
289  set_all_linkage(M, GlobalValue::ExternalLinkage);
290 
291  return true;
292 }
