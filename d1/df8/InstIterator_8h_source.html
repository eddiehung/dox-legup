108  inline void advanceToNextBB() {
109  // The only way that the II could be broken is if it is now pointing to
110  // the end() of the current BasicBlock and there are successor BBs.
111  while (BI == BB->end()) {
112  ++BB;
113  if (BB == BBs->end()) break;
114  BI = BB->begin();
115  }
116  }
