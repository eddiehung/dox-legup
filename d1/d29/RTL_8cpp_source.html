33 unsigned replaceParametersAndEval(std::string num, const RTLModule *rtl, const
34  Allocation *alloc) {
35  for (RTLModule::const_signal_iterator i = rtl->param_begin(), e =
36  rtl->param_end(); i != e; ++i) {
37  RTLSignal *param = *i;
38  replaceAll(num, param->getName(), param->getValue());
39  }
40  for (Allocation::const_define_iterator i = alloc->define_begin(), e =
41  alloc->define_end(); i != e; ++i) {
42  std::string name = i->first;
43  std::string value = i->second;
44  replaceAll(num, "`"+name, value);
45  }
46 
47  // handle simple subtraction. ie. num = "64-1" -> final = 63
48  std::stringstream ss(num);
49  std::string s;
50  unsigned final = 0;
51  while (getline(ss, s, '-')) {
52  if (final == 0) {
53  final = atoi(s.c_str());
54  } else {
55  final = final - atoi(s.c_str());
56  }
57  }
58 
59  return final;
60 }

156 RTLOp::RTLOp(Instruction *instr) : castWidth(false) {
157  if (isa<BinaryOperator>(instr)) {
158  switch (instr->getOpcode()) {
159  case Instruction::Add: op = Add; break;
160  case Instruction::Sub: op = Sub; break;
161  case Instruction::Mul: op = Mul; break;
162  case Instruction::SRem:
163  case Instruction::URem: op = Rem; break;
164  case Instruction::SDiv:
165  case Instruction::UDiv: op = Div; break;
166  case Instruction::And: op = And; break;
167  case Instruction::Or: op = Or; break;
168  case Instruction::Xor: op = Xor; break;
169  case Instruction::Shl: op = Shl; break;
170  // arithmetic shift right - MSB replaced with sign bit
171  case Instruction::AShr:
172  case Instruction::LShr: op = Shr; break;
173  case Instruction::FAdd: op = FAdd; break;
174  case Instruction::FMul: op = FMul; break;
175  case Instruction::FSub: op = FSub; break;
176  case Instruction::FDiv: op = FDiv; break;
177  case Instruction::FRem:
178  assert(0 && "Floating point operations not supported\n");
179  default: llvm_unreachable("Invalid operator type!");
180  }
181 
182  } else if (const ICmpInst *cmp = dyn_cast<ICmpInst>(instr)) {
183  switch (cmp->getPredicate()) {
184  case ICmpInst::ICMP_EQ: op = EQ; break;
185  case ICmpInst::ICMP_NE: op = NE; break;
186  case ICmpInst::ICMP_SLT:
187  case ICmpInst::ICMP_ULT: op = LT; break;
188  case ICmpInst::ICMP_SLE:
189  case ICmpInst::ICMP_ULE: op = LE; break;
190  case ICmpInst::ICMP_SGT:
191  case ICmpInst::ICMP_UGT: op = GT; break;
192  case ICmpInst::ICMP_SGE:
193  case ICmpInst::ICMP_UGE: op = GE; break;
194  default: llvm_unreachable("Illegal ICmp predicate");
195  }
196  } else if (const FCmpInst *fcmp = dyn_cast<FCmpInst>(instr)) {
197  switch (fcmp->getPredicate()) {
198  case FCmpInst::FCMP_OEQ:
199  case FCmpInst::FCMP_UEQ: op = EQ; break;
200  case FCmpInst::FCMP_ONE:
201  case FCmpInst::FCMP_UNE: op = NE; break;
202  case FCmpInst::FCMP_OLT:
203  case FCmpInst::FCMP_ULT: op = LT; break;
204  case FCmpInst::FCMP_OLE:
205  case FCmpInst::FCMP_ULE: op = LE; break;
206  case FCmpInst::FCMP_OGT:
207  case FCmpInst::FCMP_UGT: op = GT; break;
208  case FCmpInst::FCMP_OGE:
209  case FCmpInst::FCMP_UGE: op = GE; break;
210  default: llvm_unreachable("Illegal FCmp predicate");
211  }
212  }
213  else if(instr->getOpcode()==Instruction::Select) op=Sel;
214  else {
215  switch (instr->getOpcode()) {
216  case Instruction::FPTrunc: op = FPTrunc; break;
217  case Instruction::FPExt: op = FPExt; break;
218  case Instruction::FPToSI: op = FPToSI; break;
219  case Instruction::SIToFP: op = SIToFP; break;
220  default:
221  errs() << "Unrecognized instruction: " << *instr << "\n";
222  llvm_unreachable(0);
223  }
224  }

516 void RTLModule::recurseBackwards(const RTLSignal *signal) {
517  static std::set<const RTLSignal*> visited;
518  if (visited.find(signal) != visited.end()) return;
519  visited.insert(signal);
520 
521  //errs() << "recurseBackwards: " << signal->getName() << "(" << signal <<
522  //")\n";
523 
524  addCell(signal);
525  if (signal->isOp()) {
526  const RTLOp *op = (const RTLOp*)signal;
527  for (unsigned k = 0; k < op->getNumOperands(); k++) {
528  const RTLSignal *operand = op->getOperand(k);
529  connect(signal, operand);
530  recurseBackwards(operand);
531  }
532  } else {
533 
534  const RTLSignal *defaultDriver = signal->getDefaultDriver();
535  if (defaultDriver) {
536  connect(signal, defaultDriver);
537  recurseBackwards(defaultDriver);
538  }
539 
540  for (unsigned j = 0; j < signal->getNumDrivers(); ++j) {
541  const RTLSignal *driver = signal->getDriver(j);
542  assert(driver);
543  connect(signal, driver);
544  recurseBackwards(driver);
545  }
546 
547  for (unsigned j = 0; j < signal->getNumConditions(); ++j) {
548  const RTLSignal *condition = signal->getCondition(j);
549  assert(condition);
550 
551  connect(signal, condition);
552  recurseBackwards(condition);
553  }
554  }

689 void RTLModule::buildCircuitStructure() {
690  // create a cell for each input/output port
691  for (const_signal_iterator i = port_begin(), e = port_end(); i != e;
692  ++i) {
693  const RTLSignal *s = *i;
694  Cell *c = newCell(s->getName());
695  mapSignalCell[s] = c;
696  if (isInput(s)) {
697  inputs.push_back(c);
698  c->outPins.push_back(new Pin(c, "input"));
699  } else {
700  assert(isOutput(s));
701  outputs.push_back(c);
702  c->outPins.push_back(new Pin(c, "output"));
703  }
704  }
705 
706  // create a cell for each instantiated module
707  std::vector<const RTLSignal *> inst_out_ports;
708  for (const_module_iterator m = instances_begin(), me =
709  instances_end(); m != me; ++m) {
710  RTLModule *mod = *m;
711  Cell *c = newCell(mod->getName());
712  //errs() << "Adding module: " << mod->getName() << "\n";
713  for (RTLModule::const_signal_iterator i = mod->port_begin(), e =
714  mod->port_end(); i != e; ++i) {
715  const RTLSignal *port = *i;
716  mapSignalCell[port] = c;
717  if (isInput(port)) {
718  assert(port->getNumDrivers() == 1);
719  const RTLSignal *driver = port->getDriver(0);
720  connect(port, driver);
721  } else {
722  assert(isOutput(port));
723  inst_out_ports.push_back(port);
724  Pin *pin = new Pin(c, "output");
725  c->outPins.push_back(pin);
726  assert(port->getNumDrivers() == 1);
727  const RTLSignal *output = port->getDriver(0);
728  connect(output, port, pin);
729  }
730  }
731  }
732 
733  for (std::vector<const RTLSignal *>::iterator i = inst_out_ports.begin(), e
734  = inst_out_ports.end(); i != e; ++i) {
735  const RTLSignal *signal = *i;
736  recurseBackwards(signal);
737  }
738 
739  for (const_signal_iterator i = port_begin(), e = port_end(); i != e; ++i) {
740  const RTLSignal *signal = *i;
741  recurseBackwards(signal);
742  }
743 
744  for (const_signal_iterator i = signals.begin(), e = signals.end(); i != e;
745  ++i) {
746  const RTLSignal *signal = *i;
747  recurseBackwards(signal);
748  }
749 
750  // add any printf statements
751  const RTLSignal* unsynth = getUnsynthesizableSignal();
752  assert(unsynth);
753  recurseBackwards(unsynth);
754 }

758 void RTLModule::removeSignalsWithoutFanout() {
759  buildCircuitStructure();
760 
761  std::set<Cell*> marked;
762  std::queue<Cell*> queue;
763  for (std::vector<Cell*>::iterator i = outputs.begin(), e = outputs.end(); i
764  != e; ++i) {
765  queue.push(*i);
766  }
767 
768  // add any printf statements. Signals driving these statements must be kept
769  const RTLSignal* unsynth = getUnsynthesizableSignal();
770  assert(unsynth);
771  Cell *unsynthCell = mapSignalCell[unsynth];
772  assert(unsynthCell);
773  queue.push(unsynthCell);
774 
775  while (!queue.empty()) {
776  Cell *c = queue.front();
777  queue.pop();
778  if (marked.count(c)) continue;
779  marked.insert(c);
780  //errs() << "Visiting: " << c->name << "\n";
781  for (std::vector<Pin*>::iterator p = c->inPins.begin(), pe =
782  c->inPins.end(); p != pe; ++p) {
783  Pin *P = *p;
784  Net *n = P->net;
785  assert(n);
786  Pin *driver = n->driver;
787  assert(driver);
788  Cell *fanout = driver->cell;
789  assert(fanout);
790  queue.push(fanout);
791  }
792  }
793 
794  // remove signals
795  signal_iterator i = signals.begin();
796  while (i != signals.end()) {
797  Cell *c = mapSignalCell[*i];
798  assert(c);
799  if (!marked.count(c)) {
800  //outs() << "Removing signal: " << (*i)->getName() << "\n";
801  delete *i;
802  i = signals.erase(i);
803  } else {
804  ++i;
805  }
806  }

907 void RTLOp::setOperand(int i, RTLSignal *s) {
908  if (getNumOperands() == 0) {
909  switch (op) {
910  case EQ:
911  case NE:
912  case LT:
913  case LE:
914  case GT:
915  case GE:
916  // size 1:
917  setWidth(RTLWidth());
918  break;
919  case Concat:
920  if (!castWidth) {
921  setWidth(s->getWidth());
922  castWidth = true;
923  } else {

925  setWidth(RTLWidth(atoi(this->getWidth().getHi().c_str()) + 1 + atoi(s->getWidth().getHi().c_str()) + 1));

928  case Trunc:
929  case SExt:

938  "\n";
939  assert(0);
940  }
941  break;
942  case Sel:
943  // wait for second operand
944  // first operand should have size 1
945  if (!s->getWidth().str().empty()) {
946  errs() << "First select operand should has size 1!\n";

948  s->getValue() << "width: " << s->getWidth().str() <<
949  "\n";
950  assert(0);
951  }
952  break;
953  default:
954  // size of operand:
955  setWidth(s->getWidth());
956  break;
957  }
958  }
959 
960  if (i == 1 && op == Sel) {
961  setWidth(s->getWidth());
962  }

965 /* if (getWidth().str() != s->getWidth().str()) {

969  */
970  }

972  operands[i] = s;
973 }
