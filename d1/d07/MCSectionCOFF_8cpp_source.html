42 
43  // standard sections don't require the '.section'

47  }
48

63

65  OS << ",";
66  switch (Selection) {

68  OS << "one_only,";
69  break;
70  case COFF::IMAGE_COMDAT_SELECT_ANY:

72  break;
73  case COFF::IMAGE_COMDAT_SELECT_SAME_SIZE:
74  OS << "same_size,";
75  break;
76  case COFF::IMAGE_COMDAT_SELECT_EXACT_MATCH:
77  OS << "same_contents,";
