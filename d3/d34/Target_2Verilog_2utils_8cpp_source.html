33 // An example of an empty basic block at the start of main:
34 // define i32 @main() nounwind {
35 // br label %1
36 // Returns NULL if the BB is not empty
37 BasicBlock *isEmptyFirstBB(BasicBlock *BB) {
38  // basic block should only include a branch instruction
39  if (BB->size() != 1)
40  return NULL;
41 
42  BasicBlock *firstBB = BB->getParent()->begin();
43  if (firstBB != BB)
44  return NULL;
45 
46  // look for an unconditional branch to expected successor
47  TerminatorInst *TI = BB->getTerminator();
48  assert(TI);
49  if (isa<ReturnInst>(TI))
50  return NULL;
51  BranchInst *B = dyn_cast<BranchInst>(TI);
52  if (!B)
53  return NULL; // a switch statement
54  if (TI->getNumSuccessors() != 1) {
55  return NULL;
56  } else {
57  BasicBlock *succ = dyn_cast<BasicBlock>(TI->getSuccessor(0));
58  assert(succ);
59  return succ;
60  }
61 }

626 std::string charToString(unsigned char C, bool &LastWasHex) {
627 
628  std::stringstream Out;
629  // Print it out literally if it is a printable character. The only thing
630  // to be careful about is when the last letter output was a hex escape
631  // code, in which case we have to be careful not to print out hex digits
632  // explicitly (the C compiler thinks it is a continuation of the previous
633  // character, sheesh...)
634  //
635  if (isprint(C) && (!LastWasHex || !isxdigit(C))) {
636  LastWasHex = false;
637  if (C == '"' || C == '\\')
638  Out << "\\" << (char)C;
639  else
640  Out << (char)C;
641  } else {
642  LastWasHex = false;
643  switch (C) {
644  case '\n': Out << "\\n"; break;
645  case '\t': Out << "\\t"; break;
646  case '\r': Out << "\\r"; break;
647  case '\v': Out << "\\v"; break;
648  case '\a': Out << "\\a"; break;
649  case '\"': Out << "\\\""; break;
650  case '\'': Out << "\\\'"; break;
651  default:
652  Out << "\\x";
653  Out << (char)(( C/16 < 10) ? ( C/16 +'0') : ( C/16 -10+'A'));
654  Out << (char)(((C&15) < 10) ? ((C&15)+'0') : ((C&15)-10+'A'));
655  LastWasHex = true;
656  break;
657  }
658  }
659  return Out.str();
660 }

715  for (int i = 0; i < bytes; i++) {
716  *buffer = itohex((*ptr & 0xF0)>>4);

721  }

724  int x = 1;

729  buffer--; // swap two characters at once

731  char tmp = *buffer_start;

734  *(buffer_start+1) = *(buffer+1);
735  *buffer = tmp;
736  *(buffer+1) = tmp2;

738  buffer_start++;
739  buffer--;
