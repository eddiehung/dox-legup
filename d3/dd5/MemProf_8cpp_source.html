63  // adds parallel functions to parallelFunction vector
64  // function is parallel if:
65  // pthread -> called by pthread_create
66  // openmp -> called by GOMP_parallel_start
67  // or called by a parallel function
68  bool isParallelFunctionCall(CallInst *CI) {
69  if (CI->getCalledFunction() == NULL) {
70  errs() << "Could not find function being called\n";
71  return false;
72  }
73 
74  int sizeBefore = pthreadFunction.size() + openmpFunction.size();
75  std::string calledFuncName = CI->getCalledFunction()->getName();
76  Function *pF = NULL;
77  if (calledFuncName == "pthread_create") { // pthread check argument #3
78  // add name of the 3rd argument as parallel function
79  if ((pF = dyn_cast<Function>(CI->getArgOperand(2))) &&
80  !functionInParallelList(pF)) {
81  errs() << "pthread function: " << pF->getName() << "\n";
82  pthreadFunction.push_back(pF);
83  }
84  } else if (calledFuncName == "GOMP_parallel_start") { // omp
85  // add name of the 1st argument as parallel function
86  if ((pF = dyn_cast<Function>(CI->getArgOperand(0))) &&
87  !functionInParallelList(pF)) {
88  errs() << "omp function: " << pF->getName() << "\n";
89  openmpFunction.push_back(pF);
90  }
91  } else if (isPthreadFunction(CI->getParent()->getParent()) &&
92  !isAPrintCall(CI) &&
93  !functionInParallelList(CI->getCalledFunction())) {
94  errs() << "pthread function: " << CI->getCalledFunction()->getName()
95  << "\n";
96  pthreadFunction.push_back(CI->getCalledFunction());
97  } else if (isOpenMPFunction(CI->getParent()->getParent()) &&
98  !isAPrintCall(CI) &&
99  !functionInParallelList(CI->getCalledFunction())) {
100  errs() << "omp function: " << CI->getCalledFunction()->getName()
101  << "\n";
102  openmpFunction.push_back(CI->getCalledFunction());
103  }
104  int sizeAfter = pthreadFunction.size() + openmpFunction.size();
105  if (sizeBefore < sizeAfter) {
106  return true;
107  } else {
108  return false;
109  }
110  }

115  void findParallelFunctions(Module &M) {
116  bool foundParallelFunc = true;
117  while (foundParallelFunc) {
118  foundParallelFunc = false;

253  // get the number of GEP dimensions, and add an arg for each dimension
254  int getGEPDims(GEPOperator *g, std::vector<Value *> &Args) {
255  int num_dims = 0;
256  unsigned int index_offset = 0;
257  PointerType *ty = dyn_cast<PointerType>(g->getPointerOperandType());
258 
259  // if it is an array or vector, the first index, 0, can be skipped
260  // see online LLVM doc: The Often Misunderstood GEP Instruction
261  // http://llvm.org/docs/GetElementPtr.html#why-is-the-extra-0-index-required
262  if (dyn_cast<ArrayType>(ty->getElementType()) ||
263  dyn_cast<VectorType>(ty->getElementType())) {
264  // MY_DEBUG(" array or vector type ");
265  index_offset = 1;
266  }
267 
268  // skip index 0
269  for (unsigned int i = 1 + index_offset; i <= g->getNumIndices(); i++) {
270  Value *operand = g->getOperand(i);
271  if (ConstantInt *ndx = dyn_cast<ConstantInt>(operand)) {
272  Args.push_back(ConstantInt::get(
273  Type::getInt32Ty(getGlobalContext()), ndx->getSExtValue()));
274  MY_DEBUG("[" << ndx->getSExtValue() << "]");
275  } else {
276  Args.push_back(operand);
277  MY_DEBUG("[" << operand->getName() << "]");
278  }
279  num_dims++;
280  }
281  MY_DEBUG("\n");
282  return num_dims;
283  }

285  // determine if phi nodes have identical GEPs feeding each input
286  // this doesn't happend often, but does in some benchmarks
287  bool phi_inputs_are_identical_geps(PHINode *phi) {
288  GEPOperator *g0 = dyn_cast<GEPOperator>(phi->getIncomingValue(0));
289  if (!g0)
290  return false;
291  for (unsigned int i = 1; i < phi->getNumIncomingValues(); i++) {
292  GEPOperator *gi = dyn_cast<GEPOperator>(phi->getIncomingValue(i));
293  if (!gi)
294  return false;
295  // make sure number of indices are the same
296  if (g0->getNumIndices() != gi->getNumIndices())
297  return false;
298  for (unsigned int j = 0; j <= g0->getNumIndices(); j++) {
299  if (g0->getOperand(j) != gi->getOperand(j))
300  return false;
301  } // for j
302  } // for i
303  return true;
304  }

315  // create a printf call for a load or store instruction that shows:
316  // * whether it is a load or store
317  // * the address accessed
318  // * the data read/written
319  // * the name of the variable, with any relavent array indices
320  void printLoadStore(BasicBlock::iterator pos, Instruction *instr) {
321  MY_DEBUG("printLoadStore\n");
322  // want to add call _after_ the instruction
323  bool isParallel = isParallelFunction(instr->getParent()->getParent());
324  IRBuilder<> builder((Instruction *)(++pos));
325  std::vector<Value *> Args;
326  std::string operation;
327  Value *operand, *data;
328  int num_dims = 0;
329 
330  if (isa<LoadInst>(instr)) {
331  operand = instr->getOperand(0);
332  data = instr;
333  operation = "ld";
334  } else if (isa<StoreInst>(instr)) {
335  operand = instr->getOperand(1);
336  data = instr->getOperand(0);
337  operation = "st";
338  }
339 
340  // save space for format string
341  Args.push_back(NULL);
342 
343  // depending on the type of the operand that specifies the address,
344  // we may need to do different things
345  // This can be:
346  // * a GEP operator
347  // * a PHI node
348  // * a pointer
349  // * an instruction
350  // * something else??
351  if (GEPOperator *g = dyn_cast<GEPOperator>(operand)) {
352  MY_DEBUG("GEP: " << g->getOperand(0)->getName());
353  // address
354  Args.push_back((Value *)g);
355  // data
356  Args.push_back(data);
357  // name
358  // if a second level of GEP
359  // TODO: may be more than two levels of GEP?
360  if (GEPOperator *g2 = dyn_cast<GEPOperator>(g->getOperand(0))) {
361  Args.push_back(getOrAddString(
362  g2->getOperand(0)->getName().str(), builder));
363  } else {
364  Args.push_back(
365  getOrAddString(g->getOperand(0)->getName().str(), builder));
366  }
367 
368  // TODO: array indices might be wrong if multiple levels of GEP?
369  num_dims = getGEPDims(g, Args);
370 
371  } // GEP
372  else if (PHINode *phi = dyn_cast<PHINode>(operand)) {
373  // see if we can resolve the phi node, this doesn't happen often
374  if (phi_inputs_are_identical_geps(phi)) {
375  GEPOperator *g =
376  dyn_cast<GEPOperator>(phi->getIncomingValue(0));
377  MY_DEBUG("PHI: " << g->getOperand(0)->getName());
378  // address
379  Args.push_back((Value *)phi);
380  // data
381  Args.push_back(data);
382  // name
383  Args.push_back(
384  getOrAddString(g->getOperand(0)->getName().str(), builder));
385  // dims
386  num_dims = getGEPDims(g, Args);
387  } else {
388  MY_DEBUG("PHI: cannot resolve: " << phi->getName() << "\n");
389  // address
390  Args.push_back((Value *)phi);
391  // data
392  Args.push_back(data);
393  // name
394  Args.push_back(
395  getOrAddString(operand->getName().str(), builder));
396  }
397  } // PHI
398  else if (dyn_cast<PointerType>(operand->getType())) {
399  MY_DEBUG("PTR: " << operand->getName() << "[0]\n");
400  // address
401  Args.push_back(operand);
402  // data
403  Args.push_back(data);
404  // name
405  Args.push_back(getOrAddString(operand->getName().str(), builder));
406  // index 0
407  Args.push_back(
408  ConstantInt::get(Type::getInt32Ty(getGlobalContext()), 0));
409 
410  // if it is just a pointer, index is [0]
411  num_dims = 1;
412  } // PTR
413  else if (Instruction *ins = dyn_cast<Instruction>(operand)) {
414  // address
415  Args.push_back(ins);
416  // data
417  Args.push_back(data);
418  // name
419  Args.push_back(getOrAddString(ins->getName().str(), builder));
420  MY_DEBUG("INST: " << ins->getName() << "\n");
421  } // INST
422  else { // Unknown
423  MY_DEBUG("UNKNOWN: " << operand->getName() << "\n");
424  // address
425  Args.push_back(
426  ConstantInt::get(Type::getInt32Ty(getGlobalContext()), 0));
427  // data
428  Args.push_back(data);
429  // name
430  Args.push_back(getOrAddString(operand->getName().str(), builder));
431  } // UNKNOWN
432 
433  if (isParallel) {
434  if (isPthreadFunction(instr->getParent()->getParent())) {
435  Args.push_back(dyn_cast<Value>(PthreadSelfCall));
436  } else if (isOpenMPFunction(instr->getParent()->getParent())) {
437  Args.push_back(dyn_cast<Value>(OMPThreadNumCall));
438  }
439  }
440 
441  // create store string with appropriate # of dimensions
442  std::string s = "#" + operation;
443  s += " 0x%x %d (%s";
444  for (int i = 0; i < num_dims; i++)
445  s += "[%d]";
446  s += ")";
447  if (isParallel) {
448  s += " {tid: %u}\n";
449  } else {
450  s += "\n";
451  }
452 
453  // format string
454  Args[0] = getOrAddString(s, builder);
455 
456  // Create the printf call
457  builder.CreateCall(PrintFunc, Args);
458  }

460  // Create a printf() call for each alloca instruction that shows:
461  // * the name of the variable
462  // * the base address
463  // * the amount of memory allocated in bytes
464  // TODO: print size split by dimensions, eg. '[2][64]' rather than '128'
465  void printAlloca(BasicBlock::iterator pos, Instruction *i) {
466  MY_DEBUG("printAlloca\n");
467  bool isParallel = isParallelFunction(i->getParent()->getParent());
468  unsigned int size = 0;
469  // want to print call _after_ the instruction
470  IRBuilder<> builder((Instruction *)(++pos));
471  std::vector<Value *> Args;
472  if (AllocaInst *a = dyn_cast<AllocaInst>(i)) {
473  // format string
474  if (isParallel) {
475  Args.push_back(
476  getOrAddString("#alloc %s 0x%x %d {tid: %u}\n", builder));
477  } else {
478  Args.push_back(getOrAddString("#alloc %s 0x%x %d\n", builder));
479  }
480 
481  // variable name
482  Args.push_back(getOrAddString(a->getName().str(), builder));
483 
484  // address
485  Args.push_back(i);
486 
487  Type *t = a->getAllocatedType();
488  MY_DEBUG("alloca " << a->getName() << " ");
489 
490  // TODO: should probably deal with types other than array
491  // if it is an array, get the size in bytes
492  if (t->isArrayTy()) {
493  // get num elements in first dimension
494  size = t->getArrayNumElements();
495  while (t->getArrayElementType()->isArrayTy()) {
496  // go to next dim of array
497  t = t->getArrayElementType();
498  // multiply size by the next dim of array
499  size *= t->getArrayNumElements();
500  }
501  // get element size
502  size *= t->getArrayElementType()->getScalarSizeInBits();
503  // want bytes, not bits
504  size /= 8;
505  // print size
506  MY_DEBUG("size: " << size << "B");
507  }
508 
509  // size
510  Args.push_back(
511  ConstantInt::get(Type::getInt32Ty(getGlobalContext()), size));
512 
513  // thread ID
514  if (isParallel) {
515  if (isPthreadFunction(i->getParent()->getParent())) {
516  Args.push_back(dyn_cast<Value>(PthreadSelfCall));
517  } else if (isOpenMPFunction(i->getParent()->getParent())) {
518  Args.push_back(dyn_cast<Value>(OMPThreadNumCall));
519  }
520  }
521 
522  // create the printf call
523  builder.CreateCall(PrintFunc, Args);
524 
525  MY_DEBUG("\n");
526  }

542  void printCallReturn(Instruction *i) {
543 
544  IRBuilder<> builder(i);
545  std::vector<Value *> Args;
546  std::string inst_type, fn_name;
547

576  virtual bool runOnFunction(Function &F) {
577  bool fn_modified = false;
578  MY_DEBUG("Function:\t" << F.getName() << "\n");
579 
580  // add call to pthread_self() if parallel function
581  if (isPthreadFunction(&F)) {
582  PthreadSelfCall = insertPthreadSelfCall(F);
583  OMPThreadNumCall = NULL;
584  } else if (isOpenMPFunction(&F)) {
585  OMPThreadNumCall = insertOMPThreadNumCall(F);
586  PthreadSelfCall = NULL;
587  } else {
588  PthreadSelfCall = NULL;
589  OMPThreadNumCall = NULL;
590  }
591 
592  for (Function::iterator BB = F.begin(), EE = F.end(); BB != EE; ++BB) {
593  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;
594  ++I) {
595  Instruction *instr = I;
596 
597  if ((isa<LoadInst>(instr) || isa<StoreInst>(instr)) &&
598  !noLoadStore) {
599  printLoadStore(I, instr);
600  fn_modified |= true;
601  } else if (isa<AllocaInst>(instr) && !noAlloca) {
602  printAlloca(I, instr);
603  fn_modified |= true;
604  } else if ((isa<CallInst>(instr) || isa<ReturnInst>(instr)) &&
605  !noCallReturn) {
606  printCallReturn(instr);
607  fn_modified |= true;
608  }
609  }
610  }
611  return fn_modified;
612  }
