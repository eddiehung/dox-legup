65 void HwOnly::getParallelAcceleratedFunctions(Module &M,
66  std::set<GlobalValue*> &HwFcts) {
67 
68  // skip in PCIe flow
69  if (LEGUP_CONFIG->isPCIeFlow()) return;
70 
71  getInternalAccels(M, internalAccels);
72 
73  //designate all omp functions as hardware accelerators
74  for (Module::iterator F = M.begin(), FE = M.end(); F != FE; ++F) {
75  for (Function::iterator BB = F->begin(), EE = F->end(); BB != EE; ++BB) {
76  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;) {
77 
78  if (CallInst *CI = dyn_cast<CallInst>(I++)) {
79 
80  Function *calledFunction = CI->getCalledFunction();
81 
82  // ignore indirect function calls
83  if (!calledFunction) continue;
84 
85  if (getParallelFunctions(M, CI, HwFcts)) continue;
86 
87  }
88  }
89  }
90  }

107  //add noinline attribute to the OMP function, since this should never be inlined

111  // for openmp, this current function is just the wrapper
