36 SIInstrInfo::copyPhysReg(MachineBasicBlock &MBB,
37  MachineBasicBlock::iterator MI, DebugLoc DL,
38  unsigned DestReg, unsigned SrcReg,
39  bool KillSrc) const {
40 
41  // If we are trying to copy to or from SCC, there is a bug somewhere else in
42  // the backend. While it may be theoretically possible to do this, it should
43  // never be necessary.
44  assert(DestReg != AMDGPU::SCC && SrcReg != AMDGPU::SCC);
45 
46  static const int16_t Sub0_15[] = {
47  AMDGPU::sub0, AMDGPU::sub1, AMDGPU::sub2, AMDGPU::sub3,
48  AMDGPU::sub4, AMDGPU::sub5, AMDGPU::sub6, AMDGPU::sub7,
49  AMDGPU::sub8, AMDGPU::sub9, AMDGPU::sub10, AMDGPU::sub11,
50  AMDGPU::sub12, AMDGPU::sub13, AMDGPU::sub14, AMDGPU::sub15, 0
51  };
52 
53  static const int16_t Sub0_7[] = {
54  AMDGPU::sub0, AMDGPU::sub1, AMDGPU::sub2, AMDGPU::sub3,
55  AMDGPU::sub4, AMDGPU::sub5, AMDGPU::sub6, AMDGPU::sub7, 0
56  };
57 
58  static const int16_t Sub0_3[] = {
59  AMDGPU::sub0, AMDGPU::sub1, AMDGPU::sub2, AMDGPU::sub3, 0
60  };
61 
62  static const int16_t Sub0_2[] = {
63  AMDGPU::sub0, AMDGPU::sub1, AMDGPU::sub2, 0
64  };
65 
66  static const int16_t Sub0_1[] = {
67  AMDGPU::sub0, AMDGPU::sub1, 0
68  };
69 
70  unsigned Opcode;
71  const int16_t *SubIndices;
72 
73  if (AMDGPU::M0 == DestReg) {
74  // Check if M0 isn't already set to this value
75  for (MachineBasicBlock::reverse_iterator E = MBB.rend(),
76  I = MachineBasicBlock::reverse_iterator(MI); I != E; ++I) {
77 
78  if (!I->definesRegister(AMDGPU::M0))
79  continue;
80 
81  unsigned Opc = I->getOpcode();
82  if (Opc != TargetOpcode::COPY && Opc != AMDGPU::S_MOV_B32)
83  break;
84 
85  if (!I->readsRegister(SrcReg))
86  break;
87 
88  // The copy isn't necessary
89  return;
90  }
91  }
92 
93  if (AMDGPU::SReg_32RegClass.contains(DestReg)) {
94  assert(AMDGPU::SReg_32RegClass.contains(SrcReg));
95  BuildMI(MBB, MI, DL, get(AMDGPU::S_MOV_B32), DestReg)
96  .addReg(SrcReg, getKillRegState(KillSrc));
97  return;
98 
99  } else if (AMDGPU::SReg_64RegClass.contains(DestReg)) {
100  assert(AMDGPU::SReg_64RegClass.contains(SrcReg));
101  BuildMI(MBB, MI, DL, get(AMDGPU::S_MOV_B64), DestReg)
102  .addReg(SrcReg, getKillRegState(KillSrc));
103  return;
104 
105  } else if (AMDGPU::SReg_128RegClass.contains(DestReg)) {
106  assert(AMDGPU::SReg_128RegClass.contains(SrcReg));
107  Opcode = AMDGPU::S_MOV_B32;
108  SubIndices = Sub0_3;
109 
110  } else if (AMDGPU::SReg_256RegClass.contains(DestReg)) {
111  assert(AMDGPU::SReg_256RegClass.contains(SrcReg));
112  Opcode = AMDGPU::S_MOV_B32;
113  SubIndices = Sub0_7;
114 
115  } else if (AMDGPU::SReg_512RegClass.contains(DestReg)) {
116  assert(AMDGPU::SReg_512RegClass.contains(SrcReg));
117  Opcode = AMDGPU::S_MOV_B32;
118  SubIndices = Sub0_15;
119 
120  } else if (AMDGPU::VReg_32RegClass.contains(DestReg)) {
121  assert(AMDGPU::VReg_32RegClass.contains(SrcReg) ||
122  AMDGPU::SReg_32RegClass.contains(SrcReg));
123  BuildMI(MBB, MI, DL, get(AMDGPU::V_MOV_B32_e32), DestReg)
124  .addReg(SrcReg, getKillRegState(KillSrc));
125  return;
126 
127  } else if (AMDGPU::VReg_64RegClass.contains(DestReg)) {
128  assert(AMDGPU::VReg_64RegClass.contains(SrcReg) ||
129  AMDGPU::SReg_64RegClass.contains(SrcReg));
130  Opcode = AMDGPU::V_MOV_B32_e32;
131  SubIndices = Sub0_1;
132 
133  } else if (AMDGPU::VReg_96RegClass.contains(DestReg)) {
134  assert(AMDGPU::VReg_96RegClass.contains(SrcReg));
135  Opcode = AMDGPU::V_MOV_B32_e32;
136  SubIndices = Sub0_2;
137 
138  } else if (AMDGPU::VReg_128RegClass.contains(DestReg)) {
139  assert(AMDGPU::VReg_128RegClass.contains(SrcReg) ||
140  AMDGPU::SReg_128RegClass.contains(SrcReg));
141  Opcode = AMDGPU::V_MOV_B32_e32;
142  SubIndices = Sub0_3;
143 
144  } else if (AMDGPU::VReg_256RegClass.contains(DestReg)) {
145  assert(AMDGPU::VReg_256RegClass.contains(SrcReg) ||
146  AMDGPU::SReg_256RegClass.contains(SrcReg));
147  Opcode = AMDGPU::V_MOV_B32_e32;
148  SubIndices = Sub0_7;
149 
150  } else if (AMDGPU::VReg_512RegClass.contains(DestReg)) {
151  assert(AMDGPU::VReg_512RegClass.contains(SrcReg) ||
152  AMDGPU::SReg_512RegClass.contains(SrcReg));
153  Opcode = AMDGPU::V_MOV_B32_e32;
154  SubIndices = Sub0_15;
155 
156  } else {
157  llvm_unreachable("Can't copy register!");
158  }
159 
160  while (unsigned SubIdx = *SubIndices++) {
161  MachineInstrBuilder Builder = BuildMI(MBB, MI, DL,
162  get(Opcode), RI.getSubReg(DestReg, SubIdx));
163 
164  Builder.addReg(RI.getSubReg(SrcReg, SubIdx), getKillRegState(KillSrc));
165 
166  if (*SubIndices)
167  Builder.addReg(DestReg, RegState::Define | RegState::Implicit);
168  }

185 void SIInstrInfo::storeRegToStackSlot(MachineBasicBlock &MBB,
186  MachineBasicBlock::iterator MI,
187  unsigned SrcReg, bool isKill,
188  int FrameIndex,
189  const TargetRegisterClass *RC,
190  const TargetRegisterInfo *TRI) const {
191  MachineFunction *MF = MBB.getParent();
192  SIMachineFunctionInfo *MFI = MF->getInfo<SIMachineFunctionInfo>();
193  MachineRegisterInfo &MRI = MF->getRegInfo();
194  DebugLoc DL = MBB.findDebugLoc(MI);
195  unsigned KillFlag = isKill ? RegState::Kill : 0;
196 
197  if (RI.hasVGPRs(RC)) {
198  LLVMContext &Ctx = MF->getFunction()->getContext();
199  Ctx.emitError("SIInstrInfo::storeRegToStackSlot - Can't spill VGPR!");
200  BuildMI(MBB, MI, DL, get(AMDGPU::V_MOV_B32_e32), AMDGPU::VGPR0)
201  .addReg(SrcReg);
202  } else if (TRI->getCommonSubClass(RC, &AMDGPU::SGPR_32RegClass)) {
203  unsigned Lane = MFI->SpillTracker.reserveLanes(MRI, MF);
204  unsigned TgtReg = MFI->SpillTracker.LaneVGPR;
205 
206  BuildMI(MBB, MI, DL, get(AMDGPU::V_WRITELANE_B32), TgtReg)
207  .addReg(SrcReg, KillFlag)
208  .addImm(Lane);
209  MFI->SpillTracker.addSpilledReg(FrameIndex, TgtReg, Lane);
210  } else if (RI.isSGPRClass(RC)) {
211  // We are only allowed to create one new instruction when spilling
212  // registers, so we need to use pseudo instruction for vector
213  // registers.
214  //
215  // Reserve a spot in the spill tracker for each sub-register of
216  // the vector register.
217  unsigned NumSubRegs = RC->getSize() / 4;
218  unsigned FirstLane = MFI->SpillTracker.reserveLanes(MRI, MF, NumSubRegs);
219  MFI->SpillTracker.addSpilledReg(FrameIndex, MFI->SpillTracker.LaneVGPR,
220  FirstLane);
221 
222  unsigned Opcode;
223  switch (RC->getSize() * 8) {
224  case 64: Opcode = AMDGPU::SI_SPILL_S64_SAVE; break;
225  case 128: Opcode = AMDGPU::SI_SPILL_S128_SAVE; break;
226  case 256: Opcode = AMDGPU::SI_SPILL_S256_SAVE; break;
227  case 512: Opcode = AMDGPU::SI_SPILL_S512_SAVE; break;
228  default: llvm_unreachable("Cannot spill register class");
229  }
230 
231  BuildMI(MBB, MI, DL, get(Opcode), MFI->SpillTracker.LaneVGPR)
232  .addReg(SrcReg)
233  .addImm(FrameIndex);
234  } else {
235  llvm_unreachable("VGPR spilling not supported");
236  }
237 }

240  MachineBasicBlock::iterator MI,
241  unsigned DestReg, int FrameIndex,
242  const TargetRegisterClass *RC,
243  const TargetRegisterInfo *TRI) const {
244  MachineFunction *MF = MBB.getParent();
245  SIMachineFunctionInfo *MFI = MF->getInfo<SIMachineFunctionInfo>();
246  DebugLoc DL = MBB.findDebugLoc(MI);
247 
248  if (RI.hasVGPRs(RC)) {
249  LLVMContext &Ctx = MF->getFunction()->getContext();
250  Ctx.emitError("SIInstrInfo::loadRegToStackSlot - Can't retrieve spilled VGPR!");
251  BuildMI(MBB, MI, DL, get(AMDGPU::V_MOV_B32_e32), DestReg)

253  } else if (RI.isSGPRClass(RC)){
254  unsigned Opcode;
255  switch(RC->getSize() * 8) {
256  case 32: Opcode = AMDGPU::SI_SPILL_S32_RESTORE; break;
257  case 64: Opcode = AMDGPU::SI_SPILL_S64_RESTORE; break;
258  case 128: Opcode = AMDGPU::SI_SPILL_S128_RESTORE; break;
259  case 256: Opcode = AMDGPU::SI_SPILL_S256_RESTORE; break;
260  case 512: Opcode = AMDGPU::SI_SPILL_S512_RESTORE; break;
261  default: llvm_unreachable("Cannot spill register class");
262  }
263

265  MFI->SpillTracker.getSpilledReg(FrameIndex);
266 
267  BuildMI(MBB, MI, DL, get(Opcode), DestReg)

272  }

310 bool SIInstrInfo::expandPostRAPseudo(MachineBasicBlock::iterator MI) const {
311  SIMachineFunctionInfo *MFI =
312  MI->getParent()->getParent()->getInfo<SIMachineFunctionInfo>();
313  MachineBasicBlock &MBB = *MI->getParent();
314  DebugLoc DL = MBB.findDebugLoc(MI);
315  switch (MI->getOpcode()) {
316  default: return AMDGPUInstrInfo::expandPostRAPseudo(MI);
317 
318  // SGPR register spill
319  case AMDGPU::SI_SPILL_S512_SAVE:
320  case AMDGPU::SI_SPILL_S256_SAVE:
321  case AMDGPU::SI_SPILL_S128_SAVE:
322  case AMDGPU::SI_SPILL_S64_SAVE: {
323  unsigned NumSubRegs = getNumSubRegsForSpillOp(MI->getOpcode());
324  unsigned FrameIndex = MI->getOperand(2).getImm();
325 
326  for (unsigned i = 0, e = NumSubRegs; i < e; ++i) {
327  SIMachineFunctionInfo::SpilledReg Spill;
328  unsigned SubReg = RI.getPhysRegSubReg(MI->getOperand(1).getReg(),
329  &AMDGPU::SGPR_32RegClass, i);
330  Spill = MFI->SpillTracker.getSpilledReg(FrameIndex);
331 
332  BuildMI(MBB, MI, DL, get(AMDGPU::V_WRITELANE_B32),
333  MI->getOperand(0).getReg())
334  .addReg(SubReg)
335  .addImm(Spill.Lane + i);
336  }
337  MI->eraseFromParent();
338  break;
339  }
340 
341  // SGPR register restore
342  case AMDGPU::SI_SPILL_S512_RESTORE:
343  case AMDGPU::SI_SPILL_S256_RESTORE:
344  case AMDGPU::SI_SPILL_S128_RESTORE:
345  case AMDGPU::SI_SPILL_S64_RESTORE:
346  case AMDGPU::SI_SPILL_S32_RESTORE: {
347  unsigned NumSubRegs = getNumSubRegsForSpillOp(MI->getOpcode());
348 
349  for (unsigned i = 0, e = NumSubRegs; i < e; ++i) {
350  SIMachineFunctionInfo::SpilledReg Spill;
351  unsigned FrameIndex = MI->getOperand(2).getImm();
352  unsigned SubReg = RI.getPhysRegSubReg(MI->getOperand(0).getReg(),
353  &AMDGPU::SGPR_32RegClass, i);
354  Spill = MFI->SpillTracker.getSpilledReg(FrameIndex);
355 
356  BuildMI(MBB, MI, DL, get(AMDGPU::V_READLANE_B32), SubReg)
357  .addReg(MI->getOperand(1).getReg())
358  .addImm(Spill.Lane + i);
359  }
360  insertNOPs(MI, 3);
361  MI->eraseFromParent();
362  break;
363  }
364  case AMDGPU::SI_CONSTDATA_PTR: {
365  unsigned Reg = MI->getOperand(0).getReg();
366  unsigned RegLo = RI.getSubReg(Reg, AMDGPU::sub0);
367  unsigned RegHi = RI.getSubReg(Reg, AMDGPU::sub1);
368 
369  BuildMI(MBB, MI, DL, get(AMDGPU::S_GETPC_B64), Reg);
370 
371  // Add 32-bit offset from this instruction to the start of the constant data.
372  BuildMI(MBB, MI, DL, get(AMDGPU::S_ADD_I32), RegLo)
373  .addReg(RegLo)
374  .addTargetIndex(AMDGPU::TI_CONSTDATA_START)
375  .addReg(AMDGPU::SCC, RegState::Define | RegState::Implicit);
376  BuildMI(MBB, MI, DL, get(AMDGPU::S_ADDC_U32), RegHi)
377  .addReg(RegHi)
378  .addImm(0)
379  .addReg(AMDGPU::SCC, RegState::Define | RegState::Implicit)
380  .addReg(AMDGPU::SCC, RegState::Implicit);
381  MI->eraseFromParent();
382  break;
383  }
384  }
385  return true;
386 }

388 MachineInstr *SIInstrInfo::commuteInstruction(MachineInstr *MI,
389  bool NewMI) const {
390 
391  MachineRegisterInfo &MRI = MI->getParent()->getParent()->getRegInfo();
392  if (MI->getNumOperands() < 3 || !MI->getOperand(1).isReg())
393  return nullptr;
394 
395  // Cannot commute VOP2 if src0 is SGPR.
396  if (isVOP2(MI->getOpcode()) && MI->getOperand(1).isReg() &&
397  RI.isSGPRClass(MRI.getRegClass(MI->getOperand(1).getReg())))
398  return nullptr;
399 
400  if (!MI->getOperand(2).isReg()) {
401  // XXX: Commute instructions with FPImm operands
402  if (NewMI || MI->getOperand(2).isFPImm() ||
403  (!isVOP2(MI->getOpcode()) && !isVOP3(MI->getOpcode()))) {
404  return nullptr;
405  }
406 
407  // XXX: Commute VOP3 instructions with abs and neg set.
408  if (isVOP3(MI->getOpcode()) &&
409  (MI->getOperand(AMDGPU::getNamedOperandIdx(MI->getOpcode(),
410  AMDGPU::OpName::abs)).getImm() ||
411  MI->getOperand(AMDGPU::getNamedOperandIdx(MI->getOpcode(),
412  AMDGPU::OpName::neg)).getImm()))
413  return nullptr;
414 
415  unsigned Reg = MI->getOperand(1).getReg();
416  unsigned SubReg = MI->getOperand(1).getSubReg();
417  MI->getOperand(1).ChangeToImmediate(MI->getOperand(2).getImm());
418  MI->getOperand(2).ChangeToRegister(Reg, false);
419  MI->getOperand(2).setSubReg(SubReg);
420  } else {
421  MI = TargetInstrInfo::commuteInstruction(MI, NewMI);
422  }
423 
424  if (MI)
425  MI->setDesc(get(commuteOpcode(MI->getOpcode())));
426 
427  return MI;
428 }

506 bool SIInstrInfo::isInlineConstant(const APInt &Imm) const {
507  int32_t Val = Imm.getSExtValue();
508  if (Val >= -16 && Val <= 64)
509  return true;
510 
511  // The actual type of the operand does not seem to matter as long
512  // as the bits match one of the inline immediate values. For example:
513  //
514  // -nan has the hexadecimal encoding of 0xfffffffe which is -2 in decimal,
515  // so it is a legal inline immediate.
516  //
517  // 1065353216 has the hexadecimal encoding 0x3f800000 which is 1.0f in
518  // floating-point, so it is a legal inline immediate.
519 
520  return (APInt::floatToBits(0.0f) == Imm) ||
521  (APInt::floatToBits(1.0f) == Imm) ||
522  (APInt::floatToBits(-1.0f) == Imm) ||
523  (APInt::floatToBits(0.5f) == Imm) ||
524  (APInt::floatToBits(-0.5f) == Imm) ||
525  (APInt::floatToBits(2.0f) == Imm) ||
526  (APInt::floatToBits(-2.0f) == Imm) ||
527  (APInt::floatToBits(4.0f) == Imm) ||
528  (APInt::floatToBits(-4.0f) == Imm);
529 }

547 static bool compareMachineOp(const MachineOperand &Op0,
548  const MachineOperand &Op1) {

583 bool SIInstrInfo::verifyInstruction(const MachineInstr *MI,
584  StringRef &ErrInfo) const {
585  uint16_t Opcode = MI->getOpcode();
586  int Src0Idx = AMDGPU::getNamedOperandIdx(Opcode, AMDGPU::OpName::src0);
587  int Src1Idx = AMDGPU::getNamedOperandIdx(Opcode, AMDGPU::OpName::src1);
588  int Src2Idx = AMDGPU::getNamedOperandIdx(Opcode, AMDGPU::OpName::src2);
589 
590  // Make sure the number of operands is correct.
591  const MCInstrDesc &Desc = get(Opcode);
592  if (!Desc.isVariadic() &&
593  Desc.getNumOperands() != MI->getNumExplicitOperands()) {
594  ErrInfo = "Instruction has wrong number of operands.";
595  return false;
596  }
597 
598  // Make sure the register classes are correct
599  for (unsigned i = 0, e = Desc.getNumOperands(); i != e; ++i) {
600  switch (Desc.OpInfo[i].OperandType) {
601  case MCOI::OPERAND_REGISTER: {
602  int RegClass = Desc.OpInfo[i].RegClass;
603  if (!RI.regClassCanUseImmediate(RegClass) &&
604  (MI->getOperand(i).isImm() || MI->getOperand(i).isFPImm())) {
605  ErrInfo = "Expected register, but got immediate";
606  return false;
607  }
608  }
609  break;
610  case MCOI::OPERAND_IMMEDIATE:
611  // Check if this operand is an immediate.
612  // FrameIndex operands will be replaced by immediates, so they are
613  // allowed.
614  if (!MI->getOperand(i).isImm() && !MI->getOperand(i).isFPImm() &&
615  !MI->getOperand(i).isFI()) {
616  ErrInfo = "Expected immediate, but got non-immediate";
617  return false;
618  }
619  // Fall-through
620  default:
621  continue;
622  }
623 
624  if (!MI->getOperand(i).isReg())
625  continue;
626 
627  int RegClass = Desc.OpInfo[i].RegClass;
628  if (RegClass != -1) {
629  unsigned Reg = MI->getOperand(i).getReg();
630  if (TargetRegisterInfo::isVirtualRegister(Reg))
631  continue;
632 
633  const TargetRegisterClass *RC = RI.getRegClass(RegClass);
634  if (!RC->contains(Reg)) {
635  ErrInfo = "Operand has incorrect register class.";
636  return false;
637  }
638  }
639  }
640 
641 
642  // Verify VOP*
643  if (isVOP1(Opcode) || isVOP2(Opcode) || isVOP3(Opcode) || isVOPC(Opcode)) {
644  unsigned ConstantBusCount = 0;
645  unsigned SGPRUsed = AMDGPU::NoRegister;
646  for (int i = 0, e = MI->getNumOperands(); i != e; ++i) {
647  const MachineOperand &MO = MI->getOperand(i);
648  if (MO.isReg() && MO.isUse() &&
649  !TargetRegisterInfo::isVirtualRegister(MO.getReg())) {
650 
651  // EXEC register uses the constant bus.
652  if (!MO.isImplicit() && MO.getReg() == AMDGPU::EXEC)
653  ++ConstantBusCount;
654 
655  // SGPRs use the constant bus
656  if (MO.getReg() == AMDGPU::M0 || MO.getReg() == AMDGPU::VCC ||
657  (!MO.isImplicit() &&
658  (AMDGPU::SGPR_32RegClass.contains(MO.getReg()) ||
659  AMDGPU::SGPR_64RegClass.contains(MO.getReg())))) {
660  if (SGPRUsed != MO.getReg()) {
661  ++ConstantBusCount;
662  SGPRUsed = MO.getReg();
663  }
664  }
665  }
666  // Literal constants use the constant bus.
667  if (isLiteralConstant(MO))
668  ++ConstantBusCount;
669  }
670  if (ConstantBusCount > 1) {
671  ErrInfo = "VOP* instruction uses the constant bus more than once";
672  return false;
673  }
674  }
675 
676  // Verify SRC1 for VOP2 and VOPC
677  if (Src1Idx != -1 && (isVOP2(Opcode) || isVOPC(Opcode))) {
678  const MachineOperand &Src1 = MI->getOperand(Src1Idx);
679  if (Src1.isImm() || Src1.isFPImm()) {
680  ErrInfo = "VOP[2C] src1 cannot be an immediate.";
681  return false;
682  }
683  }
684 
685  // Verify VOP3
686  if (isVOP3(Opcode)) {
687  if (Src0Idx != -1 && isLiteralConstant(MI->getOperand(Src0Idx))) {
688  ErrInfo = "VOP3 src0 cannot be a literal constant.";
689  return false;
690  }
691  if (Src1Idx != -1 && isLiteralConstant(MI->getOperand(Src1Idx))) {
692  ErrInfo = "VOP3 src1 cannot be a literal constant.";
693  return false;
694  }
695  if (Src2Idx != -1 && isLiteralConstant(MI->getOperand(Src2Idx))) {
696  ErrInfo = "VOP3 src2 cannot be a literal constant.";
697  return false;
698  }
699  }
700 
701  // Verify misc. restrictions on specific instructions.
702  if (Desc.getOpcode() == AMDGPU::V_DIV_SCALE_F32 ||
703  Desc.getOpcode() == AMDGPU::V_DIV_SCALE_F64) {
704  MI->dump();
705 
706  const MachineOperand &Src0 = MI->getOperand(2);
707  const MachineOperand &Src1 = MI->getOperand(3);
708  const MachineOperand &Src2 = MI->getOperand(4);
709  if (Src0.isReg() && Src1.isReg() && Src2.isReg()) {
710  if (!compareMachineOp(Src0, Src1) &&
711  !compareMachineOp(Src0, Src2)) {
712  ErrInfo = "v_div_scale_{f32|f64} require src0 = src1 or src2";
713  return false;
714  }
715  }
716  }
717 
718  return true;
719 }

899 void SIInstrInfo::legalizeOperands(MachineInstr *MI) const {
900  MachineRegisterInfo &MRI = MI->getParent()->getParent()->getRegInfo();
901  int Src0Idx = AMDGPU::getNamedOperandIdx(MI->getOpcode(),
902  AMDGPU::OpName::src0);
903  int Src1Idx = AMDGPU::getNamedOperandIdx(MI->getOpcode(),
904  AMDGPU::OpName::src1);
905  int Src2Idx = AMDGPU::getNamedOperandIdx(MI->getOpcode(),
906  AMDGPU::OpName::src2);
907 
908  // Legalize VOP2
909  if (isVOP2(MI->getOpcode()) && Src1Idx != -1) {
910  MachineOperand &Src0 = MI->getOperand(Src0Idx);
911  MachineOperand &Src1 = MI->getOperand(Src1Idx);
912 
913  // If the instruction implicitly reads VCC, we can't have any SGPR operands,
914  // so move any.
915  bool ReadsVCC = MI->readsRegister(AMDGPU::VCC, &RI);
916  if (ReadsVCC && Src0.isReg() &&
917  RI.isSGPRClass(MRI.getRegClass(Src0.getReg()))) {
918  legalizeOpWithMove(MI, Src0Idx);
919  return;
920  }
921 
922  if (ReadsVCC && Src1.isReg() &&
923  RI.isSGPRClass(MRI.getRegClass(Src1.getReg()))) {
924  legalizeOpWithMove(MI, Src1Idx);
925  return;
926  }
927 
928  // Legalize VOP2 instructions where src1 is not a VGPR. An SGPR input must
929  // be the first operand, and there can only be one.
930  if (Src1.isImm() || Src1.isFPImm() ||
931  (Src1.isReg() && RI.isSGPRClass(MRI.getRegClass(Src1.getReg())))) {
932  if (MI->isCommutable()) {
933  if (commuteInstruction(MI))
934  return;
935  }
936  legalizeOpWithMove(MI, Src1Idx);
937  }
938  }
939 
940  // XXX - Do any VOP3 instructions read VCC?
941  // Legalize VOP3
942  if (isVOP3(MI->getOpcode())) {
943  int VOP3Idx[3] = {Src0Idx, Src1Idx, Src2Idx};
944  unsigned SGPRReg = AMDGPU::NoRegister;
945  for (unsigned i = 0; i < 3; ++i) {
946  int Idx = VOP3Idx[i];
947  if (Idx == -1)
948  continue;
949  MachineOperand &MO = MI->getOperand(Idx);
950 
951  if (MO.isReg()) {
952  if (!RI.isSGPRClass(MRI.getRegClass(MO.getReg())))
953  continue; // VGPRs are legal
954 
955  assert(MO.getReg() != AMDGPU::SCC && "SCC operand to VOP3 instruction");
956 
957  if (SGPRReg == AMDGPU::NoRegister || SGPRReg == MO.getReg()) {
958  SGPRReg = MO.getReg();
959  // We can use one SGPR in each VOP3 instruction.
960  continue;
961  }
962  } else if (!isLiteralConstant(MO)) {
963  // If it is not a register and not a literal constant, then it must be
964  // an inline constant which is always legal.
965  continue;
966  }
967  // If we make it this far, then the operand is not legal and we must
968  // legalize it.
969  legalizeOpWithMove(MI, Idx);
970  }
971  }
972 
973  // Legalize REG_SEQUENCE and PHI
974  // The register class of the operands much be the same type as the register
975  // class of the output.
976  if (MI->getOpcode() == AMDGPU::REG_SEQUENCE ||
977  MI->getOpcode() == AMDGPU::PHI) {
978  const TargetRegisterClass *RC = nullptr, *SRC = nullptr, *VRC = nullptr;
979  for (unsigned i = 1, e = MI->getNumOperands(); i != e; i+=2) {
980  if (!MI->getOperand(i).isReg() ||
981  !TargetRegisterInfo::isVirtualRegister(MI->getOperand(i).getReg()))
982  continue;
983  const TargetRegisterClass *OpRC =
984  MRI.getRegClass(MI->getOperand(i).getReg());
985  if (RI.hasVGPRs(OpRC)) {
986  VRC = OpRC;
987  } else {
988  SRC = OpRC;
989  }
990  }
991 
992  // If any of the operands are VGPR registers, then they all most be
993  // otherwise we will create illegal VGPR->SGPR copies when legalizing
994  // them.
995  if (VRC || !RI.isSGPRClass(getOpRegClass(*MI, 0))) {
996  if (!VRC) {
997  assert(SRC);
998  VRC = RI.getEquivalentVGPRClass(SRC);
999  }
1000  RC = VRC;
1001  } else {
1002  RC = SRC;
1003  }
1004 
1005  // Update all the operands so they have the same type.
1006  for (unsigned i = 1, e = MI->getNumOperands(); i != e; i+=2) {
1007  if (!MI->getOperand(i).isReg() ||
1008  !TargetRegisterInfo::isVirtualRegister(MI->getOperand(i).getReg()))
1009  continue;
1010  unsigned DstReg = MRI.createVirtualRegister(RC);
1011  MachineBasicBlock *InsertBB;
1012  MachineBasicBlock::iterator Insert;
1013  if (MI->getOpcode() == AMDGPU::REG_SEQUENCE) {
1014  InsertBB = MI->getParent();
1015  Insert = MI;
1016  } else {
1017  // MI is a PHI instruction.
1018  InsertBB = MI->getOperand(i + 1).getMBB();
1019  Insert = InsertBB->getFirstTerminator();
1020  }
1021  BuildMI(*InsertBB, Insert, MI->getDebugLoc(),
1022  get(AMDGPU::COPY), DstReg)
1023  .addOperand(MI->getOperand(i));
1024  MI->getOperand(i).setReg(DstReg);
1025  }
1026  }
1027 
1028  // Legalize INSERT_SUBREG
1029  // src0 must have the same register class as dst
1030  if (MI->getOpcode() == AMDGPU::INSERT_SUBREG) {
1031  unsigned Dst = MI->getOperand(0).getReg();
1032  unsigned Src0 = MI->getOperand(1).getReg();
1033  const TargetRegisterClass *DstRC = MRI.getRegClass(Dst);
1034  const TargetRegisterClass *Src0RC = MRI.getRegClass(Src0);
1035  if (DstRC != Src0RC) {
1036  MachineBasicBlock &MBB = *MI->getParent();
1037  unsigned NewSrc0 = MRI.createVirtualRegister(DstRC);
1038  BuildMI(MBB, MI, MI->getDebugLoc(), get(AMDGPU::COPY), NewSrc0)
1039  .addReg(Src0);
1040  MI->getOperand(1).setReg(NewSrc0);
1041  }
1042  return;
1043  }
1044 
1045  // Legalize MUBUF* instructions
1046  // FIXME: If we start using the non-addr64 instructions for compute, we
1047  // may need to legalize them here.
1048 
1049  int SRsrcIdx = AMDGPU::getNamedOperandIdx(MI->getOpcode(),
1050  AMDGPU::OpName::srsrc);
1051  int VAddrIdx = AMDGPU::getNamedOperandIdx(MI->getOpcode(),
1052  AMDGPU::OpName::vaddr);
1053  if (SRsrcIdx != -1 && VAddrIdx != -1) {
1054  const TargetRegisterClass *VAddrRC =
1055  RI.getRegClass(get(MI->getOpcode()).OpInfo[VAddrIdx].RegClass);
1056 
1057  if(VAddrRC->getSize() == 8 &&
1058  MRI.getRegClass(MI->getOperand(SRsrcIdx).getReg()) != VAddrRC) {
1059  // We have a MUBUF instruction that uses a 64-bit vaddr register and
1060  // srsrc has the incorrect register class. In order to fix this, we
1061  // need to extract the pointer from the resource descriptor (srsrc),
1062  // add it to the value of vadd, then store the result in the vaddr
1063  // operand. Then, we need to set the pointer field of the resource
1064  // descriptor to zero.
1065 
1066  MachineBasicBlock &MBB = *MI->getParent();
1067  MachineOperand &SRsrcOp = MI->getOperand(SRsrcIdx);
1068  MachineOperand &VAddrOp = MI->getOperand(VAddrIdx);
1069  unsigned SRsrcPtrLo, SRsrcPtrHi, VAddrLo, VAddrHi;
1070  unsigned NewVAddrLo = MRI.createVirtualRegister(&AMDGPU::VReg_32RegClass);
1071  unsigned NewVAddrHi = MRI.createVirtualRegister(&AMDGPU::VReg_32RegClass);
1072  unsigned NewVAddr = MRI.createVirtualRegister(&AMDGPU::VReg_64RegClass);
1073  unsigned Zero64 = MRI.createVirtualRegister(&AMDGPU::SReg_64RegClass);
1074  unsigned SRsrcFormatLo = MRI.createVirtualRegister(&AMDGPU::SGPR_32RegClass);
1075  unsigned SRsrcFormatHi = MRI.createVirtualRegister(&AMDGPU::SGPR_32RegClass);
1076  unsigned NewSRsrc = MRI.createVirtualRegister(&AMDGPU::SReg_128RegClass);
1077 
1078  // SRsrcPtrLo = srsrc:sub0
1079  SRsrcPtrLo = buildExtractSubReg(MI, MRI, SRsrcOp,
1080  &AMDGPU::VReg_128RegClass, AMDGPU::sub0, &AMDGPU::VReg_32RegClass);
1081 
1082  // SRsrcPtrHi = srsrc:sub1
1083  SRsrcPtrHi = buildExtractSubReg(MI, MRI, SRsrcOp,
1084  &AMDGPU::VReg_128RegClass, AMDGPU::sub1, &AMDGPU::VReg_32RegClass);
1085 
1086  // VAddrLo = vaddr:sub0
1087  VAddrLo = buildExtractSubReg(MI, MRI, VAddrOp,
1088  &AMDGPU::VReg_64RegClass, AMDGPU::sub0, &AMDGPU::VReg_32RegClass);
1089 
1090  // VAddrHi = vaddr:sub1
1091  VAddrHi = buildExtractSubReg(MI, MRI, VAddrOp,

1150 void SIInstrInfo::moveSMRDToVALU(MachineInstr *MI, MachineRegisterInfo &MRI) const {

1153  case AMDGPU::S_LOAD_DWORD_IMM:
1154  case AMDGPU::S_LOAD_DWORD_SGPR:
1155  case AMDGPU::S_LOAD_DWORDX2_IMM:
1156  case AMDGPU::S_LOAD_DWORDX2_SGPR:
1157  case AMDGPU::S_LOAD_DWORDX4_IMM:

1166  } else {

1176  } else {
1177  BuildMI(*MBB, MI, MI->getDebugLoc(), get(AMDGPU::S_MOV_B32),
1178  RegOffset)
1179  .addImm(ImmOffset);
1180  ImmOffset = 0;
1181  }
1182  }
1183

1186  unsigned DWord1 = MRI.createVirtualRegister(&AMDGPU::SGPR_32RegClass);
1187  unsigned DWord2 = MRI.createVirtualRegister(&AMDGPU::SGPR_32RegClass);
1188  unsigned DWord3 = MRI.createVirtualRegister(&AMDGPU::SGPR_32RegClass);

1194  BuildMI(*MBB, MI, MI->getDebugLoc(), get(AMDGPU::S_MOV_B32), DWord3)
1195  .addImm(AMDGPU::RSRC_DATA_FORMAT >> 32);
1196  BuildMI(*MBB, MI, MI->getDebugLoc(), get(AMDGPU::REG_SEQUENCE), SRsrc)
1197  .addReg(DWord0)
1198  .addImm(AMDGPU::sub0)
1199  .addReg(DWord1)
1200  .addImm(AMDGPU::sub1)
1201  .addReg(DWord2)
1202  .addImm(AMDGPU::sub2)
1203  .addReg(DWord3)
1204  .addImm(AMDGPU::sub3);
1205  MI->setDesc(get(NewOpcode));
1206  if (MI->getOperand(2).isReg()) {
1207  MI->getOperand(2).setReg(MI->getOperand(1).getReg());
1208  } else {
1209  MI->getOperand(2).ChangeToRegister(MI->getOperand(1).getReg(), false);
1210  }
1211  MI->getOperand(1).setReg(SRsrc);
1212  MI->addOperand(*MBB->getParent(), MachineOperand::CreateImm(ImmOffset));
1213  }

1216 void SIInstrInfo::moveToVALU(MachineInstr &TopInst) const {
1217  SmallVector<MachineInstr *, 128> Worklist;
1218  Worklist.push_back(&TopInst);
1219 
1220  while (!Worklist.empty()) {
1221  MachineInstr *Inst = Worklist.pop_back_val();
1222  MachineBasicBlock *MBB = Inst->getParent();
1223  MachineRegisterInfo &MRI = MBB->getParent()->getRegInfo();
1224 
1225  unsigned Opcode = Inst->getOpcode();
1226  unsigned NewOpcode = getVALUOp(*Inst);
1227 
1228  // Handle some special cases
1229  switch (Opcode) {
1230  default:
1231  if (isSMRD(Inst->getOpcode())) {
1232  moveSMRDToVALU(Inst, MRI);
1233  }
1234  break;
1235  case AMDGPU::S_MOV_B64: {
1236  DebugLoc DL = Inst->getDebugLoc();
1237 
1238  // If the source operand is a register we can replace this with a
1239  // copy.
1240  if (Inst->getOperand(1).isReg()) {
1241  MachineInstr *Copy = BuildMI(*MBB, Inst, DL, get(TargetOpcode::COPY))
1242  .addOperand(Inst->getOperand(0))
1243  .addOperand(Inst->getOperand(1));
1244  Worklist.push_back(Copy);
1245  } else {
1246  // Otherwise, we need to split this into two movs, because there is
1247  // no 64-bit VALU move instruction.
1248  unsigned Reg = Inst->getOperand(0).getReg();
1249  unsigned Dst = split64BitImm(Worklist,
1250  Inst,
1251  MRI,
1252  MRI.getRegClass(Reg),
1253  Inst->getOperand(1));
1254  MRI.replaceRegWith(Reg, Dst);
1255  }
1256  Inst->eraseFromParent();
1257  continue;
1258  }
1259  case AMDGPU::S_AND_B64:
1260  splitScalar64BitBinaryOp(Worklist, Inst, AMDGPU::S_AND_B32);
1261  Inst->eraseFromParent();
1262  continue;
1263 
1264  case AMDGPU::S_OR_B64:
1265  splitScalar64BitBinaryOp(Worklist, Inst, AMDGPU::S_OR_B32);
1266  Inst->eraseFromParent();
1267  continue;
1268 
1269  case AMDGPU::S_XOR_B64:
1270  splitScalar64BitBinaryOp(Worklist, Inst, AMDGPU::S_XOR_B32);
1271  Inst->eraseFromParent();
1272  continue;
1273 
1274  case AMDGPU::S_NOT_B64:
1275  splitScalar64BitUnaryOp(Worklist, Inst, AMDGPU::S_NOT_B32);
1276  Inst->eraseFromParent();
1277  continue;
1278 
1279  case AMDGPU::S_BCNT1_I32_B64:
1280  splitScalar64BitBCNT(Worklist, Inst);
1281  Inst->eraseFromParent();
1282  continue;
1283 
1284  case AMDGPU::S_BFE_U64:
1285  case AMDGPU::S_BFE_I64:
1286  case AMDGPU::S_BFM_B64:
1287  llvm_unreachable("Moving this op to VALU not implemented");
1288  }
1289 
1290  if (NewOpcode == AMDGPU::INSTRUCTION_LIST_END) {
1291  // We cannot move this instruction to the VALU, so we should try to
1292  // legalize its operands instead.
1293  legalizeOperands(Inst);
1294  continue;
1295  }
1296 
1297  // Use the new VALU Opcode.
1298  const MCInstrDesc &NewDesc = get(NewOpcode);
1299  Inst->setDesc(NewDesc);
1300 
1301  // Remove any references to SCC. Vector instructions can't read from it, and
1302  // We're just about to add the implicit use / defs of VCC, and we don't want
1303  // both.
1304  for (unsigned i = Inst->getNumOperands() - 1; i > 0; --i) {
1305  MachineOperand &Op = Inst->getOperand(i);
1306  if (Op.isReg() && Op.getReg() == AMDGPU::SCC)
1307  Inst->RemoveOperand(i);
1308  }
1309 
1310  if (Opcode == AMDGPU::S_SEXT_I32_I8 || Opcode == AMDGPU::S_SEXT_I32_I16) {
1311  // We are converting these to a BFE, so we need to add the missing
1312  // operands for the size and offset.
1313  unsigned Size = (Opcode == AMDGPU::S_SEXT_I32_I8) ? 8 : 16;
1314  Inst->addOperand(Inst->getOperand(1));
1315  Inst->getOperand(1).ChangeToImmediate(0);
1316  Inst->addOperand(MachineOperand::CreateImm(0));
1317  Inst->addOperand(MachineOperand::CreateImm(0));
1318  Inst->addOperand(MachineOperand::CreateImm(0));
1319  Inst->addOperand(MachineOperand::CreateImm(Size));
1320 
1321  // XXX - Other pointless operands. There are 4, but it seems you only need
1322  // 3 to not hit an assertion later in MCInstLower.
1323  Inst->addOperand(MachineOperand::CreateImm(0));
1324  Inst->addOperand(MachineOperand::CreateImm(0));
1325  } else if (Opcode == AMDGPU::S_BCNT1_I32_B32) {
1326  // The VALU version adds the second operand to the result, so insert an
1327  // extra 0 operand.
1328  Inst->addOperand(MachineOperand::CreateImm(0));
1329  }
1330 
1331  addDescImplicitUseDef(NewDesc, Inst);
1332 
1333  if (Opcode == AMDGPU::S_BFE_I32 || Opcode == AMDGPU::S_BFE_U32) {
1334  const MachineOperand &OffsetWidthOp = Inst->getOperand(2);
1335  // If we need to move this to VGPRs, we need to unpack the second operand
1336  // back into the 2 separate ones for bit offset and width.
1337  assert(OffsetWidthOp.isImm() &&
1338  "Scalar BFE is only implemented for constant width and offset");
1339  uint32_t Imm = OffsetWidthOp.getImm();
1340 
1341  uint32_t Offset = Imm & 0x3f; // Extract bits [5:0].
1342  uint32_t BitWidth = (Imm & 0x7f0000) >> 16; // Extract bits [22:16].
1343 
1344  Inst->RemoveOperand(2); // Remove old immediate.
1345  Inst->addOperand(Inst->getOperand(1));
1346  Inst->getOperand(1).ChangeToImmediate(0);
1347  Inst->addOperand(MachineOperand::CreateImm(0));
1348  Inst->addOperand(MachineOperand::CreateImm(Offset));
1349  Inst->addOperand(MachineOperand::CreateImm(0));
1350  Inst->addOperand(MachineOperand::CreateImm(BitWidth));
1351  Inst->addOperand(MachineOperand::CreateImm(0));
1352  Inst->addOperand(MachineOperand::CreateImm(0));
1353  }
1354 
1355  // Update the destination register class.
1356 
1357  const TargetRegisterClass *NewDstRC = getOpRegClass(*Inst, 0);
1358 
1359  switch (Opcode) {
1360  // For target instructions, getOpRegClass just returns the virtual
1361  // register class associated with the operand, so we need to find an
1362  // equivalent VGPR register class in order to move the instruction to the
1363  // VALU.
1364  case AMDGPU::COPY:
1365  case AMDGPU::PHI:
1366  case AMDGPU::REG_SEQUENCE:
1367  case AMDGPU::INSERT_SUBREG:
1368  if (RI.hasVGPRs(NewDstRC))
1369  continue;
1370  NewDstRC = RI.getEquivalentVGPRClass(NewDstRC);
1371  if (!NewDstRC)
1372  continue;
1373  break;
1374  default:
1375  break;
1376  }
1377 
1378  unsigned DstReg = Inst->getOperand(0).getReg();
1379  unsigned NewDstReg = MRI.createVirtualRegister(NewDstRC);
1380  MRI.replaceRegWith(DstReg, NewDstReg);
1381 
1382  // Legalize the operands
1383  legalizeOperands(Inst);
1384 
1385  for (MachineRegisterInfo::use_iterator I = MRI.use_begin(NewDstReg),
1386  E = MRI.use_end(); I != E; ++I) {
1387  MachineInstr &UseMI = *I->getParent();
1388  if (!canReadVGPR(UseMI, I.getOperandNo())) {
1389  Worklist.push_back(&UseMI);
1390  }
1391  }
1392  }
1393 }
