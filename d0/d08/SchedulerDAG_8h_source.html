38  // true data dependencies, for example:
39  // %1 = add %2, %3
40  // %4 = add %1, %5
41  // %1 depends on %2 and %3
42  // %4 depends on %1 and %5
43  // another example:
44  // %a = load
45  // store %a
46  // Here you would have to delay the store by two (if loads have a
47  // 2 cycle latency)
48  iterator dep_begin() { return depInsts.begin(); }
49  const_iterator dep_begin() const { return depInsts.begin(); }
50  iterator dep_end() { return depInsts.end(); }
51  const_iterator dep_end() const { return depInsts.end(); }
52 
53  // in the example above:
54  // %2 and %3 are used by %1
55  // %1 and %5 are used by %4
56  iterator use_begin() { return useInsts.begin(); }
57  const_iterator use_begin() const { return useInsts.begin(); }
58  iterator use_end() { return useInsts.end(); }
59  const_iterator use_end() const { return useInsts.end(); }

61  // dependencies due to alias analysis
62  // these could be RAW, WAR, or WAR dependencies, for example:
63  // store %1
64  // store %2
65  // would be a WAW dependency if pointers %1 and %2 alias
66  // another example:
67  // %a = load
68  // store %b
69  // could be a WAR dependency if %a and %b alias. In this
70  // case you should delay the store by ONE cycle (not two as in the case
71  // above) because you don't need to wait for the load to finish
72  // For example:
73  // %... = load
74  // store %...
75  // Calling mem_dep_begin() on the store would return the load since
76  // there is a WAR dependency from load -> store. The store 'depends'
77  // on the load instruction.
78  // iterate over all instructions that "depend" on this memory instruction
79  iterator mem_dep_begin() { return bbMemDepInsts.begin(); }
80  const_iterator mem_dep_begin() const { return bbMemDepInsts.begin(); }
81  iterator mem_dep_end() { return bbMemDepInsts.end(); }
82  const_iterator mem_dep_end() const { return bbMemDepInsts.end(); }
83 
84  // For example:
85  // %... = load
86  // store %...
87  // Calling mem_use_begin() on the load would return the store since
88  // there is a WAR dependency from load -> store. The load is 'used' by
89  // the store instruction.
90  // iterate over all instructions that "use" the result of this memory
91  // instruction
92  iterator mem_use_begin() { return bbMemUseInsts.begin(); }
93  const_iterator mem_use_begin() const { return bbMemUseInsts.begin(); }
94  iterator mem_use_end() { return bbMemUseInsts.end(); }
95  const_iterator mem_use_end() const { return bbMemUseInsts.end(); }
