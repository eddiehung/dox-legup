80 // pushes an instruction to the next state, creating an extra if necessary
81 void FiniteStateMachine::pushInstructionToNextState(Instruction *needle) {
82  State *st = map[needle];
83  st->remove(needle);
84  // no need to create an extra state
85  if (!st->isTerminating()) {
86  assert(st->getDefaultTransition() && "no default transition");
87  st->getDefaultTransition()->push_back(needle);
88  return;
89  }
90  // create a new state for the instruction
91  State *s = newState(st);
92  s->setBasicBlock(needle->getParent());
93  s->push_back(needle);
94  s->setTerminating(true);
95  // fix transitioning
96  s->setTransition(st->getTransition());
97  State::Transition blank;
98  st->setTransition(blank);
99  st->setTerminating(false);
100  if (st->getDefaultTransition()) {
101  s->setDefaultTransition(st->getDefaultTransition());
102  }
103  st->setDefaultTransition(s);
104 }

165 
166  // false condition

168  "label=\"~" + getTransitionOp(s) + "\"");
169  } else {

172 
173  for (unsigned i = 0, e = s->getNumTransitions()-1; i != e; ++i) {
174  std::string label = "label=\"" + getTransitionOp(s) + " == " +
175  getLabel(s->getTransitionValue(i)) + "\"";
176  graph.connectDot(out, s, s->getTransitionState(i), label);
177  }
178 
179  graph.connectDot(out, s, s->getDefaultTransition(), "Default");
180 
181  }
