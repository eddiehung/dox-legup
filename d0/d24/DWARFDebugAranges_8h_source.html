39  void setHighPC(uint64_t HighPC) {
40  if (HighPC == -1ULL || HighPC <= LowPC)
41  Length = 0;
42  else
43  Length = HighPC - LowPC;
44  }
45  uint64_t HighPC() const {
46  if (Length)
47  return LowPC + Length;
48  return -1ULL;
49  }
