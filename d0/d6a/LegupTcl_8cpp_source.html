406 /// loop_pipeline - tcl command to pipeline a loop:
407 /// loop_pipeline "loop_label" [ -ii <num> ] [-ignore-mem-deps]
408 /// optional arguments:
409 /// -ii <num>
410 /// force a specific pipeline initiation interval
411 /// -ignore-mem-deps
412 /// ignore loop carried dependencies for all memory accesses in the loop
413 /// loop_pipeline "loop" -ii 1
414 /// Examples:
415 /// loop_pipeline "loop1"
416 /// loop_pipeline "loop2" -ii 1
417 /// loop_pipeline "loop3" -ii 4 -ignore-mem-deps
418 /// loop_pipeline "loop4" -ignore-mem-deps
419 int loop_pipeline(ClientData clientData, Tcl_Interp *interp, int argc,
420  const char *argv[]) {
421  if (argc < 2) {
422  static char error[] =
423  "wrong # args: should be "
424  "\"loop_pipeline \"loop\" [ -ii 1 ] [-ignore-mem-deps]\"";
425  interp->result = error;
426  errs() << "Error: " << error << "\n";
427  assert(0);
428  return (TCL_ERROR);
429  }
430 
431  // look for optional arguments
432  bool user_II = false;
433  int II = 0;
434  bool ignoreMemDeps = false;
435  for (int i = 3; i <= argc; i++) {
436 
437  if (strcmp(argv[i - 1], "-ii") == 0) {
438  i++;
439  assert(i <= argc && "Wrong # args for loop_pipeline -ii option");
440  user_II = true;
441  II = atoi(argv[i - 1]);
442  assert(II >= 0 && "Initiation interval must be non-negative");
443 
444  } else if (strcmp(argv[i - 1], "-ignore-mem-deps") == 0) {
445  ignoreMemDeps = true;
446 
447  } else {
448  errs() << "arg value: " << argv[i - 1] << "\n";
449  assert(0 && "Unrecognized optional argument to loop_pipeline");
450  }
451  }
452 
453  const char *label = argv[1];
454 
455  LegupConfig *config = (LegupConfig *)clientData;
456  config->addLoopPipeline(label, user_II, II, ignoreMemDeps);
457  return (TCL_OK);
458 }
