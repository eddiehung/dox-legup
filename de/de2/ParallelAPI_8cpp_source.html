56 bool ParallelAPI::getAndReplaceCalls(Module &M) {
57 
58  bool modified = false;
59  // region between lock and unlock
60  bool atomicRegion = false;
61  // region between barrier and end of basic block
62  bool barrierRegion = false;
63 
64  // look for call instructions and
65  // make necessary replacements
66  for (Module::iterator F = M.begin(), FE = M.end(); F != FE; ++F) {
67  for (Function::iterator BB = F->begin(), EE = F->end(); BB != EE;
68  ++BB) {
69  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;) {
70 
71  Instruction *Inst = I++;
72  // If it's a call instruction
73  if (CallInst *CI = dyn_cast<CallInst>(Inst)) {
74 
75  Function *calledFunction = CI->getCalledFunction();
76 
77  // ignore indirect function calls
78  if (!calledFunction)
79  continue;
80 
81  // get and replace call instructions
82  if (getAndReplaceCallInsts(M, CI, modified, atomicRegion,
83  barrierRegion))
84  continue;
85 
86  // if it's accelerated function, don't inline it
87  if (LEGUP_CONFIG->isAccelerated(calledFunction->getName()))
88  calledFunction->addFnAttr(
89  Attribute::NoInline); // adding noinline attribute
90  // to function
91 
92  // if it's a load
93  } else if (LoadInst *LI = dyn_cast<LoadInst>(Inst)) {
94 
95  setLoadtoVolatile(LI, atomicRegion || barrierRegion,
96  modified);
97 
98  } else {
99

102  }

105  }

173 Function *ParallelAPI::generateOMPWrapper(Function *F, int numThreads) {
174 
175  // create the wrapper function (which will call the Pthread routine)
176  std::string wrapperName = getWrapperName(F, ompcall);
177 
178  std::vector<Type *> argTypes;
179  // only one argument exists for OMP functions
180  argTypes.push_back(*(F->getFunctionType()->param_begin()));
181 
182  // create wrapper function
183  Constant *c = F->getParent()->getOrInsertFunction(
184  wrapperName, FunctionType::get(Type::getVoidTy(getGlobalContext()),
185  argTypes, false));
186  Function *wrapperF = cast<Function>(c);
187  wrapperF->setCallingConv(CallingConv::C);
188 
189  //copy the argument names from the newly created omp function to the wrapper function
190  for (Function::arg_iterator I = F->arg_begin(), E = --F->arg_end(),
191  I2 = wrapperF->arg_begin(); I != E; ++I, ++I2) {
192  I2->setName(I->getName());
193  }
194 
195  wrapperF->addFnAttr(
196  Attribute::NoInline); // adding noinline attribute to function
197 
198  std::vector<Value *> Args;
199  // only one argument exists for OMP functions
200  Args.push_back(wrapperF->arg_begin());
201 
202  // now start to create instructions to call each accelerator:
203  //
204  // create the basic BB
205  BasicBlock *entry =
206  BasicBlock::Create(getGlobalContext(), "entry", wrapperF);
207  IRBuilder<> builder(entry);
208 
209  // create a call instruction for each instance of the accelerator
210  // each instance is assigned to a unique threadID
211  for (int i = 0; i < numThreads; i++) {
212 
213  // push the threadID as an extra argument, which is just the index
214  // number for the accelerator
215  Args.push_back(ConstantInt::get(
216  IntegerType::get(getGlobalContext(), 32), i, false));
217 
218  // create the call instruction to the accelerated function
219  CallInst *call = builder.CreateCall(F, Args, "");
220 
221  // set metadata used for later stages in LegUp
222  setMetadataStr(call, "TYPE", "omp_function");
223  setMetadataStr(call, "NUMTHREADS", utostr(numThreads));
224  setMetadataInt(call, "THREADID", i);
225 
226  Args.pop_back();
227  }
228 
229  // create a return void instruction
230  builder.CreateRetVoid();
231 
232  // return wrapper function pointer
233  return wrapperF;
234 }

240 bool ParallelAPI::replacePthreadCreateCalls(
241  CallInst *CI,
242  std::vector<std::pair<Function *, unsigned>> &pthreadFunctionThreadID) {
243 
244  // get the pthread function
245  Module &M = *CI->getParent()->getParent()->getParent();
246  std::string funcName =
247  (CI->getArgOperand(2))->stripPointerCasts()->getName().str();
248  Function *F = findFuncPtr(M, funcName.c_str());
249 
250  // get the argument into pthread function
251  CallInst *Call = CI;
252  Value *argOp = Call->getArgOperand(3);
253 
254  std::vector<Value *> Args;
255  // check if the argument is null
256  if (!dyn_cast<ConstantPointerNull>(argOp)) {
257  Args.push_back(argOp);
258  }
259 
260  // get the number of threads by analyzing loop trip count
261  LoopInfo *LI;
262  LI = &getAnalysis<LoopInfo>(*(CI->getParent()->getParent()));
263  unsigned numThreads = getLoopTripCount(CI->getParent(), LI);
264 
265  // get the threadID and functionID
266  // for this pthread function
267  unsigned threadID = 0, functionID = 0;
268  bool found = false;
269  for (vector<pair<Function *, unsigned>>::iterator it =
270  pthreadFunctionThreadID.begin();
271  it != pthreadFunctionThreadID.end(); ++it) {
272  if (it->first == F) {
273  threadID = it->second;
274  it->second = threadID + 1;
275  found = true;
276  break;
277  }
278  functionID++;
279  }
280 
281  if (!found) {
282  pthreadFunctionThreadID.push_back(make_pair(F, threadID + 1));
283  }
284 
285  // Before replacing the call instruction, we need to insert instructions for
286  // getting the threadID.
287  // This has to be done outside the wrapper function since to prevent
288  // a stale threadID value from being loaded
289  IRBuilder<> builder(Call);
290 
291  generatePthreadThreadIDInstructions(builder, F, Call, functionID);
292 
293  // now replace call
294  CallInst *New = replaceCallInstFromOldtoNewFunction(Call, F, Args, false);
295 
296  // add noinline attribute to function
297  New->setIsNoInline();
298 
299  // set metadata for the new call instruction
300  setMetadataStr(New, "PTHREADNAME", F->getName());
301  setMetadataInt(New, "NUMTHREADS", numThreads);
302  setMetadataInt(New, "THREADID", threadID);
303  setMetadataInt(New, "FUNCTIONID", functionID);
304  setMetadataStr(New, "TYPE", "legup_wrapper_pthreadcall");
305 
306  // check if we need to preserve the pthread value
307  // this was set before when replacing
308  // pthread_exit with a ret instruction
309  if (!F->hasFnAttribute(Attribute::NoReturn)) {
310  preserveLLVMInstruction(New, New->getNextNode());
311  }
312 
313  // increment threadID for this pthread function
314  // pthreadFunctionThreadID[F] = ++ID;
315 
316  // this is used to get the maximum number of parallel threads
317  // that will be forked for a function
318  // at the same time
319  unsigned numThreadsMax = numThreads;
320 
321  // threadID may be bigger, if pthread_create is not used
322  // in a loop, in which case numThreads == 1
323  // then assign threadID to numThreadsMax
324  // + 1 is there since threadID starts from 0
325  if (threadID + 1 > numThreadsMax)
326  numThreadsMax = threadID + 1;
327 
328  // set the total number of threads
329  // for this function
330  F->addFnAttr("totalNumThreads", utostr(numThreadsMax));
331 
332  // remove noreturn attribute now
333  // (llvm removes all proceeding instructions otherwise)
334  F->removeFnAttr(Attribute::NoReturn);
335 
336  return true;
337 }

342 void
343 ParallelAPI::generatePthreadThreadIDInstructions(IRBuilder<> &builder,
344  Function *F, CallInst *Call,
345  const unsigned functionID) {
346 
347  // get the thread variable pointer
348  // instanceID value will be stored here
349  Value *threadVarptr = Call->getArgOperand(0);
350 
351  assert(threadVarptr);
352 
353  // get the threadID global variable
354  // used for this function
355  Value *V0 = getPthreadIDGlobalVariable(F);
356 
357  // load from globalvar
358  // this is the threadID value for this instance of the pthread function
359  Instruction *threadIDValue = builder.CreateLoad(
360  V0, false, "legup_count_" + F->getName().str() + "_loadedValue");
361  assert(threadIDValue);
362 
363  // increment the loaded value for next thread
364  Value *add = builder.CreateAdd(
365  threadIDValue,
366  ConstantInt::get(IntegerType::get(getGlobalContext(), 32), 1, false));
367  // store back into global variable
368  builder.CreateStore(add, V0, false);
369 
370  // create the pthead function ID in LLVM IR
371  Value *functionIDVal = ConstantInt::get(
372  IntegerType::get(getGlobalContext(), 32), functionID, "functionID");
373 
374  // now using the threadID and functionID
375  // we create instanceID (= functionID << 16 | threadID)
376  // this value is used the polling wrapper to determine
377  // which function and which instance of that function to poll and retrieve
378  // the return value
379  //
380  // shift functionID << 16 and OR threadID
381  Value *shiftedFunctionID = builder.CreateShl(functionIDVal, 16);
382  Value *functionThreadID =
383  builder.CreateOr(shiftedFunctionID, threadIDValue);
384 
385  // store the instance ID into threadVarPtr
386  // make it volatile to make sure it's not optimized away
387  Instruction *st = builder.CreateStore(functionThreadID, threadVarptr, true);
388  setMetadataInt(st, "legup_pthread_functionthreadID", 1);
389  setMetadataInt(st, "legup_pthread_wrapper_inst", 1);
390 }
