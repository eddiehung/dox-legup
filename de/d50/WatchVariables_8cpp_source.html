47  void addVariableToFormatString(Instruction * instr,
48  std::string &printString) {

87  static bool isAPrintCall(Instruction *instr) {
88  CallInst *ci = dyn_cast<CallInst>(instr);
89  if (!ci) {

102  void watchInstruction(Instruction *instr, std::vector<Value *> &Args,
103  std::string &printString) {
104  // store, and gep instructions will never be assigned an integer value
105  if (isa<StoreInst>(instr) || isa<GetElementPtrInst>(instr)) {
106  return;
107  }
108  if ((isa<IntegerType>(instr->getType()) || instr->getType()->isFloatingPointTy())
109  && !isAPrintCall(instr)) {
110  addVariableToFormatString(instr, printString);
111  Args.push_back(instr);
112  // replace UndefValue's with ConstantInt's equal to 0
113  for (unsigned i = 0; i < instr->getNumOperands(); i++) {
114  if (isa<UndefValue>(instr->getOperand(i))) {
115  Type *type = instr->getOperand(i)->getType();
116  Value *value = NULL;
117  if (isa<IntegerType>(type)) {
118  value = ConstantInt::get(type, 0);
119  } else if (type->isFloatingPointTy()) {
120  value = ConstantFP::get(type, 0);
121  }
122  assert(value && "Unsupported type\n");
123  instr->setOperand(i, value);
124  }
125  }
126  }
127  }

129  void reachedFunctionTerminator(std::string &printString,
130  std::vector<Value *> &Args,
131  std::queue< std::vector<Value *> > &ArgsQueue,
132  Function::iterator &BB, Function &F, Instruction *instr) {
133  printString += '\n';
134  IRBuilder<> Builder(BB);
135  // printString is the first argument of printf(char *, ...)
136  Args[0] = Builder.CreateGlobalStringPtr(printString.c_str());
137  ArgsQueue.push(Args);
138  // print all variables that have been assigned in that basic block
139  while (!ArgsQueue.empty()) {
140  CallInst::Create(PrintFunc, ArgsQueue.front(), "", instr);
141  ArgsQueue.pop();
142  }
143  }

145  virtual bool runOnFunction(Function &F) {
146  for (Function::iterator BB = F.begin(), EE = F.end(); BB != EE; ++BB) {
147  // initialize string to print out
148  std::string printString = "\nWatch@" + F.getName().str() + ":";
149  printString += BB->getName().str() + '\n';
150  std::vector<Value *> Args;
151  // reserve first argument of printf for printString
152  Args.push_back(NULL);
153  std::queue< std::vector<Value *> > ArgsQueue;
154  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;
155  ++I) {
156  Instruction *instr = I;
157  // print variables values just before the basic block terminates
158  if (instr->isTerminator()) {
159  reachedFunctionTerminator(printString, Args, ArgsQueue, BB,
160  F, instr);
161  } else {
162  watchInstruction(instr, Args, printString);
163  }
164  }
165  }
166  // the bitcode will change as long as a basic block exists
167  return true;
168  }
