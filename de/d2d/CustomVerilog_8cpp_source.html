152 void removeFunctionsFromVectorIfNotCalled(std::vector<Function *> &functions) {
153 
154  unsigned int previousFunctionCount = functions.size();
155  std::vector<FunctionWithCallCount> calledFunctions;
156 
157  for (std::vector<Function *>::iterator it = functions.begin(); it != functions.end(); ++it) {
158  for (Function::iterator b = (*it)->begin(), be = (*it)->end(); b != be; ++b) {
159  for (BasicBlock::iterator instr = b->begin(), ie = b->end(); instr !=
160  ie; ++instr) {
161  if (isaDummyCall(instr)) continue;
162 
163  if (CallInst *CI = dyn_cast<CallInst>(instr)) {
164  Function *called = getCalledFunction(CI);
165 
166  // Get the existing FunctionWithCallCount for the called function
167  // if one exists
168  //
169  std::vector<FunctionWithCallCount>::iterator existing =
170  functionWithCallCountForFunctionInVector(called, calledFunctions);
171 
172  // If it does exist, increment its callCount
173  //
174  if (existing != calledFunctions.end())
175  existing->callCount++;
176  else {
177 
178  // It does not exist, so we make a new one and add it to our
179  // vector
180  //
181  FunctionWithCallCount newFunctionWithCallCount;
182  newFunctionWithCallCount.func = called;
183  newFunctionWithCallCount.callCount = 1;
184  calledFunctions.push_back(newFunctionWithCallCount);
185  }
186  }
187  }
188  }
189  }
190 
191 
192  // Keep going through functions and removing ones that aren't called until
193  // the size of our list of functions stops changing
194  //
195  do {
196 
197  previousFunctionCount = functions.size();
198 
199  for (std::vector<Function*>::iterator it = functions.begin(); it != functions.end();) {
200 
201  // We removing 'main' would cause all functions to be removed, but,
202  // there aren't any functions that call 'main', so we would remove
203  // it if we didn't have this condition
204  //
205  if ((*it)->getName() != "main") {
206 
207  // Get the iterator pointing to the location of the current function
208  // in our list of called functions
209  //
210  std::vector<FunctionWithCallCount>::iterator fcc =
211  functionWithCallCountForFunctionInVector(*it, calledFunctions);
212 
213  // If the function was not in our list of called functions then we
214  // can remove it
215  //
216  if (fcc == calledFunctions.end()) {
217 
218  // Save function pointer from iterator and increment iterator so
219  // that we don't destroy our iterator when we (potentially) erase
220  // functions from our vector.
221  //
222  std::vector<Function*>::iterator savedIterator = it;
223  Function *function = *it;
224  it++;
225 
226  // Before we remove our function we must decrement the call counts
227  // of any functions it calls (this function is recursive and will
228  // decrement the call counts of all of the child functions)
229  //
230  removeUncalledFunctionsFromVector(function, calledFunctions);
231  it = std::find(functions.begin(),functions.end(),function);
232  if (it == functions.end())
233  it = functions.begin();
234 
235  // Remove the current function from the vector
236  //
237  functions.erase(savedIterator);
238 
239  }
240  else {
241

243  //
244  it++;
245 
246  }
247  }
248  else {
249 
250  // We must still increment the iterator
251  //
252  it++;
253 
254  }

327 bool CustomVerilog::runOnModule(Module &M) {
328 
329  std::vector<Function*> HwFcts;
330 
331  // We don't want to synthesize functions called by custom verilog
332  // functions, but we do want to synthesize the custom verilog
333  // functions so that they can be added to the rtl.
334  //
335  // To accomplish this, we will get all of the non-custom verilog
336  // functions and remove all of the functions that aren't called
337  // by any of the functions in the vector. Then, we will add in
338  // all of the custom verilog functions.
339  //
340  getNonCustomVerilogFunctions(M, HwFcts);
341  removeFunctionsFromVectorIfNotCalled(HwFcts);
342  getCustomVerilogFunctions(M, HwFcts);
343  warnIfCustomVerilogNotCalled(M);
344 
345  std::set<GlobalValue *> HwFctsSet = getGlobalValueSetForVector(HwFcts);
346 
347  return isolateGV(M, HwFctsSet);
348
