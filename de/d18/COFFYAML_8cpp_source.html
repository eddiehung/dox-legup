290 void MappingTraits<COFFYAML::Relocation>::mapping(IO &IO,
291  COFFYAML::Relocation &Rel) {
292  IO.mapRequired("VirtualAddress", Rel.VirtualAddress);
293  IO.mapRequired("SymbolName", Rel.SymbolName);
294 
295  COFF::header &H = *static_cast<COFF::header *>(IO.getContext());
296  if (H.Machine == COFF::IMAGE_FILE_MACHINE_I386) {
297  MappingNormalization<NType<COFF::RelocationTypeI386>, uint16_t> NT(
298  IO, Rel.Type);
299  IO.mapRequired("Type", NT->Type);
300  } else if (H.Machine == COFF::IMAGE_FILE_MACHINE_AMD64) {
301  MappingNormalization<NType<COFF::RelocationTypeAMD64>, uint16_t> NT(
302  IO, Rel.Type);
303  IO.mapRequired("Type", NT->Type);
304  } else {
305  IO.mapRequired("Type", Rel.Type);
306  }
307 }
