118 bool LegupConfig::setLegupOutputPath(std::string path) {
119  // Check that the path is valid, meaning that it is a subdirectory
120  // Don't allow for blank paths
121  if (path.length() == 0) {
122  errs() << "Setting LegUp Output Path failed: input path had 0 length";
123  return false;
124  }
125  // Don't allow the first character to be '/'
126  if (path.find('/') == 0) {
127  errs() << "Setting LegUp Output Path failed: input path started with /";
128  return false;
129  }
130  // Don't allow the any character to be '~'
131  if (path.find('~') != std::string::npos) {
132  errs() << "Setting LegUp Output Path failed: input path had '~'";
133  return false;
134  }
135  // Don't allow any part of the string to be '..'
136  if (path.find("..") != std::string::npos) {
137  errs() << "Setting LegUp Output Path failed: input path had \"..\"";
138  return false;
139  }
140  // Don't allow '.'
141  if ((path.find('.') == 0) && path.length() == 1) {
142  errs() << "Setting LegUp Output Path failed: input path was '.'";
143  return false;
144  }
145 
146  // Should be a subdirectory
147  legupOutputPath = path;
148  return true;

187 bool LegupConfig::isBinaryOperatorNoOp(Instruction *instr) {
188  switch (instr->getOpcode()) {
189  case Instruction::Add:
190  // x + 0 = 0 + x = x
191  if (isAnyOfTwoOperandsZero(instr))
192  return true;
193  break;
194  case Instruction::Sub:
195  // x - 0 = x
196  if (isSecondOperandZero(instr))
197  return true;
198  break;
199  case Instruction::Mul:
200  case Instruction::URem:
201  case Instruction::UDiv:
202  if (isSecondOperandPowerOfTwo(instr))
203  return true;
204  break;
205  case Instruction::And:
206  case Instruction::Or:
207  case Instruction::Shl:
208  case Instruction::AShr:
209  case Instruction::LShr:
210  if (isSecondOperandConstant(instr)) {
211  // If the second operand is a constant
212  return true;
213  }
214  break;
215  }
216 
217  return false;
218 }

494 int findMatchingConstraint(std::string FuName,
495  std::map<std::string, int> &constraintMap,
496  int *constraint) {
497  // find longest string match from constraints:
498  int finalConstraint = 0;
499  std::string longestMatch;
500  bool foundMatch = false;
501  for (std::map<std::string, int>::iterator i = constraintMap.begin(),
502  e = constraintMap.end();
503  i != e; ++i) {
504  std::string constraintFuName = i->first;
505  int constraint = i->second;
506 
507  size_t found;
508  found = FuName.find(constraintFuName);
509  if (found == std::string::npos)
510  continue;
511 
512  // match found
513  unsigned lengthMatch = constraintFuName.size();
514  if (lengthMatch > longestMatch.size()) {
515  finalConstraint = constraint;
516  longestMatch = constraintFuName;
517  foundMatch = true;
518  }
519  }
520  // errs() << FuName << " (closest tcl constraint: " << longestMatch
521  // << ")\n";
522 
523  *constraint = finalConstraint;
524 
525  return foundMatch;
526 }

528 // This function calculates the constraint on an FuName by
529 // trying to find the most specific constraint (longest string match) that
530 // applies. For instance in the tcl file:
531 // set_resource_constraint signed_divide_16 3
532 // set_resource_constraint signed_divide 2
533 // set_resource_constraint divide 1
534 // this is what will be returned:
535 // getNumberOfFUsAllocated(signed_divide_8) = 2
536 // getNumberOfFUsAllocated(signed_divide_16) = 3
537 // getNumberOfFUsAllocated(signed_divide_32) = 2
538 // getNumberOfFUsAllocated(signed_divide_64) = 2
539 // getNumberOfFUsAllocated(unsigned_divide_8) = 1
540 // getNumberOfFUsAllocated(unsigned_divide_16) = 1
541 // getNumberOfFUsAllocated(unsigned_divide_32) = 1
542 // getNumberOfFUsAllocated(unsigned_divide_64) = 1
543 // this also applies for memories. For instance, for the local memory 'shared'
544 // the FuName would be:
545 // shared_mem_dual_port
546 // So the following tcl constraint would apply (if there was nothing more
547 // specific):
548 // set_resource_constraint mem_dual_port 2
549 // returns false if no constraint is found
550 bool LegupConfig::getNumberOfFUsAllocated(std::string FuName, int *number) {
551  // errs() << "getNumberOfFUsAllocated(" << FuName << ")\n"
552 
553  return findMatchingConstraint(FuName, FuncUnitConstraints, number);

580 std::string LegupConfig::getOpNameFromInst(Instruction *instr,
581  Allocation *alloc) {
582  assert(alloc);
583 
584  // Later take care of signed/unsigned
585  std::string params[10];
586  int width0 = 0, width1 = 0, width2 = 0;
587 
588  if (isMem(instr)) {
589  std::string mem = "mem_dual_port";
590  if (LEGUP_CONFIG->getParameterInt("LOCAL_RAMS")) {
591  assert(alloc);
592  RAM *ram = alloc->getLocalRamFromInst(instr);
593  if (ram && !alloc->isRAMGlobal(ram)) {
594  // each local ram is a unique FU
595  mem = ram->getName() + "_local_" + mem;
596  } else {
597  // global memory -- all have the same FU name
598  }
599  }
600  return mem;
601  }
602 
603  switch (instr->getNumOperands()) {
604  case 1:
605  if (isa<SIToFPInst>(instr)) {
606  // altfp_sitofp32 converts int (32) to float (32)
607  // altfp_sitofp64 converts int (32) to double (64)
608  // width0 should be width of output
