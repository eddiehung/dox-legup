22  std::map<Instruction*, Node*>::iterator GNi = g->GraphNodes.begin();
23  std::map<Instruction*, Node*>::iterator GNe = g->GraphNodes.end();
24  for ( ; GNi != GNe; ++GNi) {
25  GNi->second->label = label;
26  label++;
27  }

30 // Return true if the two nodes are of the same type.
31 // Node types are the same if either:
32 // - both nodes are operations of the same type,
33 // - both nodes are the INPUT node,
34 // - both nodes are constants with the same value
35 // Also return a mismatch if either node is NULL.
36 bool node_types_mismatch(Node *n1, Node *n2) {
37  // Return mismatch if either node is NULL
38  if (n1 == NULL || n2 == NULL) return true;
39 
40  // If both nodes are operations or both are not operations,
41  // then there is a mismatch if their type strings are unequal
42  // - for operations these strings are e.g. "and32", and
43  // - for non-operations (INPUT or CONSTANT) the strings are
44  // either "INPUT" or e.g. "POS3", "NEG4", etc.
45  if ( (n1->is_op && n2->is_op) ||
46  (!n1->is_op && !n2->is_op) )
47  {
48  return (n1->op != n2->op); // return mismatch if unequal
49  }
50 
51  // One instruction is an operation and the other is not,
52  // so there is a mismatch
53  return true;
54 }

60 // Used to copy the node labelling from one graph to another
61 void PatternMap::CopyLabels (Graph * Old, Graph * New) {
62  assert (Old->size() == New->size());
63  assert (Old->size() != 0);
64 
65  New->getRoot()->label = Old->getRoot()->label;
66  std::set <Node*> visited; // nodes visited in the Old graph
67  visited.insert(Old->getRoot());
68 
69  queue <Node*> old_nodes;
70  old_nodes.push(Old->getRoot()); // queue of nodes to visit in Old graph
71  queue <Node*> new_nodes;
72  new_nodes.push(New->getRoot()); // queue of nodes to visit in New graph
73 
74  // use these temp pointers as opposed to the graph's "current"
75  // pointer in order to not change the value of the current pointer,
76  // which may be in use by another function
77  Node * t1;
78  Node * t2;
79 
80  while (!old_nodes.empty()) {
81  t1 = old_nodes.front(); old_nodes.pop();
82  t2 = new_nodes.front(); new_nodes.pop();
83 
84  // If the predecessors are operations and haven't
85  // been visited yet, copy their label
86  if (t1->p1->is_op) {
87  if (visited.find(t1->p1) == visited.end()) {
88  t2->p1->label = t1->p1->label;
89  visited.insert(t1->p1);
90  old_nodes.push(t1->p1);
91  new_nodes.push(t2->p1);
92  }
93  }
94  if (t1->p2->is_op) {
95  if (visited.find(t1->p2) == visited.end()) {
96  t2->p2->label = t1->p2->label;
97  visited.insert(t1->p2);
98  old_nodes.push(t1->p2);
99  new_nodes.push(t2->p2);
100  }
101  }
102  }
103 }

106 // Adds all equivalent graphs to the given GraphList (recursive)
107 // This function recursively adds all equivalent copies of the base
108 // Graph to the list
109 void PatternMap::AddEquivalentCopies
110 (
111  Graph * base,
112  Instruction * start,
113  GraphList * list
114 )
115 {
116  Graph::GraphNodes_iterator i;
117  bool begin = false;
118  for (i = base->GraphNodes.begin(); i != base->GraphNodes.end(); ++i) {
119  // only start swapping commutative operations from the
120  // one sent as a parameter
121  if ( (*i).first == start ) {
122  begin = true;
123  }
124 
125  // Note this is not equivalent to if ((*i).first != start) continue,
126  // since begin never resets
127  if (!begin) continue;
128 
129  // Note: LLVM does not consider icmpeq or icmpne to be commutative
130  if ( ((*i).first)->isCommutative() ) {
131  // create a new graph which will be identical to "this"
132  // graph except the operator corresponding to (*i)->first
133  // will have its predecessors swapped
134  Graph * g = new Graph ();
135  (*g) = (*base);
136  Label(g); // label all the nodes arbitrarily
137 
138  // commutative, so swap the predecessors
139  Node * temp = g->GraphNodes[(*i).first]->p1;
140  g->GraphNodes[(*i).first]->p1 = g->GraphNodes[(*i).first]->p2;
141  g->GraphNodes[(*i).first]->p2 = temp;
142 
143  // add new graph to the list
144  list->push_back(g);
145 
146  // and add all its equivalent versions as well, starting
147  // from the next instruction
148  Graph::GraphNodes_iterator temp_iterator = i;
149  ++temp_iterator;
150  if ( temp_iterator != base->GraphNodes.end() ) {
151  AddEquivalentCopies(g, (*temp_iterator).first, list);
152  }
153  }
154  }

168 bool PatternMap::checkEqualityThroughBFS(Graph * g1, Graph * g2) {
169 
170  // BFS data structures
171  set <Node*> g1_visited; // visited nodes in each graph
172  set <Node*> g2_visited;
173  queue <Node*> g1_nodes; // queues used during traversal
174  queue <Node*> g2_nodes;
175 
176  // Initialize BFS
177  g1_nodes.push(g1->getRoot()); // root of g1
178  g2_nodes.push(g2->getRoot()); // root of g2
179  g1_visited.insert(g1->getRoot());
180  g2_visited.insert(g2->getRoot());
181 
182  // use these temp pointers as opposed to the graph's "current"
183  // pointer in order to not change the value of the current pointer,
184  // which may be in use by another function
185  Node * t1;
186  Node * t2;
187 
188  // Main BFS loop
189  while (!g1_nodes.empty()) {
190  // Pop next nodes from queue
191  t1 = g1_nodes.front(); g1_nodes.pop();
192  t2 = g2_nodes.front(); g2_nodes.pop();
193 
194  // Cache the predecessors of t1 and t2
195  Node *t1_predecessors [2], *t2_predecessors [2];
196  t1_predecessors[0] = t1->p1; // left predecessors
197  t2_predecessors[0] = t2->p1;
198  t1_predecessors[1] = t1->p2; // right predecessors
199  t2_predecessors[1] = t2->p2;
200 
201  // Check equality of both the left and right predecessors
202  for (int predecessor_number = 0; predecessor_number < 2; ++predecessor_number) {
203  Node *g1_pred = t1_predecessors[predecessor_number];
204  Node *g2_pred = t2_predecessors[predecessor_number];
205 
206  // Check for mismatching types of the predecessors
207  if (node_types_mismatch(g1_pred, g2_pred)) {
208  return false; // The two predecessors have mismatching types
209  }
210 
211  // The two predecessors have matching types. If they are operations
212  // (rather than constants / inputs), consider adding them to the queue,
213  // as long as neither node has been visited before
214  if (g1_pred->is_op) {
215 
216  // This node has not yet been visited in g1
217  if ( g1_visited.find(g1_pred) == g1_visited.end() ) {
218 
219  // If it has been visited in g2, there is a mismatch
220  if ( g2_visited.find(g2_pred) != g2_visited.end() ) return false;
221 
222  // At this point, the current predecessors in both graphs are:
223  // - Both instructions
224  // - The same instruction
225  // - Both have not been visited
226  // However, lastly we need to consider registers. E.g. if t1 and g1_pred are
227  // in the same state, but t2 and g2_pred are not, then there is a register
228  // in the way and we can't share, so these graphs are NOT equal
229 
230  // The condition below checks the following:
231  // If (both t1 and t2 are in the same states as their predecessors) OR
232  // (neither t1 and t2 are in the same states as their predecessors)
233  if ( ( fsm->getEndState(t1->I) == fsm->getEndState(g1_pred->I) ) !=
234  ( fsm->getEndState(t2->I) == fsm->getEndState(g2_pred->I) ) )
235  {
236  return false;
237  }
238 
239  g1_nodes.push(g1_pred);
240  g2_nodes.push(g2_pred);
241  g1_visited.insert(g1_pred);
242  g2_visited.insert(g2_pred);
243  }
244  // The node has been visited in g1, so if it hasn't been
245  // visited in g2, return mismatch
246  else if ( g2_visited.find(g2_pred) == g2_visited.end() ) {
247  return false;
248  }
249  }
250  }
251  }
252  return true;
253 }

304 // Add a new graph to our PatternMap
305 void PatternMap::Add( Graph*g ) {
306  PatternList_iterator i;
307  Graph_iterator j;
308 
309  // if the map is empty or does not contain this size, add it
310  if ( Patterns.find(g->size()) == Patterns.end() )
311  {
312  Label(g); // label all the nodes arbitrarily
313  Patterns[g->size()].push_back( GraphList() );
314  EquivalentCopies[g->size()].push_back( GraphList() );
315  Patterns[g->size()].back().push_back(g);
316 
317  Graph * g_copy = new Graph;
318  *(g_copy) = *g;
319  Label(g_copy); // label all the nodes arbitrarily
320  EquivalentCopies[g->size()].back().push_back(g_copy);
321  // and also add all equivalent copies
322  AddEquivalentCopies(
323  g,
324  g->GraphNodes.begin()->first,
325  &( EquivalentCopies[g->size()].back() )
326  );
327  return;
328  }
329 
330  // otherwise the size exists. Check for the graph in every GraphList of
331  // every Pattern list for this size, and if it is found then the pattern
332  // exists so increment its frequency.
333 
334  // Note this is where the EquivalentCopies GraphList is used: we need to
335  // determine if the pattern of Graph g has been found before, but it could
336  // be that it was found but in a different (equivalent) form
337 
338  // for each GraphList in the vector
339  for ( PatternList_iterator i1 = Patterns[g->size()].begin(),
340  i2 = EquivalentCopies[g->size()].begin();
341  i1 != Patterns[g->size()].end();
342  ++i1, ++i2
343  )
344  {
345  // for each graph in the GraphList
346  for (Graph_iterator j = i2->begin(); j != i2->end(); ++j) {
347  if ( checkEquality(g, *j) ) {
348  // We have a match, i.e. the 2 graphs are functionally the same,
349  // but not necessarily topologically. To ensure
350  CopyLabels (*j, g);
351  i1->push_back(g);
352  // The labels are consistent, i.e. label "1" always corresponds
353  // to the same add
354  // Copy the labels now then add the graph
355  return;
356  }
357  }
358  }
359 
360  // the pattern has not been found, so add it as above
361 
362  Label(g); // label all the nodes arbitrarily
363  Patterns[g->size()].push_back( GraphList() );
364  EquivalentCopies[g->size()].push_back( GraphList() );
365  Patterns[g->size()].back().push_back(g);
366 
367  Graph * g_copy = new Graph;
368  *(g_copy) = *g;
369  Label(g_copy); // label all the nodes arbitrarily
370  EquivalentCopies[g->size()].back().push_back(g_copy);
371  // and also add all equivalent copies
372  AddEquivalentCopies(
373  g,
374  g->GraphNodes.begin()->first,
375  &( EquivalentCopies[g->size()].back() )
376  );
377 }
