167 
168  bool modified = false;
169 
170  // replace all calls to accelerated functions with calls to the wrapper
171  for (Module::iterator F = M.begin(), FE = M.end(); F != FE; ++F) {
172  for (Function::iterator BB = F->begin(), EE = F->end(); BB != EE;
173  ++BB) {
174  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E;) {
175  if (CallInst *CI = dyn_cast<CallInst>(I++)) {
176 
177  Function *calledFunction = CI->getCalledFunction();
178  // ignore indirect function calls
179  if (!calledFunction)
180  continue;
181 
182  // add performance counters if there are any, continue if
183  // replaced
184  if (addPerformanceCounters(CI, calledFunction))
185  continue;
186 
187  // for parallel functions
188  if (isaCalltoParallelFunction(CI)) {
189 
190  if (getAndReplaceParallelFunctionCalls(M, CI, I,
191  modified))
192  continue;

194  // if it is a sequential function (a function cannot be
195  // both parallel and sequential)
196  } else {
197 
198  // replace synchronization functions (locks, barriers)
199  if (getSynchronizationFunctions(F, CI, calledFunction))
200  continue;
201 
202  // else replace sequential functions if there are any
203  modified |=

205  }
206  }
207  }
208  }
209  }
210

587 // create a new call to pthread function, with additional arguments
588 // inserted for LegUp
589 CallInst *SwOnly::createNewCalltoPthreadFunction(Module &M, Function *F,
590  CallInst *CI) {
591 
592  // here we need to make a call to the pthread create wrapper
593  // which will have two an additional argument than the pthread
594  // function
595  // 1. the pointer to the thread variable
596  // 2. the threadID
597  // the thread ID value will be used to calculate the memory-mapped
598  // address
599  // which will be saved in the pointer to the thread variable
600  //
601  // the actual body of the wrapper function will be generated later
602 
603  // create the wrapper function definition (which will call the Pthread
604  // accelerator)
605  std::string wrapperName = getWrapperName(F, pthreadcall);
606  std::vector<Type *> Params;
607  if (F->arg_size()) {
608  Params.push_back(F->arg_begin()->getType());
609  }
610  // thread ptr type
611  Params.push_back(
612  PointerType::get(IntegerType::get(getGlobalContext(), 32), 0));
613  // threadID value type
614  Params.push_back(IntegerType::get(getGlobalContext(), 32));
615 
616  // pthread_create wrappers are of void type
617  Constant *c = M.getOrInsertFunction(
618  wrapperName,
619  FunctionType::get(Type::getVoidTy(M.getContext()), Params, false));
620  Function *wrapperF = cast<Function>(c);
621  wrapperF->setCallingConv(CallingConv::C);
622 
623  Function::arg_iterator argI = --wrapperF->arg_end();
624  // set the name last argument of the wrapper function
625  argI->setName("threadIDValue");
626  argI->getPrevNode()->setName("threadID");
627 
628  // get the arguments to pass into the pthread function
629  std::vector<Value *> Args;
630  // add the original argument if there is any
631  if (F->arg_size()) {
632  Args.push_back(CI->getArgOperand(0));
633  wrapperF->arg_begin()->setName(F->arg_begin()->getName());
634  }
635 
636  // the next is the thread pointer
637  // followed by the threadID value
638  Instruction *threadID = CI->getPrevNode();
639  Value *threadPtrArg, *threadIDValueArg;
640  if (StoreInst *st = dyn_cast<StoreInst>(threadID)) {
641  if (getMetadataInt(threadID, "legup_pthread_functionthreadID")) {
642  // get the value operand of the store instruction
643  threadPtrArg = st->getPointerOperand();
644  threadIDValueArg = st->getValueOperand();
645  } else {
646  assert(0 && "The Pthread function threadID cannot be found\n");
647  }
648  } else {
649  assert(0 && "The store instruction after the call to pthread function "
650  "cannot be found\n");
651  }
652  assert(threadPtrArg);
653  assert(threadIDValueArg);
654 
655  Args.push_back(threadPtrArg);
656  // need the bottom 16 bits, which is the actual threadID value
657  // top 16 bits may be the functionID if there are multiple
658  // pthread functions
659 
660  // AND with 0x0000FFFF
661  // to truncate off the top 16 bits
662  BinaryOperator *tr = BinaryOperator::Create(
663  Instruction::And, threadIDValueArg,
664  ConstantInt::get(IntegerType::get(getGlobalContext(), 32), 0x0000FFFF,
665  false),
666  "", CI);
667  Args.push_back(tr);
668 
669  // create call the pthread wrapper
670  CallInst *newCI = CallInst::Create(wrapperF, Args, "", CI);
671 
672  return newCI;
673 }
