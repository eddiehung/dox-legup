304  // these are the the math constraints for the instructions

308  void addMulticycleConstraints() {
309 
310  int col[2];
311  REAL val[2];
312 
313  for (std::map<InstructionNode *, unsigned>::iterator
314  i = startVariableIndex.begin(),
315  e = startVariableIndex.end();
316  i != e; i++) {
317 
318  InstructionNode *iNode = i->first;
319  // Instruction *I = iNode->getInst();
320  unsigned startIndex = i->second;
321 
322  unsigned endIndex = endVariableIndex[iNode];
323 
324  if (startIndex == endIndex)
325  continue; // not a multicycle instruction
326 
327  // File() << "Multi-cycle operation startIndex: " << startIndex <<
328  // " endIndex: " << endIndex << " I: " << *I << "\n";
329 
330  // add constraints so that the variable corresponding to each
331  // cycle of a multiple cycle instruction gets assigned to
332  // contiguous states.
333  for (unsigned j = startIndex + 1; j <= endIndex; j++) {
334  col[0] = 1 + j; // variable indicies
335  col[1] = 1 + (j - 1);
336  val[0] = 1.0; // variable coefficients
337  val[1] = -1.0;
338 
339  if (SDCdebug) {
340  File() << "Adding constraint: s(" << col[0] - 1 << ") == s("
341  << col[1] - 1 << ") + 1 cycle\n";
342  }
343 
344  // there must be EXACTLY 1 cycle delay between variable j and
345  // j-1
346  sdcSolver.addConstraintIncremental(sdcSolver.lp, 2, val, col,
347  EQ, 1.0);
348  }
349  }

552  void addTimingConstraints(InstructionNode *Root, InstructionNode *Curr,
553  float PartialPathDelay,
554  std::vector<Instruction *> path) {
555  int col[2];
556  REAL val[2];
557 
558  // don't constraint multi-cycle operations
559  // dependency has more than 1 cycle latency, so this dependency will
560  // already be in another cycle.
561  if (Scheduler::getNumInstructionCycles(Root->getInst()) > 0)
562  return;
563  if (Scheduler::getNumInstructionCycles(Curr->getInst()) > 0)
564  return;
565 
566  // Walk through the dependencies
567  for (InstructionNode::iterator i = Curr->dep_begin(),
568  e = Curr->dep_end();
569  i != e; ++i) {
570  // dependency from depNode -> Curr
571  InstructionNode *depNode = *i;
572 
573  if (Scheduler::getNumInstructionCycles(depNode->getInst()) > 0)
574  continue;
575 
576  float delay = PartialPathDelay + depNode->getDelay();
577  unsigned cycleConstraint =
578  ceil((float)delay / (float)clockPeriodConstraint);
579 
580  if (cycleConstraint > 0)
581  cycleConstraint--;
582 
583  path.push_back(depNode->getInst());
584  if (cycleConstraint > 0) {
585  Instruction *I1 = depNode->getInst();
586  Instruction *I2 = Root->getInst();
587 
588  if (SDCdebug) {
589  File() << "Found path:\n";
590  for (std::vector<Instruction *>::reverse_iterator
591  j = path.rbegin(),
592  je = path.rend();
593  j != je; ++j) {
594  Instruction *p = *j;
595  File() << "\tdelay: "
596  << dag->getInstructionNode(p)->getDelay() << " "
597  << *p << "\n";
598  }
599  File() << "Path delay:" << delay << "\n";
600  }
601 
602  // don't add any timing constraints for nodes on recurrences,
603  // we want to prioritize getting the best II over meeting
604  // timing constraints...
605  if (moduloScheduler.onCriticalPath(I1) &&
606  moduloScheduler.onCriticalPath(I2)) {
607  File()
608  << "Skipping timing constraint due to loop recurrence "
609  << "on this path\n";
610  continue;
611  }
612 
613  if (SDCdebug) {
614  File() << "Adding cycle constraint: I2 >= I1 + "
615  << cycleConstraint
616  << " cycle(s). Detected delay: " << delay
617  << " (period: " << clockPeriodConstraint
618  << ") between I1 -> I2\n"
619  << "\tI1: " << *I1 << "\n"
620  << "\tI2: " << *I2 << "\n";
621  }
622 
623  // if cycleConstraint == 0, we don't need to add the constraint.
624  // the reason is that such constraints are ALREADY present in
625  // the LP
626  // formulation, as they are depedency constraints --
627  // constraints
628  // that express that an operation must happen AFTER the
629  // operations
630  // producuing results that it depends on
631  col[0] = 1 + startVariableIndex[Root];
632  val[0] = 1.0;
633  col[1] = 1 + endVariableIndex[depNode];
634  val[1] = -1.0;
635  sdcSolver.addConstraintIncremental(sdcSolver.lp, 2, val, col,
636  GE, cycleConstraint);
637 
638  } else {
639  addTimingConstraints(
640  Root, depNode, delay,
641  path); // recursive call to discover other instructions
642  }
643  }
644  }
