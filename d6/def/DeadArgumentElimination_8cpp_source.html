12 // only passed into function calls as dead arguments of other functions. This

191 /// DeleteDeadVarargs - If this is an function that takes a ... list, and if
192 /// llvm.vastart is never called, the varargs list is dead for the function.
193 bool DAE::DeleteDeadVarargs(Function &Fn) {
194  assert(Fn.getFunctionType()->isVarArg() && "Function isn't varargs!");
195  if (Fn.isDeclaration() || !Fn.hasLocalLinkage()) return false;
196 
197  // Ensure that the function is only directly called.
198  if (Fn.hasAddressTaken())
199  return false;
200 
201  // Okay, we know we can transform this function if safe. Scan its body
202  // looking for calls to llvm.vastart.
203  for (Function::iterator BB = Fn.begin(), E = Fn.end(); BB != E; ++BB) {
204  for (BasicBlock::iterator I = BB->begin(), E = BB->end(); I != E; ++I) {
205  if (IntrinsicInst *II = dyn_cast<IntrinsicInst>(I)) {
206  if (II->getIntrinsicID() == Intrinsic::vastart)
207  return false;
208  }
209  }
210  }
211 
212  // If we get here, there are no calls to llvm.vastart in the function body,
213  // remove the "..." and adjust all the calls.
214 
215  // Start by computing a new prototype for the function, which is the same as
216  // the old function, but doesn't have isVarArg set.
217  FunctionType *FTy = Fn.getFunctionType();
218 
219  std::vector<Type*> Params(FTy->param_begin(), FTy->param_end());
220  FunctionType *NFTy = FunctionType::get(FTy->getReturnType(),
221  Params, false);
222  unsigned NumArgs = Params.size();
223 
224  // Create the new function body and insert it into the module...
225  Function *NF = Function::Create(NFTy, Fn.getLinkage());
226  NF->copyAttributesFrom(&Fn);
227  Fn.getParent()->getFunctionList().insert(&Fn, NF);
228  NF->takeName(&Fn);
229 
230  // Loop over all of the callers of the function, transforming the call sites
231  // to pass in a smaller number of arguments into the new function.
232  //
233  std::vector<Value*> Args;
234  for (Value::user_iterator I = Fn.user_begin(), E = Fn.user_end(); I != E; ) {
235  CallSite CS(*I++);
236  if (!CS)
237  continue;
238  Instruction *Call = CS.getInstruction();
239 
240  // Pass all the same arguments.
241  Args.assign(CS.arg_begin(), CS.arg_begin() + NumArgs);
242 
243  // Drop any attributes that were on the vararg arguments.
244  AttributeSet PAL = CS.getAttributes();
245  if (!PAL.isEmpty() && PAL.getSlotIndex(PAL.getNumSlots() - 1) > NumArgs) {
246  SmallVector<AttributeSet, 8> AttributesVec;
247  for (unsigned i = 0; PAL.getSlotIndex(i) <= NumArgs; ++i)
248  AttributesVec.push_back(PAL.getSlotAttributes(i));
249  if (PAL.hasAttributes(AttributeSet::FunctionIndex))
250  AttributesVec.push_back(AttributeSet::get(Fn.getContext(),
251  PAL.getFnAttributes()));
252  PAL = AttributeSet::get(Fn.getContext(), AttributesVec);
253  }
254 
255  Instruction *New;
256  if (InvokeInst *II = dyn_cast<InvokeInst>(Call)) {
257  New = InvokeInst::Create(NF, II->getNormalDest(), II->getUnwindDest(),
258  Args, "", Call);
259  cast<InvokeInst>(New)->setCallingConv(CS.getCallingConv());
260  cast<InvokeInst>(New)->setAttributes(PAL);
261  } else {
262  New = CallInst::Create(NF, Args, "", Call);
263  cast<CallInst>(New)->setCallingConv(CS.getCallingConv());
264  cast<CallInst>(New)->setAttributes(PAL);
265  if (cast<CallInst>(Call)->isTailCall())
266  cast<CallInst>(New)->setTailCall();
267  }
268  New->setDebugLoc(Call->getDebugLoc());
269 
270  Args.clear();
271 
272  if (!Call->use_empty())
273  Call->replaceAllUsesWith(New);
274 
275  New->takeName(Call);
276 
277  SmallVector<std::pair<unsigned, MDNode *>, 4> MDs;
278  Call->getAllMetadata(MDs);
279  for (SmallVectorImpl<std::pair<unsigned, MDNode *> >::iterator
280  MI = MDs.begin(), ME = MDs.end(); MI != ME; ++MI) {
281  New->setMetadata(MI->first, MI->second);
282  }
283 
284  // Finally, remove the old call from the program, reducing the use-count of
285  // F.
286  Call->eraseFromParent();
287  }
288 
289  // Since we have now created the new function, splice the body of the old
290  // function right into the new function, leaving the old rotting hulk of the
291  // function empty.
292  NF->getBasicBlockList().splice(NF->begin(), Fn.getBasicBlockList());
293 
294  // Loop over the argument list, transferring uses of the old arguments over to
295  // the new arguments, also transferring over the names as well. While we're at
296  // it, remove the dead arguments from the DeadArguments list.
297  //
298  for (Function::arg_iterator I = Fn.arg_begin(), E = Fn.arg_end(),
299  I2 = NF->arg_begin(); I != E; ++I, ++I2) {
300  // Move the name and users over to the new version.
301  I->replaceAllUsesWith(I2);
302  I2->takeName(I);
303  }
304 
305  // Patch the pointer to LLVM function in debug info descriptor.
306  auto DI = FunctionDIs.find(&Fn);
307  if (DI != FunctionDIs.end())
308  DI->second.replaceFunction(NF);
309 
310  // Fix up any BlockAddresses that refer to the function.
311  Fn.replaceAllUsesWith(ConstantExpr::getBitCast(NF, Fn.getType()));
312  // Delete the bitcast that we just created, so that NF does not
313  // appear to be address-taken.
314  NF->removeDeadConstantUsers();
315  // Finally, nuke the old function.
316  Fn.eraseFromParent();
317  return true;
318 }

320 /// RemoveDeadArgumentsFromCallers - Checks if the given function has any
321 /// arguments that are unused, and changes the caller parameters to be undefined
322 /// instead.
323 bool DAE::RemoveDeadArgumentsFromCallers(Function &Fn)
324 {
325  if (Fn.isDeclaration() || Fn.mayBeOverridden())
326  return false;
327 
328  // Functions with local linkage should already have been handled, except the
329  // fragile (variadic) ones which we can improve here.
330  if (Fn.hasLocalLinkage() && !Fn.getFunctionType()->isVarArg())
331  return false;
332 
333  // If a function seen at compile time is not necessarily the one linked to
334  // the binary being built, it is illegal to change the actual arguments
335  // passed to it. These functions can be captured by isWeakForLinker().
336  // *NOTE* that mayBeOverridden() is insufficient for this purpose as it
337  // doesn't include linkage types like AvailableExternallyLinkage and
338  // LinkOnceODRLinkage. Take link_odr* as an example, it indicates a set of
339  // *EQUIVALENT* globals that can be merged at link-time. However, the
340  // semantic of *EQUIVALENT*-functions includes parameters. Changing
341  // parameters breaks this assumption.
342  //
343  if (Fn.isWeakForLinker())
344  return false;
345 
346  if (Fn.use_empty())
347  return false;
348 
349  SmallVector<unsigned, 8> UnusedArgs;
350  for (Function::arg_iterator I = Fn.arg_begin(), E = Fn.arg_end();
351  I != E; ++I) {
352  Argument *Arg = I;
353 
354  if (Arg->use_empty() && !Arg->hasByValOrInAllocaAttr())
355  UnusedArgs.push_back(Arg->getArgNo());
356  }
357 
358  if (UnusedArgs.empty())
359  return false;
360 
361  bool Changed = false;
362 
363  for (Use &U : Fn.uses()) {
364  CallSite CS(U.getUser());
365  if (!CS || !CS.isCallee(&U))
366  continue;
367 
368  // Now go through all unused args and replace them with "undef".
369  for (unsigned I = 0, E = UnusedArgs.size(); I != E; ++I) {
370  unsigned ArgNo = UnusedArgs[I];
371 
372  Value *Arg = CS.getArgument(ArgNo);
373  CS.setArgument(ArgNo, UndefValue::get(Arg->getType()));
374  ++NumArgumentsReplacedWithUndef;
375  Changed = true;
376  }
377  }
378 
379  return Changed;
380 }

409 /// SurveyUse - This looks at a single use of an argument or return value
410 /// and determines if it should be alive or not. Adds this use to MaybeLiveUses
411 /// if it causes the used value to become MaybeLive.
412 ///
413 /// RetValNum is the return value number to use when this use is used in a
414 /// return instruction. This is used in the recursion, you should always leave
415 /// it at 0.
416 DAE::Liveness DAE::SurveyUse(const Use *U,
417  UseVector &MaybeLiveUses, unsigned RetValNum) {
418  const User *V = U->getUser();
419  if (const ReturnInst *RI = dyn_cast<ReturnInst>(V)) {
420  // The value is returned from a function. It's only live when the
421  // function's return value is live. We use RetValNum here, for the case
422  // that U is really a use of an insertvalue instruction that uses the
423  // original Use.
424  RetOrArg Use = CreateRet(RI->getParent()->getParent(), RetValNum);
425  // We might be live, depending on the liveness of Use.
426  return MarkIfNotLive(Use, MaybeLiveUses);
427  }
428  if (const InsertValueInst *IV = dyn_cast<InsertValueInst>(V)) {
429  if (U->getOperandNo() != InsertValueInst::getAggregateOperandIndex()
430  && IV->hasIndices())
431  // The use we are examining is inserted into an aggregate. Our liveness
432  // depends on all uses of that aggregate, but if it is used as a return
433  // value, only index at which we were inserted counts.
434  RetValNum = *IV->idx_begin();
435 
436  // Note that if we are used as the aggregate operand to the insertvalue,
437  // we don't change RetValNum, but do survey all our uses.
438 
439  Liveness Result = MaybeLive;
440  for (const Use &UU : IV->uses()) {
441  Result = SurveyUse(&UU, MaybeLiveUses, RetValNum);
442  if (Result == Live)
443  break;
444  }
445  return Result;
446  }
447 
448  if (ImmutableCallSite CS = V) {
449  const Function *F = CS.getCalledFunction();
450  if (F) {
451  // Used in a direct call.
452 
453  // Find the argument number. We know for sure that this use is an
454  // argument, since if it was the function argument this would be an
455  // indirect call and the we know can't be looking at a value of the
456  // label type (for the invoke instruction).
457  unsigned ArgNo = CS.getArgumentNo(U);
458 
459  if (ArgNo >= F->getFunctionType()->getNumParams())
460  // The value is passed in through a vararg! Must be live.
461  return Live;
462 
463  assert(CS.getArgument(ArgNo)
464  == CS->getOperand(U->getOperandNo())
465  && "Argument is not where we expected it");
466 
467  // Value passed to a normal call. It's only live when the corresponding
468  // argument to the called function turns out live.
469  RetOrArg Use = CreateArg(F, ArgNo);
470  return MarkIfNotLive(Use, MaybeLiveUses);
471  }
472  }
473  // Used in any other way? Value must be live.
474  return Live;
475 }

477 /// SurveyUses - This looks at all the uses of the given value
478 /// Returns the Liveness deduced from the uses of this value.
479 ///
480 /// Adds all uses that cause the result to be MaybeLive to MaybeLiveRetUses. If
481 /// the result is Live, MaybeLiveUses might be modified but its content should
482 /// be ignored (since it might not be complete).
483 DAE::Liveness DAE::SurveyUses(const Value *V, UseVector &MaybeLiveUses) {
484  // Assume it's dead (which will only hold if there are no uses at all..).
485  Liveness Result = MaybeLive;
486  // Check each use.
487  for (const Use &U : V->uses()) {
488  Result = SurveyUse(&U, MaybeLiveUses);
489  if (Result == Live)
490  break;
491  }
492  return Result;
493 }

495 // SurveyFunction - This performs the initial survey of the specified function,
496 // checking out whether or not it uses any of its incoming arguments or whether
497 // any callers use the return value. This fills in the LiveValues set and Uses
498 // map.
499 //
500 // We consider arguments of non-internal functions to be intrinsically alive as
501 // well as arguments to functions which have their "address taken".
502 //
503 void DAE::SurveyFunction(const Function &F) {
504  // Functions with inalloca parameters are expecting args in a particular
505  // register and memory layout.
506  if (F.getAttributes().hasAttrSomewhere(Attribute::InAlloca)) {
507  MarkLive(F);
508  return;
509  }
510 
511  unsigned RetCount = NumRetVals(&F);
512  // Assume all return values are dead
513  typedef SmallVector<Liveness, 5> RetVals;
514  RetVals RetValLiveness(RetCount, MaybeLive);
515 
516  typedef SmallVector<UseVector, 5> RetUses;
517  // These vectors map each return value to the uses that make it MaybeLive, so
518  // we can add those to the Uses map if the return value really turns out to be
519  // MaybeLive. Initialized to a list of RetCount empty lists.
520  RetUses MaybeLiveRetUses(RetCount);
521 
522  for (Function::const_iterator BB = F.begin(), E = F.end(); BB != E; ++BB)
523  if (const ReturnInst *RI = dyn_cast<ReturnInst>(BB->getTerminator()))
524  if (RI->getNumOperands() != 0 && RI->getOperand(0)->getType()
525  != F.getFunctionType()->getReturnType()) {
526  // We don't support old style multiple return values.
527  MarkLive(F);
528  return;
529  }
530 
531  if (!F.hasLocalLinkage() && (!ShouldHackArguments() || F.isIntrinsic())) {
532  MarkLive(F);
533  return;
534  }
535 
536  DEBUG(dbgs() << "DAE - Inspecting callers for fn: " << F.getName() << "\n");
537  // Keep track of the number of live retvals, so we can skip checks once all
538  // of them turn out to be live.
539  unsigned NumLiveRetVals = 0;
540  Type *STy = dyn_cast<StructType>(F.getReturnType());
541  // Loop all uses of the function.
542  for (const Use &U : F.uses()) {
543  // If the function is PASSED IN as an argument, its address has been
544  // taken.
545  ImmutableCallSite CS(U.getUser());
546  if (!CS || !CS.isCallee(&U)) {
547  MarkLive(F);
548  return;
549  }
550 
551  // If this use is anything other than a call site, the function is alive.
552  const Instruction *TheCall = CS.getInstruction();
553  if (!TheCall) { // Not a direct call site?
554  MarkLive(F);
555  return;
556  }
557 
558  // If we end up here, we are looking at a direct call to our function.
559 
560  // Now, check how our return value(s) is/are used in this caller. Don't
561  // bother checking return values if all of them are live already.
562  if (NumLiveRetVals != RetCount) {
563  if (STy) {
564  // Check all uses of the return value.
565  for (const User *U : TheCall->users()) {
566  const ExtractValueInst *Ext = dyn_cast<ExtractValueInst>(U);
567  if (Ext && Ext->hasIndices()) {
568  // This use uses a part of our return value, survey the uses of
569  // that part and store the results for this index only.
570  unsigned Idx = *Ext->idx_begin();
571  if (RetValLiveness[Idx] != Live) {
572  RetValLiveness[Idx] = SurveyUses(Ext, MaybeLiveRetUses[Idx]);
573  if (RetValLiveness[Idx] == Live)
574  NumLiveRetVals++;
575  }
576  } else {
577  // Used by something else than extractvalue. Mark all return
578  // values as live.
579  for (unsigned i = 0; i != RetCount; ++i )
580  RetValLiveness[i] = Live;
581  NumLiveRetVals = RetCount;
582  break;
583  }
584  }
585  } else {
586  // Single return value
587  RetValLiveness[0] = SurveyUses(TheCall, MaybeLiveRetUses[0]);
588  if (RetValLiveness[0] == Live)
589  NumLiveRetVals = RetCount;
590  }
591  }
592  }
593 
594  // Now we've inspected all callers, record the liveness of our return values.
595  for (unsigned i = 0; i != RetCount; ++i)
596  MarkValue(CreateRet(&F, i), RetValLiveness[i], MaybeLiveRetUses[i]);
597 
598  DEBUG(dbgs() << "DAE - Inspecting args for fn: " << F.getName() << "\n");
599 
600  // Now, check all of our arguments.
601  unsigned i = 0;
602  UseVector MaybeLiveArgUses;
603  for (Function::const_arg_iterator AI = F.arg_begin(),

608  // them, making them potentially very sensitive to ABI changes resulting
609  // from removing arguments entirely, so don't. For example AArch64 handles
610  // register and stack HFAs very differently, and this is reflected in the
611  // IR which has already been generated.
612  Result = Live;
613  } else {

615  // MaybeLive in MaybeLiveArgUses).

627 /// MaybeLive, it also takes all uses in MaybeLiveUses and records them in Uses,
628 /// such that RA will be marked live if any use in MaybeLiveUses gets marked

637  // marked live whenever one of the uses becomes live.

639  UE = MaybeLiveUses.end(); UI != UE; ++UI)

641  break;

676 /// PropagateLiveness - Given that RA is a live value, propagate it's liveness
677 /// to any other values it uses (according to Uses).
678 void DAE::PropagateLiveness(const RetOrArg &RA) {
679  // We don't use upper_bound (or equal_range) here, because our recursive call
680  // to ourselves is likely to cause the upper_bound (which is the first value
681  // not belonging to RA) to become erased and the iterator invalidated.
682  UseMap::iterator Begin = Uses.lower_bound(RA);
683  UseMap::iterator E = Uses.end();
684  UseMap::iterator I;
685  for (I = Begin; I != E && I->first == RA; ++I)
686  MarkLive(I->second);
687 
688  // Erase RA from the Uses map (from the lower bound to wherever we ended up
689  // after the loop).
690  Uses.erase(Begin, I);
691 }

693 // RemoveDeadStuffFromFunction - Remove any arguments and return values from F
694 // that are not in LiveValues. Transform the function and all of the callees of
695 // the function to not have these arguments and return values.
696 //
697 bool DAE::RemoveDeadStuffFromFunction(Function *F) {
698  // Don't modify fully live functions
699  if (LiveFunctions.count(F))
700  return false;
701 
702  // Start by computing a new prototype for the function, which is the same as
703  // the old function, but has fewer arguments and a different return type.
704  FunctionType *FTy = F->getFunctionType();
705  std::vector<Type*> Params;
706 
707  // Keep track of if we have a live 'returned' argument
708  bool HasLiveReturnedArg = false;
709 
710  // Set up to build a new list of parameter attributes.
711  SmallVector<AttributeSet, 8> AttributesVec;
712  const AttributeSet &PAL = F->getAttributes();
713 
714  // Remember which arguments are still alive.
715  SmallVector<bool, 10> ArgAlive(FTy->getNumParams(), false);
716  // Construct the new parameter list from non-dead arguments. Also construct
717  // a new set of parameter attributes to correspond. Skip the first parameter
718  // attribute, since that belongs to the return value.
719  unsigned i = 0;
720  for (Function::arg_iterator I = F->arg_begin(), E = F->arg_end();
721  I != E; ++I, ++i) {
722  RetOrArg Arg = CreateArg(F, i);
723  if (LiveValues.erase(Arg)) {
724  Params.push_back(I->getType());
725  ArgAlive[i] = true;
726 
727  // Get the original parameter attributes (skipping the first one, that is
728  // for the return value.
729  if (PAL.hasAttributes(i + 1)) {
730  AttrBuilder B(PAL, i + 1);
731  if (B.contains(Attribute::Returned))
732  HasLiveReturnedArg = true;
733  AttributesVec.
734  push_back(AttributeSet::get(F->getContext(), Params.size(), B));
735  }
736  } else {
737  ++NumArgumentsEliminated;
738  DEBUG(dbgs() << "DAE - Removing argument " << i << " (" << I->getName()
739  << ") from " << F->getName() << "\n");
740  }
741  }
742 
743  // Find out the new return value.
744  Type *RetTy = FTy->getReturnType();
745  Type *NRetTy = nullptr;
746  unsigned RetCount = NumRetVals(F);
747 
748  // -1 means unused, other numbers are the new index
749  SmallVector<int, 5> NewRetIdxs(RetCount, -1);
750  std::vector<Type*> RetTypes;
751 
752  // If there is a function with a live 'returned' argument but a dead return
753  // value, then there are two possible actions:
754  // 1) Eliminate the return value and take off the 'returned' attribute on the
755  // argument.
756  // 2) Retain the 'returned' attribute and treat the return value (but not the
757  // entire function) as live so that it is not eliminated.
758  //
759  // It's not clear in the general case which option is more profitable because,
760  // even in the absence of explicit uses of the return value, code generation
761  // is free to use the 'returned' attribute to do things like eliding
762  // save/restores of registers across calls. Whether or not this happens is
763  // target and ABI-specific as well as depending on the amount of register
764  // pressure, so there's no good way for an IR-level pass to figure this out.
765  //
766  // Fortunately, the only places where 'returned' is currently generated by
767  // the FE are places where 'returned' is basically free and almost always a
768  // performance win, so the second option can just be used always for now.
769  //
770  // This should be revisited if 'returned' is ever applied more liberally.
771  if (RetTy->isVoidTy() || HasLiveReturnedArg) {
772  NRetTy = RetTy;
773  } else {
774  StructType *STy = dyn_cast<StructType>(RetTy);
775  if (STy)
776  // Look at each of the original return values individually.
777  for (unsigned i = 0; i != RetCount; ++i) {
778  RetOrArg Ret = CreateRet(F, i);
779  if (LiveValues.erase(Ret)) {
780  RetTypes.push_back(STy->getElementType(i));
781  NewRetIdxs[i] = RetTypes.size() - 1;
782  } else {
783  ++NumRetValsEliminated;
784  DEBUG(dbgs() << "DAE - Removing return value " << i << " from "
785  << F->getName() << "\n");
786  }
787  }
788  else
789  // We used to return a single value.
790  if (LiveValues.erase(CreateRet(F, 0))) {
791  RetTypes.push_back(RetTy);
792  NewRetIdxs[0] = 0;
793  } else {
794  DEBUG(dbgs() << "DAE - Removing return value from " << F->getName()
795  << "\n");
796  ++NumRetValsEliminated;
797  }
798  if (RetTypes.size() > 1)
799  // More than one return type? Return a struct with them. Also, if we used
800  // to return a struct and didn't change the number of return values,
801  // return a struct again. This prevents changing {something} into
802  // something and {} into void.
803  // Make the new struct packed if we used to return a packed struct
804  // already.
805  NRetTy = StructType::get(STy->getContext(), RetTypes, STy->isPacked());
806  else if (RetTypes.size() == 1)
807  // One return type? Just a simple value then, but only if we didn't use to
808  // return a struct with that simple value before.
809  NRetTy = RetTypes.front();
810  else if (RetTypes.size() == 0)
811  // No return types? Make it void, but only if we didn't use to return {}.
812  NRetTy = Type::getVoidTy(F->getContext());
813  }
814 
815  assert(NRetTy && "No new return type found?");
816 
817  // The existing function return attributes.
818  AttributeSet RAttrs = PAL.getRetAttributes();
819 
820  // Remove any incompatible attributes, but only if we removed all return
821  // values. Otherwise, ensure that we don't have any conflicting attributes
822  // here. Currently, this should not be possible, but special handling might be
823  // required when new return value attributes are added.
824  if (NRetTy->isVoidTy())
825  RAttrs =
826  AttributeSet::get(NRetTy->getContext(), AttributeSet::ReturnIndex,
827  AttrBuilder(RAttrs, AttributeSet::ReturnIndex).
828  removeAttributes(AttributeFuncs::
829  typeIncompatible(NRetTy, AttributeSet::ReturnIndex),
830  AttributeSet::ReturnIndex));
831  else
832  assert(!AttrBuilder(RAttrs, AttributeSet::ReturnIndex).
833  hasAttributes(AttributeFuncs::
834  typeIncompatible(NRetTy, AttributeSet::ReturnIndex),
835  AttributeSet::ReturnIndex) &&
836  "Return attributes no longer compatible?");
837 
838  if (RAttrs.hasAttributes(AttributeSet::ReturnIndex))
839  AttributesVec.push_back(AttributeSet::get(NRetTy->getContext(), RAttrs));
840 
841  if (PAL.hasAttributes(AttributeSet::FunctionIndex))
842  AttributesVec.push_back(AttributeSet::get(F->getContext(),
843  PAL.getFnAttributes()));
844 
845  // Reconstruct the AttributesList based on the vector we constructed.
846  AttributeSet NewPAL = AttributeSet::get(F->getContext(), AttributesVec);
847 
848  // Create the new function type based on the recomputed parameters.
849  FunctionType *NFTy = FunctionType::get(NRetTy, Params, FTy->isVarArg());
850 
851  // No change?
852  if (NFTy == FTy)
853  return false;

855  // Create the new function body and insert it into the module...
856  Function *NF = Function::Create(NFTy, F->getLinkage());
857  NF->copyAttributesFrom(F);
858  NF->setAttributes(NewPAL);
859  // Insert the new function before the old function, so we won't be processing
860  // it again.
861  F->getParent()->getFunctionList().insert(F, NF);
862  NF->takeName(F);
863 
864  // Loop over all of the callers of the function, transforming the call sites
865  // to pass in a smaller number of arguments into the new function.
866  //
867  std::vector<Value*> Args;
868  while (!F->use_empty()) {
869  CallSite CS(F->user_back());
870  Instruction *Call = CS.getInstruction();
871 
872  AttributesVec.clear();
873  const AttributeSet &CallPAL = CS.getAttributes();
874 
875  // The call return attributes.
876  AttributeSet RAttrs = CallPAL.getRetAttributes();
877 
878  // Adjust in case the function was changed to return void.
879  RAttrs =
880  AttributeSet::get(NF->getContext(), AttributeSet::ReturnIndex,
881  AttrBuilder(RAttrs, AttributeSet::ReturnIndex).
882  removeAttributes(AttributeFuncs::
883  typeIncompatible(NF->getReturnType(),
884  AttributeSet::ReturnIndex),
885  AttributeSet::ReturnIndex));
886  if (RAttrs.hasAttributes(AttributeSet::ReturnIndex))
887  AttributesVec.push_back(AttributeSet::get(NF->getContext(), RAttrs));
888 
889  // Declare these outside of the loops, so we can reuse them for the second
890  // loop, which loops the varargs.
891  CallSite::arg_iterator I = CS.arg_begin();
892  unsigned i = 0;
893  // Loop over those operands, corresponding to the normal arguments to the
894  // original function, and add those that are still alive.
895  for (unsigned e = FTy->getNumParams(); i != e; ++I, ++i)
896  if (ArgAlive[i]) {
897  Args.push_back(*I);
898  // Get original parameter attributes, but skip return attributes.
899  if (CallPAL.hasAttributes(i + 1)) {
900  AttrBuilder B(CallPAL, i + 1);
901  // If the return type has changed, then get rid of 'returned' on the
902  // call site. The alternative is to make all 'returned' attributes on
903  // call sites keep the return value alive just like 'returned'

905  // and this is not an expected case anyway
906  if (NRetTy != RetTy && B.contains(Attribute::Returned))
907  B.removeAttribute(Attribute::Returned);
908  AttributesVec.
909  push_back(AttributeSet::get(F->getContext(), Args.size(), B));
910  }
911  }
912 
913  // Push any varargs arguments on the list. Don't forget their attributes.
914  for (CallSite::arg_iterator E = CS.arg_end(); I != E; ++I, ++i) {
915  Args.push_back(*I);
916  if (CallPAL.hasAttributes(i + 1)) {
917  AttrBuilder B(CallPAL, i + 1);

919  push_back(AttributeSet::get(F->getContext(), Args.size(), B));
920  }

922 
923  if (CallPAL.hasAttributes(AttributeSet::FunctionIndex))
924  AttributesVec.push_back(AttributeSet::get(Call->getContext(),
925  CallPAL.getFnAttributes()));

927  // Reconstruct the AttributesList based on the vector we constructed.
928  AttributeSet NewCallPAL = AttributeSet::get(F->getContext(), AttributesVec);
929

932  New = InvokeInst::Create(NF, II->getNormalDest(), II->getUnwindDest(),
933  Args, "", Call);
934  cast<InvokeInst>(New)->setCallingConv(CS.getCallingConv());
935  cast<InvokeInst>(New)->setAttributes(NewCallPAL);
936  } else {
937  New = CallInst::Create(NF, Args, "", Call);
938  cast<CallInst>(New)->setCallingConv(CS.getCallingConv());
939  cast<CallInst>(New)->setAttributes(NewCallPAL);
940  if (cast<CallInst>(Call)->isTailCall())

942  }

950  Call->replaceAllUsesWith(New);

954  // Replace by null for now.
955  if (!Call->getType()->isX86_MMXTy())

1026  if (!I->getType()->isX86_MMXTy())

1039  } else {

1042  // extractvalue/insertvalue chains to extract only the values we need
1043  // to return and insert them into our new result.
1044  // This does generate messy code, but we'll let it to instcombine to
1045  // clean that up.

1049  for (unsigned i = 0; i != RetCount; ++i)
1050  if (NewRetIdxs[i] != -1) {
1051  ExtractValueInst *EV = ExtractValueInst::Create(OldRet, i,
1052  "oldret", RI);

1054  // We're still returning a struct, so reinsert the value into
1055  // our new return value at the new index
1056 
1057  RetVal = InsertValueInst::Create(RetVal, EV, NewRetIdxs[i],
1058  "newret", RI);
1059  } else {

1061  // extracted value.
1062  RetVal = EV;
1063  }
1064  }
1065  }
1066  // Replace the return instruction with one returning the new return
1067  // value (possibly 0 if we became void).
1068  ReturnInst::Create(F->getContext(), RetVal, RI);
1069  BB->getInstList().erase(RI);
1070  }

1073  auto DI = FunctionDIs.find(F);
1074  if (DI != FunctionDIs.end())
1075  DI->second.replaceFunction(NF);
1076 
1077  // Now that the old function is dead, delete it.
1078  F->eraseFromParent();
1079 
1080  return true;

1090  // removed. We can do this if they never call va_start. This loop cannot be

1092  // information computed while surveying other functions.

1095  Function &F = *I++;

1102  // determine that dead arguments passed into recursive functions are dead).
1103  //

1106  SurveyFunction(*I);

1109  // turn.

1121
