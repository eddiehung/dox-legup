38  * Assign lattice bits to definitions.

40  * @param instToLatticeBit Mapping from definitions to lattice bits.
41  * @param F LLVM function object.

44 void LiveVariableAnalysis::initInstrToLatticeBit(Function &F) {
45  unsigned instIdx = 0;
46 
47  // Arguments count as definitions as well - first F.arg_size() bits.
48  for (Function::arg_iterator argIt = F.arg_begin(); argIt != F.arg_end(); argIt++) {
49  instToLatticeBit.insert(std::make_pair(&*argIt, instIdx));
50  instIdx++;
51  }
52  // Don't bother filtering out non-definition instructions. Their
53  // corresponding use / in / out bits will be always zero and the
54  // resulting in / out sets will look right.
55  for (inst_iterator instIt = inst_begin(F); instIt != inst_end(F); instIt++) {
56  instToLatticeBit.insert(std::make_pair(&*instIt, instIdx));
57  instIdx++;
58  }
59 }

103 /**
104  * Set corresponding use bit for all blocks using a given definition.
105  *
106  * @param use_begin First use of the given definition.
107  * @param use_end One-past-last use of the given definition.
108  * @param blockToInfo Mapping from LLVM basic blocks to liveness information.
109  * @param defInst Definition for which use bits are set
110  * (NULL if argument, not instruction).
111  * @param defBit Corresponding bit for the definition in the lattice.
112  *
113  */
114 void LiveVariableAnalysis::markUses(Value::user_iterator user_begin, Value::user_iterator user_end,
115  Instruction *defInst, unsigned defBit) {
116  for (Value::user_iterator useIt = user_begin; useIt != user_end; useIt++) {
117  Instruction *useInst = dyn_cast<Instruction>(*useIt);
118  if (!useInst) continue;
119  BasicBlock *useBlock = useInst->getParent();
120 
121  // Avoid setting use bit for uses in the same block as the
122  // definition. This is equivalent with the way use sets are
123  // defined for the liveness analysis when working with SSA
124  // form (thus having a textual-unique, dominant definition
125  // for every use of a value).
126 
127  if (defInst != NULL) {
128  // STEFAN: The assumption that in SSA you never have a use followed
129  // by a def neglects the case where an instruction enters a block
130  // through a phi but is then also defined in that block (see
131  // adpcm).
132  if (defInst->getParent() == useBlock && !isa<PHINode>(useInst)) {
133  continue;
134  }
135  }
136 
137  BasicBlockLivenessInfo *useBlockInfo = blockToInfo[useBlock];
138  useBlockInfo->use->set(defBit);
139  }
140 }

143  * Set use sets for all the blocks.

149 void LiveVariableAnalysis::initBlocksUse(Function &F) {
150  for (inst_iterator defIt = inst_begin(F); defIt != inst_end(F); defIt++) {
151  Instruction *defInst = &*defIt;
152  markUses(defIt->user_begin(), defIt->user_end(), defInst, getLatticeBit(defInst));
153  }

155  markUses(defIt->user_begin(), defIt->user_end(), NULL, getLatticeBit(&*defIt));
156  }

159 // create a mask that is 1 for every instruction except instructions that are
160 // only used in a phi node
161 // ie. %i = phi i32 [ %b, %bb1 ], [ %c, %bb2 ]
162 // %d = add i32 %c, 0
163 // Here we would set the mask to 1 for %i, %c (because of the use by %d), and
164 // %d and set the mask to 0 for %b
165 BitVector LiveVariableAnalysis::getMaskPhiValues(BasicBlock *phiBlock) {
166  BitVector mask = BitVector(instToLatticeBit.size(), true);
167  for (BasicBlock::iterator instIt = phiBlock->begin(); instIt != phiBlock->end(); instIt++) {
168  if (!isa<PHINode>(&*instIt))
169  continue;
170  PHINode *phiNode = dyn_cast<PHINode>(&*instIt);
171  for (unsigned crtIncomingIdx = 0; crtIncomingIdx < phiNode->getNumIncomingValues(); crtIncomingIdx++) {
172  Value *crtValue = phiNode->getIncomingValue(crtIncomingIdx);
173  Instruction *crtInst = dyn_cast<Instruction>(crtValue);
174  if (!crtInst) continue;
175  BasicBlock *defBlock = crtInst->getParent();
176 
177  // check if the incoming value is only used in a phi instr
178  bool usedInNonPhi = false;
179  for (Value::user_iterator useIt = crtInst->user_begin(); useIt !=
180  crtInst->user_end(); useIt++) {
181  Instruction *useInst = dyn_cast<Instruction>(*useIt);
182  if (!useInst) continue;
183  BasicBlock *useBlock = useInst->getParent();
184  // use/def in the same basic block hides upwards use
185  if (useBlock == defBlock) continue;
186  // ignore phi uses
187  if (isa<PHINode>(useInst)) continue;
188  // should be in the same basic block as phi
189  if (useBlock != phiBlock) continue;
190 
191  usedInNonPhi = true;
192  }
193 
194  if (!usedInNonPhi) {
195  mask.reset( getLatticeBit(crtValue) );
196  }
197  }
198  }
199  return mask;
200 }

203 /**
204  * Set flow mask for the current function.
205  *
206  * Liveness analysis is tricky using the SSA form because of the phi nodes.
207  * A phi node appears to use more than one value, but this is actually
208  * flow-sensitive. Basically, a phi node looks - data-flow wise - as a
209  * different instruction (using a single value) from each incoming block.
210  * Therefore, when computing out set for each incoming block, the in sets
211  * for the successors containing phi nodes will be adjusted accordingly by
212  * using this mask.
213  *
214  * @param mask The flow mask between blocks containing phi-nodes and the
215  * incoming blocks.
216  * @param blockToInfo Mapping from LLVM basic blocks to liveness information.
217  * @param instToLatticeBit Mapping from definitions to lattice bits.
218  * @param F LLVM function object.
219  */
220 void LiveVariableAnalysis::initMask(FlowMask &mask, Function &F) {
221 
222  for (Function::iterator blockIt = F.begin(); blockIt != F.end(); blockIt++) {
223  BasicBlock *phiBlock = blockIt;
224 
225  BitVector initialMask = getMaskPhiValues(phiBlock);
226 
227  // now that we have masked out all the phi dependent uses, create a mask
228  // for each crtBlock -> phiBlock control flow
229  // ie. %i = phi i32 [ %b, %bb1 ], [ %c, %bb2 ]
230  // we should change the mask for %b to 1 for %bb1 -> phiBlock
231  // and leave the mask for %b at 0 for %bb2 -> phiBlock
232  for (BasicBlock::iterator instIt = phiBlock->begin(); instIt != phiBlock->end(); instIt++) {
233  if (!isa<PHINode>(&*instIt))
234  continue;
235  PHINode *phiNode = dyn_cast<PHINode>(&*instIt);
236  for (unsigned crtIncomingIdx = 0; crtIncomingIdx < phiNode->getNumIncomingValues(); crtIncomingIdx++) {
237  BasicBlock *crtBlock = phiNode->getIncomingBlock(crtIncomingIdx);
238  Value *crtValue = phiNode->getIncomingValue(crtIncomingIdx);
239  if (!isa<Instruction>(crtValue)) continue;
240  std::pair<BasicBlock *, BasicBlock *> maskKey = std::make_pair(crtBlock, phiBlock);
241  std::map<std::pair<BasicBlock *, BasicBlock *>, BitVector *>::iterator maskIt =
242  mask.find(maskKey);
243  if (maskIt == mask.end()) {
244  mask[maskKey] = new BitVector(initialMask);
245  }
246  mask[maskKey]->set(getLatticeBit(crtValue));
247  }
248  }
249  }

286  // initialize block to info mapping
287  for (BlockInfoMapping::iterator it = blockToInfo.begin(); it != blockToInfo.end(); it++) {
288  delete it->second;
289  }
290  blockToInfo.clear();
291  initBlocksInfo(defcount, F);
292 
293  // initialize def and use sets
294  initEntryArgDef(blockToInfo[F.begin()], F.arg_size());
295  for (BlockInfoMapping::iterator it = blockToInfo.begin(); it != blockToInfo.end(); it++) {
296  initBlockDef(it->second);

299 
300  // initialize flow mask
301  FlowMask mask;

305  // blocks in quasi-topological order, works anyway
306  bool inChanged = true;
307  while (inChanged) {

311  (blockToInfo[B]->out)->reset();
