82 void EdmondMatching::backtrace (int vert, int pathNum, int stop, int parity, int direction){
83  if (vert==stop) return; //pathNum is the number of the path to store
84  else if (parity==0){ //vert and parity determine the label to be read.
85  if (direction==reverse){
86  backtrace(label[vert].even,pathNum,stop,(parity+1)%2,reverse);
87  path[pathNum][endPath[pathNum]++]=vert;
88  } //forward means the vertices called first enter
89  else if (direction==forward){ //the path first, reverse is the opposite.
90  path[pathNum][endPath[pathNum]++]=vert;
91  backtrace(label[vert].even,pathNum,stop,(parity+1)%2,forward);
92  }
93  }
94  /*
95  stop is the stopping condition for the recursion.
96  Recursion is necessary because of the possible dual odd labels.
97  having empty at stop means the recursion will only stop after
98  the whole tree has been climbed. If assigned to a vertex, it'll stop
99  once it's reached.
100  */
101  else if (parity==1 && label[vert].odd[1]==undef){
102  if (direction==reverse){
103  backtrace(label[vert].odd[0],pathNum,stop,(parity+1)%2,reverse);
104  path[pathNum][endPath[pathNum]++]=vert;
105  }
106  else if (direction==forward){
107  path[pathNum][endPath[pathNum]++]=vert;
108  backtrace(label[vert].odd[0],pathNum,stop,(parity+1)%2,forward);
109  }
110  }
111  /*
112  Dual odd labels are interpreted as follows:
113  There exists an odd length alternating path starting from the root to this
114  vertex. To find this path, backtrace from odd[0] to the top of the tree and
115  from odd[1] to the vertex itself. This, put in the right order, will
116  constitute said path.
117  */
118  else if (parity==1 && label[vert].odd[1]!=undef){
119  if (direction==reverse){
120  backtrace(label[vert].odd[0],pathNum,empty,(parity+1)%2,reverse);
121  backtrace(label[vert].odd[1],pathNum,vert,(parity+1)%2,forward);
122  path[pathNum][endPath[pathNum]++]=vert;
123  }
124  else if (direction==forward){

126  backtrace(label[vert].odd[0],pathNum,empty,(parity+1)%2,forward);

220 bool EdmondMatching::augmentMatching (int n){ //The main analyzing function, with the
221  int node,nodeLabel; //goal of finding augmenting paths or
222  initAlg(n); //concluding that the matching is maximum.
223  for (int i=0; i<n; i++) if (!match[i]){
224  label[i].even=empty;
225  enqueue(i,0); //Initialize the queue with the exposed vertices,
226  } //making them the roots in the forest.

230  nodeLabel=queue[queueFront].type;
231  if (nodeLabel==0){
232  for (int i=0; i<n; i++) if (g[node][i]==unmatched){
233  if (blossom[node]==blossom[i]);
234  //Do nothing. Edges inside the same blossom have no meaning.
235  else if (label[i].even!=undef){
236  /*
237  The tree has reached a vertex with a label.
238  The parity of this label indicates that an odd length
239  alternating path has been found. If this path is between
240  roots, we have an augmenting path, else there's an
241  alternating cycle, a blossom.
242  */

244  backtrace(node,0,empty,0,reverse);
245  backtrace(i,1,empty,0,reverse);
246  //Call the backtracing function to find out.
247  if (path[0][0]==path[1][0]) newBlossom(node,i);
248  /*
249  If the same root node is reached, a blossom was found.
250  Start the labelling procedure to create pseudo-contraction.
251  */
252  else {

255  /*
256  If the roots are different, we have an augmenting path.

261  }
