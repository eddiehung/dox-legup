187  limit(40), out(o), printNodeLabel(callback_node_label),

196  void printLabel(formatted_raw_ostream &out, T *I) {
197  out << "[label=\"";
198 
199  std::string stripped;
200  raw_string_ostream stream(stripped);
201 
202  printNodeLabel(stream, I);
203  // need to flush the stream to write to string!
204  stream.flush();
205 
206  // newlines aren't allowed in dot labels
207  replaceAll(stripped, "\n", " ");
208 
209  // limit the size of the instruction string
210  limitString(stripped, limit);
211 
212  out << stripped;
213 
214  out << "\"";
215 
216  if (printNodeLabelExtras) {
217  printNodeLabelExtras(out, I);
218  }
219 
220  out << "]";
221  }

227  void connectDot(formatted_raw_ostream &out, T *driver,
228  T *signal, std::string label) {
229 
230  if (seen.find(signal) == seen.end()) {
231  printNode(out, signal);
232  printLabel(out, signal);
233  out << ";\n";
234  seen.insert(signal);
235  }
236 
237  if (seen.find(driver) == seen.end()) {
238  printNode(out, driver);
239  printLabel(out, driver);
240  out << ";\n";
241  seen.insert(driver);
242  }
243 
244  printNode(out, driver);
245  out << " -> ";
246  printNode(out, signal);
247  if (!label.empty()) out << "[" << label << "]";
248  out << ";\n";
249  }
