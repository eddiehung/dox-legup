76 unsigned ARMELFObjectWriter::GetRelocTypeInner(const MCValue &Target,
77  const MCFixup &Fixup,
78  bool IsPCRel) const {
79  MCSymbolRefExpr::VariantKind Modifier = Target.getAccessVariant();
80 
81  unsigned Type = 0;
82  if (IsPCRel) {
83  switch ((unsigned)Fixup.getKind()) {
84  default: llvm_unreachable("Unimplemented");
85  case FK_Data_4:
86  switch (Modifier) {
87  default: llvm_unreachable("Unsupported Modifier");
88  case MCSymbolRefExpr::VK_None:
89  Type = ELF::R_ARM_REL32;
90  break;
91  case MCSymbolRefExpr::VK_TLSGD:
92  llvm_unreachable("unimplemented");
93  case MCSymbolRefExpr::VK_GOTTPOFF:
94  Type = ELF::R_ARM_TLS_IE32;
95  break;
96  case MCSymbolRefExpr::VK_GOTPCREL:
97  Type = ELF::R_ARM_GOT_PREL;
98  break;
99  }
100  break;
101  case ARM::fixup_arm_blx:
102  case ARM::fixup_arm_uncondbl:
103  switch (Modifier) {
104  case MCSymbolRefExpr::VK_PLT:
105  Type = ELF::R_ARM_PLT32;
106  break;
107  case MCSymbolRefExpr::VK_ARM_TLSCALL:
108  Type = ELF::R_ARM_TLS_CALL;
109  break;
110  default:
111  Type = ELF::R_ARM_CALL;
112  break;
113  }
114  break;
115  case ARM::fixup_arm_condbl:
116  case ARM::fixup_arm_condbranch:
117  case ARM::fixup_arm_uncondbranch:
118  Type = ELF::R_ARM_JUMP24;
119  break;
120  case ARM::fixup_t2_condbranch:
121  case ARM::fixup_t2_uncondbranch:
122  Type = ELF::R_ARM_THM_JUMP24;
123  break;
124  case ARM::fixup_arm_movt_hi16:
125  Type = ELF::R_ARM_MOVT_PREL;
126  break;
127  case ARM::fixup_arm_movw_lo16:
128  Type = ELF::R_ARM_MOVW_PREL_NC;
129  break;
130  case ARM::fixup_t2_movt_hi16:
131  Type = ELF::R_ARM_THM_MOVT_PREL;
132  break;
133  case ARM::fixup_t2_movw_lo16:
134  Type = ELF::R_ARM_THM_MOVW_PREL_NC;
135  break;
136  case ARM::fixup_arm_thumb_bl:
137  case ARM::fixup_arm_thumb_blx:
138  switch (Modifier) {
139  case MCSymbolRefExpr::VK_ARM_TLSCALL:
140  Type = ELF::R_ARM_THM_TLS_CALL;
141  break;
142  default:
143  Type = ELF::R_ARM_THM_CALL;
144  break;
145  }
146  break;
147  }
148  } else {
149  switch ((unsigned)Fixup.getKind()) {
150  default: llvm_unreachable("invalid fixup kind!");
151  case FK_Data_4:
152  switch (Modifier) {
153  default: llvm_unreachable("Unsupported Modifier");
154  case MCSymbolRefExpr::VK_ARM_NONE:
155  Type = ELF::R_ARM_NONE;
156  break;
157  case MCSymbolRefExpr::VK_GOT:
158  Type = ELF::R_ARM_GOT_BREL;
159  break;
160  case MCSymbolRefExpr::VK_TLSGD:
161  Type = ELF::R_ARM_TLS_GD32;
162  break;
163  case MCSymbolRefExpr::VK_TPOFF:
164  Type = ELF::R_ARM_TLS_LE32;
165  break;
166  case MCSymbolRefExpr::VK_GOTTPOFF:
167  Type = ELF::R_ARM_TLS_IE32;
168  break;
169  case MCSymbolRefExpr::VK_None:
170  Type = ELF::R_ARM_ABS32;
171  break;
172  case MCSymbolRefExpr::VK_GOTOFF:
173  Type = ELF::R_ARM_GOTOFF32;
174  break;
175  case MCSymbolRefExpr::VK_GOTPCREL:
176  Type = ELF::R_ARM_GOT_PREL;
177  break;
178  case MCSymbolRefExpr::VK_ARM_TARGET1:
179  Type = ELF::R_ARM_TARGET1;
180  break;
181  case MCSymbolRefExpr::VK_ARM_TARGET2:
182  Type = ELF::R_ARM_TARGET2;
183  break;
184  case MCSymbolRefExpr::VK_ARM_PREL31:
185  Type = ELF::R_ARM_PREL31;
186  break;
187  case MCSymbolRefExpr::VK_ARM_TLSLDO:
188  Type = ELF::R_ARM_TLS_LDO32;
189  break;
190  case MCSymbolRefExpr::VK_ARM_TLSCALL:
191  Type = ELF::R_ARM_TLS_CALL;
192  break;
193  case MCSymbolRefExpr::VK_ARM_TLSDESC:
194  Type = ELF::R_ARM_TLS_GOTDESC;
195  break;
196  case MCSymbolRefExpr::VK_ARM_TLSDESCSEQ:
197  Type = ELF::R_ARM_TLS_DESCSEQ;
198  break;
199  }
200  break;
201  case ARM::fixup_arm_ldst_pcrel_12:
202  case ARM::fixup_arm_pcrel_10:
203  case ARM::fixup_arm_adr_pcrel_12:
204  case ARM::fixup_arm_thumb_bl:
205  case ARM::fixup_arm_thumb_cb:
206  case ARM::fixup_arm_thumb_cp:
207  case ARM::fixup_arm_thumb_br:
208  llvm_unreachable("Unimplemented");
209  case ARM::fixup_arm_condbranch:
210  case ARM::fixup_arm_uncondbranch:
211  Type = ELF::R_ARM_JUMP24;
212  break;
213  case ARM::fixup_arm_movt_hi16:
214  Type = ELF::R_ARM_MOVT_ABS;
215  break;
216  case ARM::fixup_arm_movw_lo16:
217  Type = ELF::R_ARM_MOVW_ABS_NC;
218  break;
219  case ARM::fixup_t2_movt_hi16:
220  Type = ELF::R_ARM_THM_MOVT_ABS;
221  break;
222  case ARM::fixup_t2_movw_lo16:
223  Type = ELF::R_ARM_THM_MOVW_ABS_NC;
224  break;
225  }
226  }
227 
228  return Type;
229 }
